\begin{question}
证明Laplace变换的性质，其中 \(f(t) \to F(p)\)，对于前三个定理，写出Fourier变换的类似结论
\begin{enumerate}
  \item 延迟定理 \(f(t-t_0) \eta(t-t_0) \to \ee^{-p t_0} F(p)\)；
  \item 平移定理 \(\ee^{p_0 t} f(t) \eta(t) \to F(p-p_0)\)；
  \item 相似定理 \(f(a t) \eta(t) \to \frac1a F(p/a),\quad a>0\)；
  \item \(\displaystyle\int_t^\infty \frac{f(\tau)}{\tau}\dd{}\tau\, \eta(t) \to \frac1p \int_0^p F(q)\dd{q}\)。
\end{enumerate}

\begin{solution}
\begin{enumerate}
  \item 作代换 \(u=t-t_0\)，得到
  \[
  \begin{aligned}
  \mathcal{L}\{f(t-t_0)\eta(t-t_0)\}
  &=\int_{t_0}^{\infty}f(t-t_0)\ee^{-pt}\dd{t}\\
  &=\ee^{-pt_0}\int_0^\infty f(u)\ee^{-pu}\dd{u}
  =\ee^{-pt_0}F(p).
  \end{aligned}
  \]
  相应的 Fourier 变换结论为
  \[
  \mathcal{F}\{f(x-x_0)\}(k)=\ee^{-\ii kx_0}F(k).
  \]

  \item
  \[
  \mathcal{L}\{\ee^{p_0t}f(t)\eta(t)\}
  =\int_0^\infty f(t)\ee^{-(p-p_0)t}\dd{t}
  =F(p-p_0).
  \]
  相应的 Fourier 变换结论为
  \[
  \mathcal{F}\{\ee^{\ii k_0x}f(x)\}(k)=F(k-k_0).
  \]

  \item 当 \(a>0\) 时，作代换 \(u=at\)，得到
  \[
  \mathcal{L}\{f(at)\eta(t)\}
  =\frac1a\int_0^\infty f(u)\ee^{-(p/a)u}\dd{u}
  =\frac1aF\left(\frac pa\right).
  \]
  相应的 Fourier 变换结论为
  \[
  \mathcal{F}\{f(ax)\}(k)=\frac1aF\left(\frac ka\right).
  \]

  \item 对等号左边作 Laplace 变换并交换积分次序：
  \[
  \begin{aligned}
  \mathcal{L}\left\{\int_t^\infty\frac{f(\tau)}{\tau}\dd{\tau}\right\}
  &=\int_0^\infty\frac{f(\tau)}{\tau}
    \left(\int_0^\tau\ee^{-pt}\dd{t}\right)\dd{\tau}\\
  &=\frac1p\int_0^\infty\frac{f(\tau)}{\tau}
    \bigl(1-\ee^{-p\tau}\bigr)\dd{\tau}.
  \end{aligned}
  \]
  另一方面，
  \[
  \frac1p\int_0^pF(q)\dd{q}
  =\frac1p\int_0^\infty\frac{f(t)}{t}
    \bigl(1-\ee^{-pt}\bigr)\dd{t},
  \]
  两式相同。
\end{enumerate}

\end{question}

\begin{question}
求下列函数的Laplace换式：
\begin{enumerate}
  \item \(t^\alpha,\ \operatorname{Re}\alpha >-1\);
  \item \(t-a\lfloor t/a\rfloor,\ a>0\);
  \item \(\dfrac{1-\cos\omega t}{t^2},\ \omega>0\);
  \item \(\displaystyle\int_t^\infty \frac{\sin\tau}{\tau}\dd{}\tau\).
\end{enumerate}
\begin{solution}

\begin{enumerate}
  \item 即计算积分 \(\displaystyle\int_0^\infty \ee^{-p t} t^{\alpha}\dd{t}\)。作换元 \(\tau = p t\)，有 \[ \mathcal{L}\{t^\alpha\}=\int_0^\infty \ee^{-\tau} (\tau/p)^{\alpha}\,\frac{d\tau}{p} = \frac{\Gamma(\alpha+1)}{p^{\alpha+1}} \]

  \item 即计算积分 \(\displaystyle\int_0^\infty (t-a\lfloor t/a\rfloor) \ee^{-p t}\dd{t}\)。分段积分，有 \[ \mathcal{L}\{t-a\lfloor t/a\rfloor\} = \sum_{n=0}^\infty \int_{n a}^{(n+1) a} (t-n a) \ee^{-p t}\dd{t} = \sum_{n=0}^\infty \int_0^a t \ee^{-p (t+n a)}\dd{t} \] \[ = \frac{1}{1-\ee^{-p a}} \int_0^a t \ee^{-p t}\dd{t} = \frac{1}{1-\ee^{-p a}}\frac{1-\ee^{-p a}(p a+1)}{p^2} \]

  \item 利用像函数积分的反演公式 \(\displaystyle\frac{f(t)}{t} \to \int_p^\infty F(q)\dd{q}\)，有 \[ \mathcal{L}\{1-\cos\omega t\}=\frac1p-\frac12\frac1{p+\ii\omega}-\frac12\frac1{p-\ii\omega} \] \[ \mathcal{L}\Bigl\{\frac{1-\cos\omega t}{t}\Bigr\} = \int_p^\infty \Bigl(\frac1q-\frac12\frac1{q+\ii\omega}-\frac12\frac1{q-\ii\omega}\Bigr)\dd{q} = \frac12\ln\frac{p^2+\omega^2}{p^2} \] \[ \mathcal{L}\Bigl\{\frac{1-\cos\omega t}{t^2}\Bigr\} = \int_p^\infty \frac12\ln\frac{q^2+\omega^2}{q^2}\dd{q} = \frac12 p\ln\frac{p^2}{p^2+\omega^2}+\omega\arctan\frac{\omega}{p} \]

  \item 根据 Problem 1 中所证明的，
  \[
  \int_t^\infty \frac{\sin\tau}{\tau}\dd{\tau}
  \longrightarrow
  \frac1p\int_0^p\frac{1}{q^2+1}\dd{q}
  =\frac1p\arctan p.
  \]
\end{enumerate}
\end{solution}
\end{question}

\begin{question}
求下列函数的Fourier换式：（下列各式中 \(a>0\)）
\begin{enumerate}
  \item \(\Pi_a(x)=\begin{cases}0 & |x|>a/2,\\ 1 & |x|<a/2\end{cases}\);
  \item \(\ee^{-x^2/a^2}\);
  \item \(\operatorname{Ш}_a(x)=\displaystyle\sum_{n=-\infty}^\infty \delta(x-n a)\).
\end{enumerate}
\begin{solution}
\begin{enumerate}
  \item
  \[
  \mathcal{F}\{\Pi_a(x)\}
  =\frac1{\sqrt{2\pi}}\int_{-a/2}^{a/2}\ee^{-\ii kx}\dd{x}
  =\sqrt{\frac2\pi}\frac{\sin(ka/2)}{k}.
  \]

  \item 配方后得到
  \[
  \begin{aligned}
  \mathcal{F}\{\ee^{-x^2/a^2}\}
  &=\frac1{\sqrt{2\pi}}\int_{-\infty}^\infty
    \ee^{-x^2/a^2-\ii kx}\dd{x}\\
  &=\frac{a}{\sqrt2}\ee^{-a^2k^2/4}.

  &=\frac1{\sqrt{2\pi}}\sum_{n=-\infty}^\infty\ee^{-\ii kan}\\
  &=\frac{\sqrt{2\pi}}{a}
    \sum_{q=-\infty}^\infty
    \delta\left(k-\frac{2\pi q}{a}\right).
  \end{aligned}
  \]
\end{enumerate}
\end{solution}
\end{question}

\begin{question}

\begin{question}
利用Laplace变换计算下列积分：
\begin{enumerate}
  \item \(\displaystyle\int_0^\infty \frac{\ee^{-a x}-\ee^{-b x}}{x}\cos c x\dd{x},\ a>0,\ b>0,\ c>0\);

  \item \(\displaystyle\int_0^\infty \frac{1-\cos b x}{x^2}\dd{x},\ b>0\);

  \item \(\displaystyle\int_0^\infty \frac{\sin x t}{x(x^2+1)}\dd{x}\).
\end{enumerate}
\begin{solution}
\begin{enumerate}
  \item \[ \ee^{-a x}\cos c x - \ee^{-b x}\cos c x \to \frac12\Bigl[\frac1{p+a+\ii c}+\frac1{p+a-\ii c}-\frac1{p+b+\ii c}-\frac1{p+b-\ii c}\Bigr] \]因此有 \[ \int_0^\infty \frac{\ee^{-a x}-\ee^{-b x}}{x}\cos c x\dd{x} = \int_0^\infty \frac12\Bigl[\frac1{p+a+\ii c}+\frac1{p+a-\ii c}-\frac1{p+b+\ii c}-\frac1{p+b-\ii c}\Bigr]dp \] \[ = \bigl.\frac12\ln\frac{(p+a)^2+c^2}{(p+b)^2+c^2}\bigr|_0^\infty = \frac12\ln\frac{b^2+c^2}{a^2+c^2} \]

  \item \[ 1-\cos b x \to \frac1p-\frac12\frac1{p+\ii b}-\frac12\frac1{p-\ii b} \]因此有 \[ \frac{1-\cos b x}{x} \to \int_p^\infty \Bigl(\frac1p-\frac12\frac1{p+\ii b}-\frac12\frac1{p-\ii b}\Bigr)dp = \frac12\ln\frac{p^2+b^2}{p^2} \]故 \[ \int_0^\infty \frac{1-\cos b x}{x^2}\dd{x} = \int_0^\infty \frac12\ln\frac{p^2+b^2}{p^2}\dd{p} = \frac12\pi b \]

  \item \[ \frac{\sin x t}{x(x^2+1)} \to \frac1{(x^2+1)(p^2+x^2)} \]因此有 \[ \int_0^\infty \frac{\sin x t}{x(x^2+1)}\dd{x} \to \int_0^\infty \frac1{(x^2+1)(p^2+x^2)}\dd{x} = \int_0^\infty \frac1{p^2-1}\Bigl[\frac1{x^2+1}-\frac1{x^2+p^2}\Bigr]dx = \frac{\pi}{2}\frac1{(p+1)p} \]而同时有 \(\displaystyle\frac1{p(p+1)} = \frac1p-\frac1{p+1} \gets 1-\ee^{-t}\)，因此上述结果为 \[ \int_0^\infty \frac{\sin x t}{x(x^2+1)}\dd{x} = \frac{\pi}{2}(1-\ee^{-t})\quad (t>0) \]而上述积分关于 \(t\) 为奇函数，因此有 \[ \int_0^\infty \frac{\sin x t}{x(x^2+1)}\dd{x} = \frac{\pi}{2}\operatorname{sgn}(t)(1-\ee^{-|t|}) \]
\end{enumerate}
\end{solution}
\end{question}

\begin{question}
求解下列积分方程，其中 \(\alpha\) 为已知复数，\(f(t)\) 为已知函数，\(u(\xi)\) 是未知函数。\[ f(t)=\int_0^t \frac{u(\xi)}{(t-\xi)^\alpha}\dd{}\xi,\quad t>0,\ 0<\operatorname{Re}\alpha<1. \]
\begin{solution}
两边作 Laplace 变换并使用卷积定理：
\[
F(p)=\Gamma(1-\alpha)p^{\alpha-1}U(p),
\qquad
U(p)=\frac{p^{1-\alpha}F(p)}{\Gamma(1-\alpha)}.
\]
又因为
\[
\frac1{p^\alpha}
\longleftrightarrow
\frac{t^{\alpha-1}}{\Gamma(\alpha)},
\]
可将反演结果写成 Riemann--Liouville 形式
\[
u(t)
=\frac{1}{\Gamma(\alpha)\Gamma(1-\alpha)}
\dv{}{t}\int_0^t f(\xi)(t-\xi)^{\alpha-1}\dd{\xi}.
\]
利用 Euler 反射公式，最终得到
\[
u(t)
=\frac{\sin(\pi\alpha)}{\pi}
\dv{}{t}\int_0^t
\frac{f(\xi)}{(t-\xi)^{1-\alpha}}\dd{\xi}.
\]
这里的导数按分数阶积分意义理解；这样无需在上端点对奇异核作分部积分。
\end{solution}
\end{question}

\begin{question}
利用 Laplace 变换，求解下列常微分方程边值问题：
\[ \dv[2]{y}{x}-k^2 y(x)=\delta(x-x'),\quad x>0,\ k>0,\ x'>0, \] \[ y'(0)=0,\ \bigl.y\bigr|_{x\to\infty}=0 \]
\begin{solution}
对等式两边作拉普拉斯变换，可以得到 \[ p^2 Y(p) - p y(0)-k^2 Y(p)= \ee^{-p x'} \]因此有 \(Y(p)=\dfrac{\ee^{-p x'}+p y(0)}{p^2-k^2}\)。故对其反演可以得到，\[ y(x)= \frac{\eta(x-x')}{k}\sinh\bigl(k(x-x')\bigr) + y(0)\cosh(k x) \]当 \(x\to\infty\) 时，若 \(y(x)\) 有界，则有 \(\dfrac{\ee^{-k x'}}{k}+y(0)=0\)，因此有 \(y(0)=-\dfrac{\ee^{-k x'}}{k}\)，代入可以得到 \[ y(x)=\frac1k\Bigl(\eta(x-x')\sinh(k(x-x'))-\ee^{-k x'}\cosh(k x)\Bigr) = \begin{cases} -\dfrac1k \ee^{-k x'}\cosh(k x) & x<x',\\[2mm] -\dfrac1k \cosh(k x') \ee^{-k x} & x>x' \end{cases} \]
\end{solution}
\end{question}

\begin{question}

\end{solution}

\begin{question}
求下列常微分方程边值问题相应的Green函数

\begin{enumerate}
  \item \[
  \begin{cases} \dfrac{d^2y}{dx^2}-k^2 y(x)=f(x), & x>0,\ k>0,\\ \bigl.\dfrac{dy}{dx}\bigr|_{x=0}=v_0,\ \bigl.y(x)\bigr|_{x\to\infty}\text{有界}, & \text{其中 }v_0\text{ 是已知常数} \end{cases}
  \]
  \item \[
  \begin{cases} \dfrac{d^2y}{dx^2}+k^2 y(x)=f(x), & a<x<b,\ k>0,\\ y(a)=y_0,\ \bigl.\dfrac{dy}{dx}\bigr|_{x=b}=v_0, & \text{其中 }y_0,\ v_0\text{ 是已知常数} \end{cases}
  \]
\end{enumerate}

\begin{solution}
\begin{enumerate}
  \item 格林函数 \(G(x;s)\) 满足 \[ \dv[2]{G(x;s)}{x} - k^2 G(x;s)=\delta(x-s),\quad \bigl.\dv{G(x;s)}{x}\bigr|_{x=0}=0,\quad \bigl.G(x;s)\bigr|_{x\to\infty}\text{有界} \]对于 \(x<s\) 的部分，我们可以轻松给出解 \(G(x;s)=A\cosh(k x)\)。而对于 \(x>s\) 的部分，设解为 \(G(x;s)=B \ee^{k x}+C \ee^{-k x}\)，由于函数有界，故 \(B=0\)，因此代入 \(x=s\) 处连续性条件，可以得到 \[ \begin{cases} A\cosh(k s)= C \ee^{-k s},\\ -C k \ee^{-k s} - A k\sinh(k s) = 1 \end{cases} \]解得 \[ \begin{cases} A=-\dfrac1k \ee^{-k s},\\ C=-\dfrac1k\cosh(k s) \end{cases} \]故我们解得 \[ G(x;s)=\begin{cases} -\dfrac1k \ee^{-k s}\cosh(k x) & 0<x<s,\\[2mm] -\dfrac1k \cosh(k s) \ee^{-k x} & x>s \end{cases} \]

  \item 格林函数 \(G(x;s)\) 满足
  \[
  \dv[2]{G(x;s)}{x}+k^2G(x;s)=\delta(x-s),
  \quad G(a;s)=0,\quad G_x(b;s)=0.
  \]
  当 \(a<x<s\) 时取 \(G=A\sin(k(x-a))\)，当 \(s<x<b\) 时取
  \(G=B\cos(k(x-b))\)。由连续性和导数跳跃条件得到
  \[
  \begin{cases}
  A\sin(k(s-a))=B\cos(k(s-b)),\\
  -Bk\sin(k(s-b))-Ak\cos(k(s-a))=1.
  \end{cases}
  \]
  当 \(\cos(k(b-a))\ne0\) 时，
  \[
  A=-\frac{\cos(k(s-b))}{k\cos(k(b-a))},
  \qquad
  B=-\frac{\sin(k(s-a))}{k\cos(k(b-a))}.
  \]
  因而
  \[
  G(x;s)=
  -\frac{\sin(k(x_<-a))\cos(k(x_>-b))}
  {k\cos(k(b-a))},
  \]
  其中 \(x_<=\min(x,s)\)、\(x_>=\max(x,s)\)。当
  \(\cos(k(b-a))=0\) 时，齐次边值问题存在非零解，通常的 Green
  函数不存在。
\end{enumerate}
\end{solution}
\end{question}

\begin{question}
分别用下列三种方法求解常微分方程初值问题：
\begin{enumerate}
  \item 常数变易法；
  \item 利用 Laplace 变换；
  \item Green 函数方法。
\end{enumerate}
\[ \dv[2]{y(t)}{t} - k^2 y(t)=f(t),\quad t>0,\ k>0, \] \[ y(0)=A,\ \bigl.\dv{y(t)}{t}\bigr|_{t=0}=B \]
\begin{solution}
\textbf{常数变易法}\par

考察齐次方程 \(\dfrac{d^2y}{dt^2}=k^2 y,\ y(0)=A,\ \bigl.\dfrac{dy}{dt}\bigr|_{t=0}=B\)，有 \(y(t)=A\cosh(k t)+\dfrac{B}{k}\sinh(k t)\)。对于非齐次方程，设 \(y(t)=A\cosh(k t)+\dfrac{B}{k}\sinh(k t) + u(t) \ee^{k t}\)，则有 \[ \dv[2]{u}{t}+2k\dv{u}{t}=f(t) \ee^{-k t},\quad u(0)=0,\quad \bigl.\dv{u}{t}\bigr|_{t=0}=0 \]即有 \[ \dv{}{t}\bigl[\ee^{2k t}\dv{u}{t}\bigr]=f(t) \ee^{k t}\ \Rightarrow\ u(t)=\int_0^t \ee^{-2k t_1}\Bigl(\int_0^{t_1} f(t_2) \ee^{k t_2}\dd{t}_2\Bigr)\dd{t}_1 \]改变积分顺序，有
\begin{center}
\begin{tikzpicture}[scale=0.8]
  \draw[->] (0,0) -- (3.5,0) node[right] {$t_1$};
  \draw[->] (0,0) -- (0,3.5) node[above] {$t_2$};
  \draw[thick] (0,0) -- (2.5,2.5);
  \draw[thick,fill=black!10] (0,0) -- (2.5,2.5) -- (2.5,0) -- cycle;
  \node at (2.5,0) [below] {$t$};
  \node at (0,2.5) [left] {$t$};
  \node at (1.25,1.25) {$\mathcal{D}$};
\end{tikzpicture}
\end{center}

改变积分次序，可得
\[
u(t) = \int_0^t f(t_2)\,\ee^{k t_2}\Bigl(\int_{t_2}^t \ee^{-2k t_1}\dd{t}_1\Bigr)\dd{t}_2
= \frac{1}{2k}\int_0^t f(t_2)\bigl(\ee^{-k t_2}-\ee^{k(t_2-2t)}\bigr)\dd{t}_2.
\]
因此原方程的解为
\[
y(t)=A\cosh(kt)+\frac{B}{k}\sinh(kt)
+\frac1k\int_0^t\sinh\bigl(k(t-\tau)\bigr)f(\tau)\dd{\tau}.
\]

\textbf{Laplace 变换}\par

记 \(Y(s)=\mathcal{L}\{y\}(s)\)、\(F(s)=\mathcal{L}\{f\}(s)\)。
对方程作 Laplace 变换，得到
\[
s^2Y(s)-sA-B-k^2Y(s)=F(s),
\]
即
\[
Y(s)=\frac{sA+B}{s^2-k^2}+\frac{F(s)}{s^2-k^2}.
\]
由
\[
\mathcal{L}^{-1}\!\left\{\frac{s}{s^2-k^2}\right\}=\cosh(kt),
\qquad
\mathcal{L}^{-1}\!\left\{\frac{1}{s^2-k^2}\right\}
=\frac1k\sinh(kt),
\]
以及卷积定理，得到
\[
y(t)=A\cosh(kt)+\frac{B}{k}\sinh(kt)
+\frac1k\int_0^t\sinh\bigl(k(t-\tau)\bigr)f(\tau)\dd{\tau}.
\]

\textbf{Green 函数方法}\par

取满足零初值条件的推迟 Green 函数
\[
G(t,\tau)=\frac1k\sinh\bigl(k(t-\tau)\bigr)H(t-\tau).
\]
它满足
\[
\left(\pdv[2]{}{t}-k^2\right)G(t,\tau)=\delta(t-\tau),
\qquad
G(0,\tau)=\left.\pdv{G(t,\tau)}{t}\right|_{t=0}=0.
\]
齐次初值所对应的解为
\[
y_{\mathrm h}(t)=A\cosh(kt)+\frac{B}{k}\sinh(kt).
\]
因此
\[
\begin{aligned}
y(t)
&=y_{\mathrm h}(t)+\int_0^\infty G(t,\tau)f(\tau)\dd{\tau}\\
&=A\cosh(kt)+\frac{B}{k}\sinh(kt)
  +\frac1k\int_0^t\sinh\bigl(k(t-\tau)\bigr)f(\tau)\dd{\tau}.
\end{aligned}
\]
\end{solution}
\end{question}
