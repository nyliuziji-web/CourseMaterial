\heading{数学物理方法（上）-第9次作业}

\begin{question}
求下列函数的Laurent展开：

1. \(\displaystyle\frac{1}{z^2(z-1)}\)，在 \(z=1\) 附近展开；

2. \(\displaystyle\frac{1}{z^2(z-1)}\)，展开区域 \(1<|z|<\infty\)；

3. \(\displaystyle\frac{1}{1-\cos z}\)，在 \(z=2n\pi\) 附近展开（可只求出不为0的前四项系数）。

\begin{solution}
1. 令 \(u = z-1\)，有原式即为
\[
\frac{1}{u(u+1)^2} = \frac{1}{u} \sum_{n=0}^\infty \binom{-2}{n} u^n
= \sum_{n=-1}^\infty \binom{-2}{n+1} (z-1)^n.
\]

2. 有原式即为
\[
\frac{1}{z^2(z-1)} = \frac{1}{z^3} \frac{1}{1-1/z}
= \frac{1}{z^3} \sum_{n=0}^\infty z^{-n}
= \sum_{n=0}^\infty z^{-n-3}.
\]

3. 在 \(z=2n\pi\) 附近，有
\[
\cos z = \sum_{n=0}^\infty \frac{(-1)^n}{(2n)!} (z-2n\pi)^{2n},
\]
因此有
\[
\frac{1}{1-\cos z}
= \frac{1}{\displaystyle\sum_{n=1}^\infty \frac{(-1)^{n-1}}{(2n)!} (z-2n\pi)^{2n}}
= \frac{1}{\displaystyle\frac{(z-2n\pi)^2}{2!} - \frac{(z-2n\pi)^4}{4!} + \frac{(z-2n\pi)^6}{6!} - \frac{(z-2n\pi)^8}{8!} + \cdots}
\]
设
\[
\frac{1}{1-\cos z}
= a_0 (z-2n\pi)^{-2} + a_1 + a_2 (z-2n\pi)^2 + a_3 (z-2n\pi)^4 + \cdots,
\]
则有
\[
\bigl(a_0 (z-2n\pi)^{-2}+a_1+a_2 (z-2n\pi)^2+a_3 (z-2n\pi)^4+\cdots\bigr)
\]
\[
\times \biggl(\frac{(z-2n\pi)^2}{2!}-\frac{(z-2n\pi)^4}{4!}+\frac{(z-2n\pi)^6}{6!}-\frac{(z-2n\pi)^8}{8!}+\cdots\biggr)
\]
\[
= \frac{a_0}{2!} + \Bigl(\frac{a_1}{2!}-\frac{a_0}{4!}\Bigr)(z-2n\pi)^2
+ \Bigl(\frac{a_2}{2!}-\frac{a_1}{4!}+\frac{a_0}{6!}\Bigr)(z-2n\pi)^4
+ \Bigl(\frac{a_3}{2!}-\frac{a_2}{4!}+\frac{a_1}{6!}-\frac{a_0}{8!}\Bigr)(z-2n\pi)^6+\cdots
\]
故有方程
\[
\begin{cases}
\dfrac{a_0}{2!} = 1,\\[6pt]
\dfrac{a_1}{2!} - \dfrac{a_0}{4!} = 0,\\[6pt]
\dfrac{a_2}{2!} - \dfrac{a_1}{4!} + \dfrac{a_0}{6!} = 0,\\[6pt]
\dfrac{a_3}{2!} - \dfrac{a_2}{4!} + \dfrac{a_1}{6!} - \dfrac{a_0}{8!} = 0
\end{cases}
\]

解得
\[
\begin{cases}
a_0 = 2,\\
a_1 = \dfrac{1}{6},\\
a_2 = 2\left(\dfrac{1}{6\cdot 4!} - \dfrac{2}{6!}\right) = \dfrac{1}{5!},\\
a_3 = 2\left(\dfrac{1}{5!\cdot 4!} - \dfrac{1}{6\cdot 6!} + \dfrac{2}{8!}\right) = \dfrac{40}{3\cdot 8!}
\end{cases}
\]

因此有
\[
\frac{1}{1-\cos z}
= 2(z-2n\pi)^{-2} + \frac{1}{6} + \frac{1}{5!}(z-2n\pi)^2 + \frac{40}{3\cdot 8!}(z-2n\pi)^4 + \cdots
\]
\end{solution}
\end{question}

\begin{question}
判断下列函数孤立奇点的性质，如果是极点，确定其阶数：

1. \(\displaystyle\frac{1}{z^2+a^2},\,a\neq 0,\,n\in\mathbb{Z}\)；

2. \(\displaystyle\frac{\cos a z}{z^2}\)；

3. \(\displaystyle\frac{\cos a z - \cos b z}{z^2},\,a^2\neq b^2\)；

4. \(\displaystyle\frac{\sin z}{z^2} - \frac{1}{z}\)；

5. \(\displaystyle\cosh\frac{1}{\sqrt{z}}\)；

6. \(\displaystyle\frac{\sqrt{z}}{\sinh\sqrt{z}}\)；

7. \(\displaystyle\frac{\ln z}{z^2-1}\)；

8. \(\displaystyle\int_0^z \frac{\sin\sqrt{\zeta}}{\sqrt{\zeta}}\dd{}\zeta\)；

9. \(\displaystyle z^n + a^n,\,a\neq 0,\,n\in\mathbb{Z}\)；

10. \(\displaystyle\frac{z^2}{\sin a z},\,a\neq 0\)。

\begin{solution}
1. 函数的孤立奇点为 \(z=\pm \ii a\)，有
\[
\lim_{z\to\ii a} \frac{1}{z^2+a^2} = \infty,\quad
\lim_{z\to-\ii a} \frac{1}{z^2+a^2} = \infty,
\]
故其为函数的极点。有
\[
\lim_{z\to\ii a} (z-\ii a) \cdot \frac{1}{z^2+a^2}
= \lim_{z\to\ii a} \frac{1}{z+\ii a} = \frac{1}{2\ii a} \neq 0
\]
\[
\lim_{z\to-\ii a} (z+\ii a) \cdot \frac{1}{z^2+a^2}
= \lim_{z\to-\ii a} \frac{1}{z-\ii a} = \frac{1}{-2\ii a} \neq 0
\]
因此 \(\pm \ii a\) 为一阶极点。

2. 函数的孤立奇点为 \(z=0,\infty\)，有
\[
\lim_{z\to 0} \frac{\cos a z}{z^2} = \infty,
\]
故其为函数的极点。有
\[
\lim_{z\to 0} z^2 \cdot \frac{\cos a z}{z^2} = 1 \neq 0
\]
因此 \(0\) 为二阶极点。

   (1) 若 \(a=0\)，则 \(\displaystyle\lim_{z\to\infty} \frac{\cos a z}{z^2} = 0\)，故 \(\infty\) 为函数的可去奇点。
   (2) 若 \(a\neq 0\)，则 \(\displaystyle\lim_{z\to\infty} \frac{\cos a z}{z^2}\) 不存在，因此 \(\infty\) 为函数的本性奇点。

3. 函数的孤立奇点为 \(z=0,\infty\)，有
\[
\lim_{z\to 0} \frac{\cos (a z) - \cos (b z)}{z^2}
= \lim_{z\to 0} \frac{\frac12 b z^2 - \frac12 a z^2}{z^2}
= \frac12(b-a),
\]
故 \(0\) 为函数的可去奇点。而 \(\displaystyle\lim_{z\to\infty} \frac{\cos a z - \cos b z}{z^2}\) 不存在，因此 \(\infty\) 为函数的本性奇点。

4. 函数的孤立奇点为 \(z=0,\infty\)，有
\[
\lim_{z\to 0} \frac{\sin z}{z^2} - \frac{1}{z} = 0,
\]
故其为函数的可去奇点。\(\displaystyle\lim_{z\to\infty} \frac{\sin z}{z^2} - \frac{1}{z}\) 不存在，因此其为函数的本性奇点。

5. 函数的孤立奇点为 \(z=0\)，但 \(\displaystyle\lim_{z\to 0} \cosh\frac{1}{\sqrt{z}}\) 不存在，因此其为函数的本性奇点。

6. 函数的孤立奇点为 \(z = -n^2\pi^2,\,n\in\mathbb{N}\)，(因为 \(\sinh\sqrt{z} = \sinh(\pm \ii n\pi) = 0\))。

   (1) 若 \(n=0\)，有 \(z=0\)，则 \(\displaystyle\lim_{z\to 0} \frac{\sqrt{z}}{\sinh\sqrt{z}} = 1\)，因此其为可去奇点。
   (2) 若 \(n\neq 0\)，有 \(z=-n^2\pi^2\)，则 \(\displaystyle\lim_{z\to -n^2\pi^2} \frac{\sqrt{z}}{\sinh\sqrt{z}} = \infty\)，因此其为极点，有
   \[
   \lim_{z\to -n^2\pi^2} (z+n^2\pi^2) \frac{\sqrt{z}}{\sinh\sqrt{z}}
   = \lim_{\varepsilon\to 0} \varepsilon \cdot \frac{\sqrt{\varepsilon - n^2\pi^2}}{\sinh\sqrt{\varepsilon - n^2\pi^2}}
   \]
   \[
   = \lim_{\varepsilon\to 0} \frac{\varepsilon\sqrt{\varepsilon - n^2\pi^2}}
        {\sinh\!\bigl(\sqrt{-n^2\pi^2}\,(1-\frac{\varepsilon}{2n^2\pi^2})\bigr)}
   = \lim_{\varepsilon\to 0} \frac{\varepsilon\sqrt{-n^2\pi^2}}
        {-\frac{\varepsilon}{2n^2\pi^2}\sqrt{-n^2\pi^2}}
   = -2n^2\pi^2 \neq 0
   \]
   因此其为一阶极点。

   *(无穷远点 \(\infty\) 为非孤立奇点。)*

7. 函数的孤立奇点为 \(z=0,\pm 1,\infty\)，有 \(\displaystyle\lim_{z\to 0} \frac{\ln z}{z^2-1}\) 不存在，因此 \(z=0\) 为本性奇点，而 \(z\to\pm 1\)，

   (1) 若 \(\bigl.\ln(z)\bigr|_{z=1}=0\)，则
   \[
   \lim_{z\to 1} \frac{\ln z}{z^2-1}
   = \lim_{z\to 1} \frac{\ln z}{2(z-1)}
   = \lim_{z\to 0} \frac{\ln(1+z)}{2z}
   = \frac12,
   \]
   因此 \(z=1\) 为可去奇点。
   (2) 若 \(\bigl.\ln(z)\bigr|_{z=1}=2\pi n\ii,\,n\in\mathbb{Z}\setminus\{0\}\)，则
   \[
   \lim_{z\to 1} \frac{\ln z}{z^2-1} = \infty,
   \]
   因此 \(z=1\) 为极点。而
   \[
   \lim_{z\to 1} (z-1) \frac{\ln z}{z^2-1}
   = \lim_{z\to 1} \frac{\ln z}{z+1} = \pi n\ii \neq 0,
   \]
   因此 \(z=1\) 为一阶极点。

   有 \(\bigl.\ln(z)\bigr|_{z=-1}=(2\pi n + \pi)\ii,\,n\in\mathbb{Z}\)，则
   \[
   \lim_{z\to -1} \frac{\ln z}{z^2-1} = \infty,
   \]
   因此 \(z=-1\) 为极点。而
   \[
   \lim_{z\to -1} (z+1) \frac{\ln z}{z^2-1}
   = \lim_{z\to -1} \frac{\ln z}{z-1}
   = \frac{(2\pi n + \pi)}{2}\ii \neq 0,
   \]
   因此 \(z=-1\) 为一阶极点。有 \(\displaystyle\lim_{z\to\infty} \frac{\ln z}{z^2-1}=0\)，因此其为可去奇点。

8. \[
\int_0^z \frac{\sin\sqrt{\zeta}}{\sqrt{\zeta}}\dd{}\zeta
= \int_0^z 2\sin\sqrt{\zeta}\dd{}(\sqrt{\zeta})
= \bigl.-2\cos\sqrt{\zeta}\bigr|_0^z
= 2 - 2\cos\sqrt{z}.
\]
其孤立奇点位于 \(z=\infty\)。而 \(\displaystyle\lim_{z\to\infty} 2 - 2\cos\sqrt{z}\) 不存在，因此 \(z=\infty\) 为本性奇点。

9. (1) 当 \(n>0\) 时，函数在 \(z=\infty\) 为孤立奇点。有 \(\displaystyle\lim_{z\to\infty} z^n + a^n = \infty\)，因此其为极点。作代换 \(t=1/z\) 后有
   \[
   \lim_{t\to 0} t^n \left(\frac{1}{t^n} + a^n\right) = 1 \neq 0,
   \]
   因此其为 \(n\) 阶极点。
   (2) 当 \(n=0\) 时，函数在 \(z=0\) 为孤立奇点。有 \(\displaystyle\lim_{z\to 0} z^0 + a^0 = 1\)。因此 \(z=0\) 为可去奇点。
   (3) 当 \(n<0\) 时，函数在 \(z=0\) 为孤立奇点。有 \(\displaystyle\lim_{z\to 0} z^n + a^n = \infty\)，因此 \(z=0\) 为极点。同时有
   \[
   \lim_{z\to 0} z^{-n} (z^n + a^n) = \lim_{z\to 0} (1 + z^{-n}a^n) = 1 \neq 0,
   \]
   因此 \(z=0\) 为 \(-n\) 阶极点。

10. 函数的孤立奇点为 \(z=n\pi/a,\,n\in\mathbb{Z}\)。
   (1) \(n = 0\)，有 \(\displaystyle\lim_{z\to 0} \frac{z^2}{\sin a z} = 0\)，因此 \(z=0\) 为可去奇点。
   (2) \(n \neq 0\)，有 \(\displaystyle\lim_{z\to n\pi/a} \frac{z^2}{\sin a z} = \infty\)，因此 \(z=n\pi/a\) 为极点。同时有
   \[
   \lim_{z\to n\pi/a} \left(z - \frac{n\pi}{a}\right) \frac{z^2}{\sin a z}
   = \frac{1}{a} \left(\frac{n\pi}{a}\right)^2 \neq 0,
   \]
   因此其为一阶极点。*(\(z=\infty\) 为非孤立奇点)*

\hrule
\textbf{综上：}
\begin{enumerate}
\item \(z = \pm \ii a\) 为一阶极点。
\item \(z=0\) 为二阶极点；\(z=\infty\)：若 \(a=0\) 为可去奇点，若 \(a\neq 0\) 为本性奇点。
\item \(z=0\) 为可去奇点；\(z=\infty\) 为本性奇点。
\item \(z=0\) 为可去奇点；\(z=\infty\) 为本性奇点。
\item \(z=0\) 为本性奇点。
\item \(z=0\) 为可去奇点；\(z=-n^2\pi^2\,(n\neq 0)\) 为一阶极点；\(z=\infty\) 为非孤立奇点。
\item \(z=0\) 为本性奇点；\(z=1\)：若取主值分支为可去奇点，否则为一阶极点；\(z=-1\) 为一阶极点；\(z=\infty\) 为可去奇点。
\item \(z=\infty\) 为本性奇点。
\item \(n>0\)：\(z=\infty\) 为 \(n\) 阶极点；\(n=0\)：\(z=0\) 为可去奇点；\(n<0\)：\(z=0\) 为 \(-n\) 阶极点。
\item \(z=0\) 为可去奇点；\(z=n\pi/a\,(n\neq 0)\) 为一阶极点；\(z=\infty\) 为非孤立奇点。
\end{enumerate}
\end{solution}
\end{question}
