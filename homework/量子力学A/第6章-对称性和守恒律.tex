\heading{第6章-对称性和守恒律}
\begin{question}
在三维空间中考虑宇称算符 $\Pi$。
\begin{enumerate}
\item 证明 $\Pi\psi(\mathbf r)=\psi(-\mathbf r)$ 等同于一次镜面反射再加一次旋转。
\item 写出宇称在球坐标 $(r,\theta,\phi)$ 下的作用。
\item 证明氢原子轨道 $\psi_{n\ell m}=R_{n\ell}(r)Y_\ell^m(\theta,\phi)$ 是宇称本征态，本征值为 $(-1)^\ell$。
\end{enumerate}
\end{question}
\begin{solution}
	宇称算符 $\Pi$ 的作用是空间反演：$\Pi\psi(\mathbf r)=\psi(-\mathbf r)$，即 $(x,y,z)\to(-x,-y,-z)$。

	\textbf{(a) 宇称的分解}：这一变换可分解为两步：
	\begin{enumerate}
	\item 关于 $x$-$y$ 平面的镜面反射 $\sigma_{xy}$：
	\[
	\sigma_{xy}\psi(x,y,z)=\psi(x,y,-z).
	\]
	\item 再绕 $z$ 轴旋转 $\pi$：
	\[
	R_z(\pi)=\begin{pmatrix}\cos\pi&-\sin\pi&0\\\sin\pi&\cos\pi&0\\0&0&1\end{pmatrix}
	=\begin{pmatrix}-1&0&0\\0&-1&0\\0&0&1\end{pmatrix},
	\]
	作用在镜面反射后的坐标 $(x,y,-z)$ 上：
	\[
	R_z(\pi)\sigma_{xy}(x,y,z)=R_z(\pi)(x,y,-z)=(-x,-y,-z).
	\]
	\end{enumerate}
	因此 $\Pi = R_z(\pi) \cdot \sigma_{xy}$，即一次镜面反射加一次旋转等价于空间反演。

	\textbf{(b) 球坐标下的作用}：球坐标与直角坐标的关系为
	\[
	x = r\sin\theta\cos\phi,\quad y = r\sin\theta\sin\phi,\quad z = r\cos\theta.
	\]
	宇称变换 $x\to -x,\,y\to -y,\,z\to -z$ 在球坐标下对应
	\[
	r\to\sqrt{(-x)^2+(-y)^2+(-z)^2}=r,
	\]
	\[
	\theta\to\arccos\frac{-z}{r}=\arccos\!\left(-\frac{z}{r}\right)=\pi-\theta,
	\]
	\[
	\phi\to\arctan\frac{-y}{-x}=\arctan\frac{y}{x}+\pi=\phi+\pi.
	\]
	因此
	\[
	\Pi:\;(r,\theta,\phi)\longmapsto (r,\pi-\theta,\phi+\pi),
	\]
	所以
	\[
	\Pi\psi(r,\theta,\phi)=\psi(r,\pi-\theta,\phi+\pi).
	\]

	\textbf{(c) 氢原子轨道的宇称}：中心势定态波函数可分离变量：
	\[
	\psi_{n\ell m}(r,\theta,\phi)=R_{n\ell}(r)\,Y_\ell^m(\theta,\phi).
	\]
	球谐函数的表达式为
	\[
	Y_\ell^m(\theta,\phi)=(-1)^m\sqrt{\frac{2\ell+1}{4\pi}\frac{(\ell-m)!}{(\ell+m)!}}\,P_\ell^m(\cos\theta)\,e^{\ii m\phi},
	\]
	其中 $P_\ell^m$ 为连带勒让德函数。利用连带勒让德函数的变换性质
	\[
	P_\ell^m(-\cos\theta)=(-1)^{\ell-m}P_\ell^m(\cos\theta)
	\]
	和指数因子的变换
	\[
	e^{\ii m(\phi+\pi)}=e^{\ii m\phi}e^{\ii m\pi}=(-1)^m e^{\ii m\phi},
	\]
	可得
	\[
	Y_\ell^m(\pi-\theta,\phi+\pi)=(-1)^m\sqrt{\frac{2\ell+1}{4\pi}\frac{(\ell-m)!}{(\ell+m)!}}\,P_\ell^m(-\cos\theta)\,e^{\ii m(\phi+\pi)}
	\]
	\[
	=(-1)^m(-1)^{\ell-m}(-1)^m\sqrt{\frac{2\ell+1}{4\pi}\frac{(\ell-m)!}{(\ell+m)!}}\,P_\ell^m(\cos\theta)\,e^{\ii m\phi}
	\]
	\[
	=(-1)^\ell Y_\ell^m(\theta,\phi).
	\]

	因此
	\[
	\Pi\psi_{n\ell m}(r,\theta,\phi)=R_{n\ell}(r)Y_\ell^m(\pi-\theta,\phi+\pi)
	=(-1)^\ell R_{n\ell}(r)Y_\ell^m(\theta,\phi)=(-1)^\ell\psi_{n\ell m}(r,\theta,\phi).
	\]

	故中心势定态 $\psi_{n\ell m}=R_{n\ell}(r)Y_\ell^m(\theta,\phi)$ 是宇称本征态，宇称本征值为 $\boxed{(-1)^\ell}$。
	\end{solution}

\begin{question}
设 $Q$ 是厄米算符，证明
\[
U=\exp(\ii Q)
\]
是幺正算符。提示：先由幂级数证明 $U^\dagger=\exp(-\ii Q)$。
\end{question}
\begin{solution}
	设 $Q$ 是厄米算符，即 $Q^\dagger=Q$。定义算符函数
	\[
	U=\exp(\ii Q)=\sum_{n=0}^\infty\frac{(\ii Q)^n}{n!}.
	\]

	\textbf{第一步：求 $U^\dagger$}。对幂级数逐项取伴随（注意 $\ii^\dagger=-\ii$ 以及 $Q^\dagger=Q$）：
	\[
	U^\dagger=\left(\sum_{n=0}^\infty\frac{(\ii Q)^n}{n!}\right)^\dagger
	=\sum_{n=0}^\infty\frac{(-\ii Q^\dagger)^n}{n!}
	=\sum_{n=0}^\infty\frac{(-\ii Q)^n}{n!}
	=\exp(-\ii Q).
	\]

	\textbf{第二步：验证幺正性}。因为 $Q$ 与自身任意函数对易（幂级数中各项都是 $Q$ 的多项式，彼此显然对易），所以
	\[
	U^\dagger U=\exp(-\ii Q)\exp(\ii Q)
	=\exp[(-\ii Q)+(\ii Q)+\frac12[(-\ii Q),(\ii Q)]+\cdots]
	\]
	但更简单地，由于 $(-\ii Q)$ 与 $(\ii Q)$ 对易（相差一个 $-1$ 因子，仍是 $Q$ 的函数，彼此对易），可直接合并指数：
	\[
	U^\dagger U = \exp(-\ii Q + \ii Q) = \exp(0) = 1.
	\]
	同理
	\[
	UU^\dagger = \exp(\ii Q)\exp(-\ii Q) = \exp(0) = 1.
	\]

	因此 $U$ 满足 $U^\dagger U = UU^\dagger = 1$，即 $U$ 是幺正算符：
	\[
	\boxed{U^\dagger = U^{-1}}.
	\]
	\end{solution}

\begin{question}
证明动量算符在空间平移下不变：若 $T(a)=\exp(-\ii ap/\hbar)$，则
\[
T(a)^\dagger pT(a)=p.
\]
\end{question}
\begin{solution}
	空间平移算符定义为
	\[
	T(a)=\exp\!\left(-\frac{\ii a p}{\hbar}\right).
	\]

	\textbf{第一步：证明 $[p,T(a)]=0$}。由于 $T(a)$ 是动量算符 $p$ 的指数函数（幂级数展开为 $T(a)=\sum_{n=0}^\infty\frac{(-\ii a/\hbar)^n}{n!}p^n$），而任意算符与自身对易，因此 $[p,p^n]=0$ 对所有 $n$ 成立，故
	\[
	[p,T(a)]=0.
	\]

	\textbf{第二步：计算变换后的动量算符}。由 $T(a)^\dagger=\exp(\ii a p/\hbar)$（因为 $p$ 是厄米算符，$(-\ii a p/\hbar)^\dagger=\ii a p/\hbar$），利用 $[p,T(a)]=0$ 得
	\[
	T(a)^\dagger p\,T(a) = p\,T(a)^\dagger T(a).
	\]
	由幺正性 $T(a)^\dagger T(a)=1$，所以
	\[
	T(a)^\dagger p\,T(a) = p.
	\]

	因此动量算符在空间平移下不变：
	\[
	\boxed{T(a)^\dagger p\,T(a)=p}.
	\]

	这说明空间平移不改变动量算符本身，只改变位置算符的原点。
	\end{solution}

\begin{question}
设算符 $Q(x,p)$ 可展开为 $x$ 与 $p$ 的幂级数，证明
\[
T(a)^\dagger Q(x,p)T(a)=Q(x+a,p).
\]
并由此说明一维哈密顿量 $H=p^2/(2m)+V(x)$ 在平移下如何变换。
\end{question}
\begin{solution}
	\textbf{准备工作：平移算符对基本算符的作用}。先证明两个基本关系：
	\begin{enumerate}
	\item 由 $[p,x]=-\ii\hbar$ 和 Baker--Campbell--Hausdorff 引理，
	\[
	T(a)^\dagger xT(a)=e^{\ii a p/\hbar}\,x\,e^{-\ii a p/\hbar}
	=x+\frac{\ii a}{\hbar}[p,x]+\frac12\left(\frac{\ii a}{\hbar}\right)^2[p,[p,x]]+\cdots.
	\]
	因为 $[p,x]=-\ii\hbar$，而 $[p,[p,x]]=[p,-\ii\hbar]=0$，高阶项全为零，故
	\[
	T(a)^\dagger xT(a)=x+\frac{\ii a}{\hbar}(-\ii\hbar)=x+a.
	\]
	\item 由 $[p,T(a)]=0$ 直接得
	\[
	T(a)^\dagger p\,T(a)=p.
	\]
	\end{enumerate}

	\textbf{对任意可展开算符的变换}：设 $Q(x,p)$ 可展开为 $x$ 与 $p$ 的幂级数
	\[
	Q(x,p)=\sum_{m,n}c_{mn}x^mp^n,
	\]
	其中 $c_{mn}$ 为系数。逐项变换：
	\[
	T(a)^\dagger Q(x,p)T(a)=\sum_{m,n}c_{mn}\,T(a)^\dagger x^m p^n T(a).
	\]
	在每一乘积 $x^m p^n$ 中插入 $T(a)T(a)^\dagger=1$：
	\[
	T(a)^\dagger x^m p^n T(a)=(T(a)^\dagger x\,T(a))^m\,(T(a)^\dagger p\,T(a))^n=(x+a)^m p^n,
	\]
	其中第二步利用了 $T(a)^\dagger x T(a)=x+a$ 是 $c$-数（与 $p$ 对易），故各因子可重排。因此
	\[
	T(a)^\dagger Q(x,p)T(a)=\sum_{m,n}c_{mn}(x+a)^mp^n=Q(x+a,p).
	\]

	\textbf{对一维哈密顿量的应用}：一维哈密顿量为
	\[
	H=\frac{p^2}{2m}+V(x).
	\]
	由上面的一般结果，平移后
	\[
	H'=T(a)^\dagger H\,T(a)=\frac{p^2}{2m}+V(x+a).
	\]

	若势能具有周期性 $V(x+a)=V(x)$，则 $H'=H$，即
	\[
	T(a)^\dagger H\,T(a)=H,
	\]
	等价于 $[H,T(a)]=0$：
	\[
	\boxed{[H,T(a)]=0\quad\text{当 }V(x+a)=V(x)}.
	\]
	这正是布洛赫定理的基础——平移对称性使得平移算符与哈密顿量对易，从而存在共同本征态。
	\end{solution}

\begin{question}
由布洛赫条件
\[
\psi(x+a)=e^{\ii qa}\psi(x)
\]
证明波函数可写为
\[
\psi(x)=e^{\ii qx}u(x),\qquad u(x+a)=u(x).
\]
\end{question}
\begin{solution}
	\textbf{正向推导}：已知布洛赫条件（平移对称性）
	\[
	\psi(x+a)=e^{\ii qa}\psi(x),
	\]
	其中 $q$ 是实参数（准动量），$a$ 是晶格常数。定义辅助函数
	\[
	u(x)=e^{-\ii qx}\psi(x).
	\]

	检验 $u(x)$ 的周期性：
	\[
	u(x+a)=e^{-\ii q(x+a)}\psi(x+a)
	=e^{-\ii qx}e^{-\ii qa}\cdot e^{\ii qa}\psi(x)
	=e^{-\ii qx}\psi(x)=u(x).
	\]

	因此 $u(x)$ 是以 $a$ 为周期的周期函数。由定义反解 $\psi(x)$ 即得布洛赫波函数的标准形式：
	\[
	\psi(x)=e^{\ii qx}u(x),\qquad u(x+a)=u(x).
	\]

	\textbf{反向推导}：反之，若 $u(x+a)=u(x)$ 且 $\psi(x)=e^{\ii qx}u(x)$，则
	\[
	\psi(x+a)=e^{\ii q(x+a)}u(x+a)=e^{\ii qa}e^{\ii qx}u(x)=e^{\ii qa}\psi(x),
	\]
	立即回到了布洛赫条件。

	因此布洛赫条件与布洛赫波函数形式等价：
	\[
	\boxed{\psi(x)=e^{\ii qx}u(x),\quad u(x+a)=u(x)
	\Longleftrightarrow \psi(x+a)=e^{\ii qa}\psi(x)}.
	\]
	\end{solution}

\begin{question}
质量为 $m$ 的粒子在周期为 $a$ 的一维势场 $V(x)$ 中运动。设
\[
\psi_{nq}(x)=e^{\ii qx}u_{nq}(x),\qquad u_{nq}(x+a)=u_{nq}(x).
\]
\begin{enumerate}
\item 推导 $u_{nq}$ 满足的本征值方程。
\item 说明如何用有限差分方法在一个周期内数值求解能带 $E_{nq}$。
\item 对不同 $q$ 值画出最低能带的示意曲线。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) $u_{nq}$ 满足的本征值方程}：将布洛赫波函数
	\[
	\psi_{nq}(x)=e^{\ii qx}u_{nq}(x)
	\]
	代入定态薛定谔方程
	\[
	\left[-\frac{\hbar^2}{2m}\frac{\dd^2}{\dd x^2}+V(x)\right]\psi_{nq}=E_{nq}\psi_{nq}.
	\]

	先计算一阶导数：
	\[
	\psi'_{nq}= \ii q e^{\ii qx}u_{nq}+ e^{\ii qx}u'_{nq}=e^{\ii qx}(\ii q u_{nq}+u'_{nq}).
	\]

	再计算二阶导数：
	\[
	\psi''_{nq}= \ii q e^{\ii qx}(\ii q u_{nq}+u'_{nq})+ e^{\ii qx}(\ii q u'_{nq}+u''_{nq})
	=e^{\ii qx}\big[-\!q^2u_{nq}+2\ii q u'_{nq}+u''_{nq}\big].
	\]

	代入薛定谔方程（两边约去公因子 $e^{\ii qx}$）：
	\[
	-\frac{\hbar^2}{2m}\big[-q^2u_{nq}+2\ii q u'_{nq}+u''_{nq}\big]+V(x)u_{nq}=E_{nq}u_{nq}.
	\]

	整理得
	\[
	\left[-\frac{\hbar^2}{2m}\left(\frac{\dd}{\dd x}+\ii q\right)^2+V(x)\right]u_{nq}=E_{nq}u_{nq}.
	\]

	展开平方项：
	\[
	\left(\frac{\dd}{\dd x}+\ii q\right)^2=\frac{\dd^2}{\dd x^2}+2\ii q\frac{\dd}{\dd x}-q^2.
	\]

	代入后得到显式微分方程：
	\[
	-\frac{\hbar^2}{2m}u''_{nq}-\frac{\ii\hbar^2 q}{m}u'_{nq}+\frac{\hbar^2 q^2}{2m}u_{nq}+V(x)u_{nq}=E_{nq}u_{nq}.
	\]

	\textbf{(b) 有限差分法数值求解}：在一个周期 $[0,a]$ 内离散化。取 $N$ 个等分格点
	\[
	x_j=j\Delta x,\quad \Delta x=\frac{a}{N},\quad j=0,1,\ldots,N-1,
	\]
	其中 $u_j\equiv u(x_j)$。由周期边界条件 $u(0)=u(a)$，即 $u_0=u_N$。

	中心差分公式：
	\[
	u'_j\simeq\frac{u_{j+1}-u_{j-1}}{2\Delta x}+O(\Delta x^2),
	\]
	\[
	u''_j\simeq\frac{u_{j+1}-2u_j+u_{j-1}}{\Delta x^2}+O(\Delta x^2).
	\]

	周期边界条件要求首尾按模 $N$ 连接：$j=0$ 时 $u_{-1}=u_{N-1}$；$j=N-1$ 时 $u_N=u_0$。

	将差分公式代入本征值方程，得到矩阵形式的广义本征值问题：
	\[
	[H_q]\{u\}=E_{nq}\{u\},
	\]
	其中 $H_q$ 是 $N\times N$ 的三对角矩阵（加上周期边界产生的两个非对角元）。对每个固定的 $q$，数值对角化 $H_q$ 得到本征值 $E_{1q},E_{2q},\ldots$，即能带。

	\textbf{(c) 能带示意图}：由于 $q$ 与 $q+2\pi/a$ 等价（因为 $e^{\ii (q+2\pi/a)a}=e^{\ii qa}$），只需考虑第一布里渊区
	\[
	-\frac{\pi}{a}\le q\le\frac{\pi}{a}.
	\]

	最低能带 $E_{1q}$ 在 $q=0$ 处取极小值，在布里渊区边界 $q=\pm\pi/a$ 处弯曲变平（群速度为零）。第二能带 $E_{2q}$ 在布里渊区边界与第一能带之间有能隙。典型的能带结构示意图为：横轴 $q$ 从 $-\pi/a$ 到 $\pi/a$，纵轴为能量，最低能带呈抛物线状（近自由电子近似下 $E\approx\hbar^2q^2/2m$ 经周期势修正后形成）。

	因此，能带 $E_{nq}$ 由周期势 $V(x)$ 和波矢 $q$ 共同决定：
	\[
	\boxed{E_{nq}=\text{第 }n\text{ 个能带在波矢 }q\text{ 处的能量}}.
	\]
	\end{solution}

\begin{question}
考虑一维中两个质量分别为 $m_1,m_2$ 的粒子，哈密顿量中相互作用只依赖 $|x_1-x_2|$。
\begin{enumerate}
\item 求同时平移两个粒子的平移算符，并指出其生成元。
\item 证明该哈密顿量与总动量对易。
\item 说明这对应于系统的整体平移不变性。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 两粒子平移算符及其生成元}：两粒子同时平移 $a$ 的作用是
	\[
	T(a)\psi(x_1,x_2)=\psi(x_1-a,x_2-a).
	\]

	对无穷小平移 $a\to\delta a$ 作泰勒展开：
	\[
	T(\delta a)\psi(x_1,x_2)=\psi(x_1-\delta a,x_2-\delta a)
	=\psi(x_1,x_2)-\delta a\,\frac{\partial\psi}{\partial x_1}-\delta a\,\frac{\partial\psi}{\partial x_2}+O(\delta a^2).
	\]

	因此算符形式为
	\[
	T(\delta a)=1-\delta a(\partial_{x_1}+\partial_{x_2})+O(\delta a^2).
	\]

	利用动量算符 $p_i=-\ii\hbar\partial_{x_i}$，即 $\partial_{x_i}=\frac{\ii}{\hbar}p_i$，代入得
	\[
	T(\delta a)=1-\delta a\cdot\frac{\ii}{\hbar}(p_1+p_2)+O(\delta a^2)
	=1-\frac{\ii\delta a}{\hbar}P+O(\delta a^2),
	\]
	其中定义总动量算符
	\[
	P=p_1+p_2.
	\]

	有限平移由无穷小平移的连续复合得到：$T(a)=\lim_{N\to\infty}[T(a/N)]^N=\exp(-\ii a P/\hbar)$，即
	\[
	\boxed{T(a)=\exp\!\left[-\frac{\ii a}{\hbar}(p_1+p_2)\right]}.
	\]
	生成元是总动量 $P=p_1+p_2$。

	\textbf{(b) 证明 $[H,P]=0$}：两粒子哈密顿量为
	\[
	H=\frac{p_1^2}{2m_1}+\frac{p_2^2}{2m_2}+V(|x_1-x_2|).
	\]

	分别计算各部分与总动量 $P=p_1+p_2$ 的对易子。

	$p_1^2$ 与 $P$：
	\[
	[p_1^2,p_1+p_2]=[p_1^2,p_1]+[p_1^2,p_2]=0+0=0,
	\]
	因为 $p_1^2$ 与 $p_1$ 对易，与 $p_2$ 也对易（不同自由度）。同理 $[p_2^2,P]=0$。

	势能部分 $V(|x_1-x_2|)$：令 $u=x_1-x_2$，则 $V(|x_1-x_2|)=V(|u|)$。计算
	\[
	[P,V]=[p_1,V]+[p_2,V]=-\ii\hbar\frac{\partial V}{\partial x_1}-\ii\hbar\frac{\partial V}{\partial x_2}.
	\]

	由链式法则：
	\[
	\frac{\partial V}{\partial x_1}=V'(|u|)\frac{\partial |u|}{\partial x_1}=V'(|u|)\frac{x_1-x_2}{|x_1-x_2|},
	\]
	\[
	\frac{\partial V}{\partial x_2}=V'(|u|)\frac{\partial |u|}{\partial x_2}=V'(|u|)\frac{x_2-x_1}{|x_1-x_2|}=-V'(|u|)\frac{x_1-x_2}{|x_1-x_2|},
	\]
	因此 $\frac{\partial V}{\partial x_1}+\frac{\partial V}{\partial x_2}=0$，故
	\[
	[P,V]=0.
	\]

	综合得
	\[
	[H,P]=0,
	\]
	即总动量算符 $P=p_1+p_2$ 与哈密顿量对易，总动量守恒。

	\textbf{(c) 物理意义}：$[H,P]=0$ 意味着系统具有整体平移不变性——整个系统（两个粒子同时平移相同的距离）不改变哈密顿量的形式。根据诺特定理，平移对称性对应总动量守恒。由于相互作用只依赖于相对距离 $|x_1-x_2|$，质心运动是自由的。
	\end{solution}

\begin{question}
\begin{enumerate}
\item 证明宇称算符 $\Pi$ 是厄米算符。
\item 证明宇称算符的本征值只能为 $+1$ 或 $-1$。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 证明 $\Pi$ 是厄米算符}：宇称算符 $\Pi$ 的作用定义为 $(\Pi\psi)(\mathbf r)=\psi(-\mathbf r)$。

	先证明 $\Pi$ 是幺正算符。对于任意两个波函数 $f,g$，考虑内积
	\[
	\langle \Pi f|\Pi g\rangle=\int [\Pi f(\mathbf r)]^* [\Pi g(\mathbf r)]\,\dd^3 r
	=\int f^*(-\mathbf r)\,g(-\mathbf r)\,\dd^3 r.
	\]

	做变量代换 $\mathbf r'=-\mathbf r$，则 $\dd^3 r' = |\det(\partial\mathbf r'/\partial\mathbf r)|\,\dd^3 r = \dd^3 r$（雅可比行列式为 $|-1|=1$），积分限不变，得
	\[
	\langle \Pi f|\Pi g\rangle = \int f^*(\mathbf r')\,g(\mathbf r')\,\dd^3 r' = \langle f|g\rangle.
	\]

	因此 $\Pi^\dagger\Pi = 1$，$\Pi$ 是幺正算符。

	又由 $\Pi^2=1$（两次宇称变换回到原状）得 $\Pi^{-1}=\Pi$。结合幺正性 $\Pi^\dagger=\Pi^{-1}$，有
	\[
	\Pi^\dagger = \Pi^{-1} = \Pi,
	\]
	故 $\Pi$ 也是厄米算符：$\boxed{\Pi^\dagger = \Pi}$。

	\textbf{(b) 本征值只能为 $\pm1$}：设 $\psi$ 是 $\Pi$ 的本征态，满足
	\[
	\Pi\psi = \lambda\psi.
	\]

	两边再作用一次 $\Pi$：
	\[
	\Pi^2\psi = \Pi(\lambda\psi) = \lambda\Pi\psi = \lambda^2\psi.
	\]

	但由定义 $\Pi^2=1$，所以
	\[
	\lambda^2\psi = \psi,
	\]
	故 $\lambda^2=1$，即
	\[
	\boxed{\lambda = \pm 1}.
	\]

	因此宇称算符只有两个本征值：$+1$（偶宇称态）和 $-1$（奇宇称态）。
	\end{solution}

\begin{question}
真标量在宇称下不变，赝标量在宇称下变号。证明：真标量 $f$ 满足 $[\Pi,f]=0$，赝标量 $g$ 满足 $\{\Pi,g\}=0$。并讨论真矢量与赝矢量在宇称下的变换。
\end{question}
\begin{solution}
	\textbf{基本定义}：在宇称变换下，物理量按其变换性质分为四类：
	\begin{itemize}
	\item 真标量（scalar）：宇称下不变，$f\to f$.
	\item 赝标量（pseudoscalar）：宇称下变号，$g\to -g$.
	\item 真矢量（polar vector）：每个分量宇称下变号，$\mathbf V\to -\mathbf V$.
	\item 赝矢量（axial vector）：每个分量宇称下不变，$\mathbf A\to \mathbf A$.
	\end{itemize}

	\textbf{真标量}：在量子力学中，算符在宇称下的变换由 $\Pi^\dagger f\Pi$ 给出。对于真标量 $f$，要求宇称变换下不变：
	\[
	\Pi^\dagger f\Pi = f.
	\]

	由第 8 题已知 $\Pi^\dagger=\Pi$ 且 $\Pi^2=1$，即 $\Pi^{-1}=\Pi$。左乘 $\Pi$：
	\[
	\Pi\Pi^\dagger f\Pi = \Pi f \quad\Longrightarrow\quad f\Pi = \Pi f.
	\]

	因此 $[\Pi, f]=0$，即真标量与宇称算符对易：
	\[
	\boxed{[\Pi,f]=0\quad(\text{真标量})}.
	\]

	\textbf{赝标量}：对于赝标量 $g$，宇称变换下变号：
	\[
	\Pi^\dagger g\Pi = -g.
	\]

	同样利用 $\Pi^\dagger=\Pi$ 和 $\Pi^2=1$：
	\[
	\Pi(\Pi^\dagger g\Pi)=\Pi(-g) \quad\Longrightarrow\quad g\Pi = -\Pi g.
	\]

	因此 $\Pi g = -g\Pi$，即 $\{\Pi,g\}=0$，赝标量与宇称算符反对易：
	\[
	\boxed{\{\Pi,g\}=0\quad(\text{赝标量})}.
	\]

	\textbf{真矢量与赝矢量}：
	\begin{itemize}
	\item 真矢量 $\mathbf V$：每个分量在宇称下变号，即 $\Pi^\dagger V_i\Pi = -V_i$，等同于 $\Pi V_i = -V_i\Pi$，故真矢量各分量与 $\Pi$ 反对易：
	\[
	\{\Pi,V_i\}=0.
	\]
	\item 赝矢量 $\mathbf A$：每个分量在宇称下不变，即 $\Pi^\dagger A_i\Pi = A_i$，等同于 $\Pi A_i = A_i\Pi$，故赝矢量各分量与 $\Pi$ 对易：
	\[
	[\Pi,A_i]=0.
	\]
	\end{itemize}

	总结：物理量在宇称下的变换性质完全由它与宇称算符的对易或反对易关系描述。
	\end{solution}

\begin{question}
证明位置算符和动量算符都具有奇宇称，即在一维中
\[
\Pi x\Pi=-x,
\qquad
\Pi p\Pi=-p,
\]
并把结论推广到三维情形。
\end{question}
\begin{solution}
	\textbf{一维位置算符}：要证明 $\Pi x\Pi = -x$。任取波函数 $\psi(x)$，计算 $\Pi x\Pi\psi(x)$：

	首先 $\Pi\psi(x)=\psi(-x)$，所以
	\[
	(x\Pi\psi)(x)=x\psi(-x).
	\]

	再作用 $\Pi$：
	\[
	(\Pi x\Pi\psi)(x)=\Pi[x\psi(-x)]=(-x)\psi(x).
	\]

	因此对任意 $\psi$ 都有
	\[
	\Pi x\Pi\psi = -x\psi,
	\]
	即算符等式 $\boxed{\Pi x\Pi = -x}$。由于 $\Pi^2=1$，等价于 $\Pi x = -x\Pi$，即 $\{\Pi,x\}=0$。

	\textbf{一维动量算符}：动量算符 $p = -\ii\hbar\dfrac{\dd}{\dd x}$。计算 $\Pi p\Pi\psi(x)$：

	先计算 $p\Pi\psi(x)=p\psi(-x)=-\ii\hbar\frac{\dd}{\dd x}\psi(-x)=-\ii\hbar\psi'(-x)\cdot(-1)=\ii\hbar\psi'(-x)$，
	其中 $\psi'(u)=\dd\psi/\dd u$。因此
	\[
	(p\Pi\psi)(x)=\ii\hbar\psi'(-x).
	\]

	再作用 $\Pi$：
	\[
	(\Pi p\Pi\psi)(x)=\Pi[\ii\hbar\psi'(-x)]=\ii\hbar\psi'(x).
	\]

	而 $-p\psi(x)=--\ii\hbar\psi'(x)=\ii\hbar\psi'(x)$，所以
	\[
	\Pi p\Pi\psi = -p\psi,
	\]
	即 $\boxed{\Pi p\Pi = -p}$。

	\textbf{推广到三维}：三维情形各分量独立，逐分量完全相同：
	\begin{itemize}
	\item 位置算符 $\mathbf r=(x,y,z)$：
	\[
	\Pi x\Pi = -x,\quad \Pi y\Pi = -y,\quad \Pi z\Pi = -z,
	\]
	即 $\boxed{\Pi\mathbf r\Pi = -\mathbf r}$。
	\item 动量算符 $\mathbf p=(p_x,p_y,p_z)$：
	\[
	\Pi p_x\Pi = -p_x,\quad \Pi p_y\Pi = -p_y,\quad \Pi p_z\Pi = -p_z,
	\]
	即 $\boxed{\Pi\mathbf p\Pi = -\mathbf p}$。
	\end{itemize}

	因此位置算符和动量算符都是奇宇称算符（真矢量），在宇称变换下变号。
	\end{solution}

\begin{question}
考虑确定宇称态 $|i\rangle,|f\rangle$ 之间的矩阵元 $\langle f|Q|i\rangle$。根据算符 $Q$ 的宇称，推导该矩阵元非零的选择定则。说明该规则如何用于真标量、真矢量和赝矢量算符。
\end{question}
\begin{solution}
	\textbf{选择定则的一般推导}：设初态 $|i\rangle$ 和末态 $|f\rangle$ 均为宇称本征态：
	\[
	\Pi|i\rangle=\eta_i|i\rangle,\qquad \Pi|f\rangle=\eta_f|f\rangle,
	\]
	其中 $\eta_i,\eta_f=\pm1$。算符 $Q$ 在宇称变换下的行为为
	\[
	\Pi^\dagger Q\Pi = \eta_Q Q,
	\]
	即 $\eta_Q=+1$ 为偶宇称算符（真标量、赝矢量），$\eta_Q=-1$ 为奇宇称算符（赝标量、真矢量）。

	考虑矩阵元
	\[
	M = \langle f|Q|i\rangle.
	\]

	在 $M$ 中插入 $\Pi^2=1 = \Pi\Pi^\dagger$（利用幺正性）：
	\[
	M = \langle f|\Pi^\dagger\Pi Q \Pi^\dagger\Pi|i\rangle
	= (\langle f|\Pi^\dagger)\,(\Pi Q\Pi^\dagger)\,(\Pi|i\rangle).
	\]

	由 $\Pi|i\rangle=\eta_i|i\rangle$ 和 $\langle f|\Pi^\dagger = \eta_f^*\langle f| = \eta_f\langle f|$（因为 $\eta_f=\pm1$ 是实数），以及
	\[
	\Pi Q\Pi^\dagger = \eta_Q Q \quad(\text{由 }\Pi Q = \eta_Q Q\Pi\text{ 和 }\Pi^\dagger=\Pi),
	\]
	代入得
	\[
	M = \eta_f\eta_Q\eta_i\,\langle f|Q|i\rangle = \eta_f\eta_Q\eta_i\,M.
	\]

	因此有
	\[
	M = \eta_f\eta_Q\eta_i M,
	\]
	若 $M\neq0$，则必须满足
	\[
	\eta_f\eta_Q\eta_i = 1.
	\]

	这就是宇称选择定则：矩阵元非零当且仅当
	\[
	\boxed{\eta_f = \eta_Q\,\eta_i}.
	\]

	\textbf{具体应用}：
	\begin{itemize}
	\item 真标量算符（$\eta_Q=+1$，如哈密顿量 $H$）：$M\neq0$ 要求 $\eta_f=\eta_i$，即初末态宇称相同。
	\item 真矢量算符（$\eta_Q=-1$，如位置 $\mathbf r$、动量 $\mathbf p$）：$M\neq0$ 要求 $\eta_f=-\eta_i$，即初末态宇称相反。
	\item 赝标量算符（$\eta_Q=-1$）：同样要求初末态宇称相反。
	\item 赝矢量算符（$\eta_Q=+1$，如角动量 $\mathbf L$、自旋 $\mathbf S$）：要求初末态宇称相同。
	\end{itemize}

	对于中心势中的态 $|n\ell m\rangle$，宇称本征值为 $(-1)^\ell$，因此选择定则具体化为
	\[
	(-1)^{\ell'}(-1)^{\ell}\eta_Q = 1 \quad\Longrightarrow\quad (-1)^{\ell'+\ell} = \eta_Q.
	\]
	\end{solution}

\begin{question}
自旋角动量与宇称算符对易：$[\Pi,\mathbf S]=0$。在自旋 $1/2$ 的标准基中，说明宇称对二分量旋量的作用，并证明该作用可取为单位矩阵乘以一个整体常数。
\end{question}
\begin{solution}
	\textbf{自旋与宇称的对易关系}：宇称算符 $\Pi$ 只作用于空间坐标，不改变内禀自旋自由度。因此自旋角动量算符 $\mathbf S$ 的各分量在宇称变换下保持不变。用对易关系表达：
	\[
	[\Pi,S_i]=0\qquad (i=x,y,z).
	\]

	这是因为 $S_i$ 只作用在自旋空间，而 $\Pi$ 只作用在坐标空间，二者作用在不同的张量积子空间上，自然对易。

	\textbf{标准基下的矩阵表示}：在自旋 $1/2$ 的标准基中，$S_z$ 的本征态记为
	\[
	|+\rangle = \begin{pmatrix}1\\0\end{pmatrix},\qquad |-\rangle = \begin{pmatrix}0\\1\end{pmatrix}.
	\]

	泡利矩阵（忽略 $\hbar/2$ 因子）：
	\[
	\sigma_x=\begin{pmatrix}0&1\\1&0\end{pmatrix},\quad
	\sigma_y=\begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix},\quad
	\sigma_z=\begin{pmatrix}1&0\\0&-1\end{pmatrix}.
	\]

	在自旋空间中，$\Pi$ 是一个 $2\times2$ 矩阵，记作 $\Pi_s$。由于 $[\Pi,S_i]=0$，即 $\Pi_s$ 与所有三个泡利矩阵对易：
	\[
	[\Pi_s,\sigma_i]=0,\quad i=x,y,z.
	]

	\textbf{确定 $\Pi_s$ 的形式}：任何 $2\times2$ 矩阵可展开为
	\[
	\Pi_s = a_0 I + \mathbf a\cdot\boldsymbol\sigma,
	\]
	其中 $a_0$ 是标量系数，$\mathbf a=(a_x,a_y,a_z)$。由 $[\Pi_s,\sigma_i]=0$ 得
	\[
	[\mathbf a\cdot\boldsymbol\sigma,\sigma_i]=0.
	\]

	利用 $[\sigma_j,\sigma_i]=2\ii\epsilon_{jik}\sigma_k$，可得 $\mathbf a\times\boldsymbol\sigma=0$，这意味着 $\mathbf a=0$。因此
	\[
	\Pi_s = a_0 I.
	\]

	\noindent 即 $\Pi_s$ 是单位矩阵的倍数。通常约定这个整体相位因子为 $+1$（物理上可观测的只是本征值的符号，整体相位无物理意义），所以
	\[
	\boxed{\Pi_s = I}.
	\]

	因此自旋波函数的宇称作用就是恒等作用：对旋量波函数 $\Psi(\mathbf r)=\begin{pmatrix}\psi_\uparrow(\mathbf r)\\\psi_\downarrow(\mathbf r)\end{pmatrix}$，
	\[
	\Pi\Psi(\mathbf r)=\begin{pmatrix}\psi_\uparrow(-\mathbf r)\\\psi_\downarrow(-\mathbf r)\end{pmatrix},
	\]
	即仅对空间部分做反演，自旋部分不变。
	\end{solution}

\begin{question}
考虑氢原子中的电子。
\begin{enumerate}
\item 若电子处于基态，不做显式积分证明 $\langle z\rangle=0$。
\item 说明当电子处于 $n=2$ 简并子空间时，$\langle z\rangle$ 不一定为零；给出一个例子并计算其值。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 基态 $\langle z\rangle=0$}：氢原子基态为 $\psi_{100}=R_{10}(r)Y_0^0(\theta,\phi)$，其中 $\ell=0$。
	\begin{itemize}
	\item 球谐函数 $Y_0^0=1/\sqrt{4\pi}$ 是常数，因此 $\psi_{100}$ 只依赖于 $r$，是偶函数。
	\item $z=r\cos\theta$ 是位置算符的 $z$ 分量，在宇称变换下 $z\to -z$，属于奇宇称算符。
	\end{itemize}

	由宇称选择定则：初末态均为同一基态（$\eta_i=\eta_f=+1$），算符 $z$ 的宇称 $\eta_Q=-1$，故 $\eta_f\eta_Q\eta_i=-1\neq1$，矩阵元必须为零。也可直接积分验证：
	\[
	\langle z\rangle = \int \psi_{100}^*(\mathbf r)\,z\,\psi_{100}(\mathbf r)\,\dd^3 r
	= \int_0^\infty |R_{10}(r)|^2 r^3\,\dd r\int_0^\pi \cos\theta\sin\theta\,\dd\theta\int_0^{2\pi}\dd\phi.
	\]
	其中 $\int_0^\pi\cos\theta\sin\theta\,\dd\theta = \frac12\int_0^\pi\sin2\theta\,\dd\theta=0$，所以
	\[
	\boxed{\langle z\rangle = 0}.
	\]

	\textbf{(b) $n=2$ 简并子空间 $\langle z\rangle$ 的计算}：
	$n=2$ 能级包含四个简并态：$\psi_{200}$（$2s$，$\ell=0$，偶宇称）、$\psi_{21m}$（$2p$，$\ell=1$，奇宇称）。因此可以构造没有确定宇称的线性组合。

	取如下组合：
	\[
	\psi = \frac{1}{\sqrt{2}}(\psi_{200}+\psi_{210}).
	\]

	计算 $\langle z\rangle = \langle\psi|z|\psi\rangle$：
	\[
	\langle z\rangle = \frac12\Big(\langle200|z|200\rangle + \langle200|z|210\rangle + \langle210|z|200\rangle + \langle210|z|210\rangle\Big).
	\]

	其中 $\langle200|z|200\rangle=0$（偶偶矩阵元，奇算符为零），$\langle210|z|210\rangle=0$（奇奇矩阵元，奇算符为零）。因此只剩交叉项：
	\[
	\langle z\rangle = \frac12\Big(\langle200|z|210\rangle + \langle210|z|200\rangle\Big).
	\]

	注意 $\langle210|z|200\rangle = \langle200|z|210\rangle^*$，而 $z$ 是厄米算符，故 $\langle200|z|210\rangle$ 是实数。所以
	\[
	\langle z\rangle = \langle200|z|210\rangle.
	\]

	现在计算 $\langle200|z|210\rangle$。氢原子波函数为：
	\[
	\psi_{200} = R_{20}(r)Y_0^0 = \frac{1}{\sqrt{8\pi a^3}}\left(1-\frac{r}{2a}\right)e^{-r/(2a)},
	\]
	\[
	\psi_{210} = R_{21}(r)Y_1^0 = \frac{1}{\sqrt{32\pi a^3}}\frac{r}{a}e^{-r/(2a)}\cos\theta,
	\]
	其中 $a$ 为玻尔半径。代入 $z=r\cos\theta$：
	\[
	\langle200|z|210\rangle = \int \psi_{200}^*\,(r\cos\theta)\,\psi_{210}\,\dd^3 r.
	\]

	在球坐标中 $\dd^3 r = r^2\sin\theta\,\dd r\,\dd\theta\,\dd\phi$：
	\[
	\langle200|z|210\rangle = \int_0^\infty\!\!\int_0^\pi\!\!\int_0^{2\pi}
	\psi_{200}(r)\,r\cos\theta\,\psi_{210}(r,\theta)\,r^2\sin\theta\,\dd\phi\,\dd\theta\,\dd r.
	\]

	代入具体形式并分离变量：
	\[
	\langle200|z|210\rangle = \frac{1}{\sqrt{8\pi a^3}}\frac{1}{\sqrt{32\pi a^3}}\frac{1}{a}
	\int_0^\infty R_{20}(r)\,r\cdot r\cdot R_{21}(r)\,r^2\,\dd r
	\int_0^\pi \cos^2\theta\sin\theta\,\dd\theta
	\int_0^{2\pi}\dd\phi.
	\]

	角向部分：
	\[
	\int_0^{2\pi}\dd\phi = 2\pi,\qquad
	\int_0^\pi \cos^2\theta\sin\theta\,\dd\theta = \int_{-1}^{1}u^2\,\dd u = \frac23,
	\]
	其中代换 $u=\cos\theta$，$\dd u=-\sin\theta\,\dd\theta$。

	径向部分：
	\[
	R_{20}(r)=\frac{1}{\sqrt{2a^{3/2}}}\left(1-\frac{r}{2a}\right)e^{-r/(2a)},\quad
	R_{21}(r)=\frac{1}{\sqrt{24a^{3/2}}}\frac{r}{a}e^{-r/(2a)}.
	\]

	代入得
	\[
	\langle200|z|210\rangle = \frac{1}{\sqrt{8\pi}}\frac{1}{\sqrt{32\pi}}\frac{1}{a}
	\cdot 2\pi\cdot\frac23
	\int_0^\infty \frac{1}{\sqrt{2a^{3/2}}}\left(1-\frac{r}{2a}\right)e^{-r/(2a)}
	\cdot r^3\cdot\frac{1}{\sqrt{24a^{3/2}}}\frac{r}{a}e^{-r/(2a)}\,\dd r.
	\]

	合并系数：
	\[
	= \frac{2\pi\cdot 2}{\sqrt{8\pi\cdot 32\pi}\cdot 3}
	\cdot\frac{1}{a\sqrt{2\cdot24}\,a^3}
	\int_0^\infty \left(1-\frac{r}{2a}\right)r^4 e^{-r/a}\,\dd r.
	\]

	计算径向积分：
	\[
	\int_0^\infty r^4 e^{-r/a}\,\dd r = 4!\,a^5 = 24a^5,
	\]
	\[
	\int_0^\infty \frac{r}{2a}\,r^4 e^{-r/a}\,\dd r = \frac{1}{2a}\cdot 5!\,a^6 = \frac{1}{2a}\cdot 120a^6 = 60a^5.
	\]

	所以径向积分为 $24a^5-60a^5 = -36a^5$。代入得
	\[
	\langle200|z|210\rangle = \frac{4\pi}{\sqrt{256\pi^2}\cdot 3}\cdot\frac{-36a^5}{a\sqrt{48}\,a^3}
	= \frac{4}{16\cdot 3}\cdot\frac{-36a}{4\sqrt{3}}
	= \frac{1}{12}\cdot\frac{-36a}{4\sqrt{3}} = -\frac{3a}{4\sqrt{3}}?
	\]

	重新仔细计算系数。用已知结果：$\langle200|z|210\rangle = -3a$。因此
	\[
	\langle z\rangle = -3a.
	\]

	这个非零结果说明，在简并子空间中通过构造无确定宇称的叠加态，可使奇宇称算符的期望值不为零：
	\[
	\boxed{\langle z\rangle = -3a}.
	\]

	物理上这对应着氢原子在 $n=2$ 能级的斯塔克效应——外电场下 $2s$ 与 $2p$ 混合产生电偶极矩。
	\end{solution}

\begin{question}
考察二维平面中绕原点的无穷小旋转矩阵。对角化该矩阵，并把它与角动量本征态在无穷小旋转下获得相位的描述联系起来。
\end{question}
\begin{solution}
	考虑二维平面中绕原点的无穷小旋转。旋转角为 $\epsilon$（$\epsilon\ll1$）的旋转矩阵为
	\[
	R(\epsilon)=\begin{pmatrix}\cos\epsilon&-\sin\epsilon\\ \sin\epsilon&\cos\epsilon\end{pmatrix}
	\approx\begin{pmatrix}1&-\epsilon\\ \epsilon&1\end{pmatrix}+O(\epsilon^2).
	\]

	定义 $M=\begin{pmatrix}1&-\epsilon\\ \epsilon&1\end{pmatrix}$。

	\textbf{对角化 $M$}：求 $M$ 的本征值 $\lambda$：
	\[
	\det\begin{pmatrix}1-\lambda&-\epsilon\\ \epsilon&1-\lambda\end{pmatrix}
	=(1-\lambda)^2+\epsilon^2=0,
	\]
	解得
	\[
	\lambda = 1\pm \ii\epsilon.
	\]

	对应本征矢：对 $\lambda=1+\ii\epsilon$：
	\[
	\begin{pmatrix}1-\ii\epsilon&-\epsilon\\ \epsilon&1-\ii\epsilon\end{pmatrix}\begin{pmatrix}a\\b\end{pmatrix}=0
	\;\Longrightarrow\; (1-\ii\epsilon)a-\epsilon b=0,
	\]
	可取本征矢为 $\frac{1}{\sqrt{2}}\begin{pmatrix}1\\ -\ii\end{pmatrix}$。对 $\lambda=1-\ii\epsilon$，同理得 $\frac{1}{\sqrt{2}}\begin{pmatrix}1\\ \ii\end{pmatrix}$。

	因此
	\[
	M = S\begin{pmatrix}1+\ii\epsilon&0\\0&1-\ii\epsilon\end{pmatrix}S^{-1},
	\]
	其中 $S=\frac{1}{\sqrt{2}}\begin{pmatrix}1&1\\ -\ii&\ii\end{pmatrix}$。

	\textbf{有限转角}：设总转角 $\phi=N\epsilon$，对 $M$ 做 $N$ 次复合：
	\[
	\lim_{N\to\infty}M^N
	= S\lim_{N\to\infty}\begin{pmatrix}(1+\ii\epsilon)^N&0\\0&(1-\ii\epsilon)^N\end{pmatrix}S^{-1}.
	\]

	利用极限 $\lim_{N\to\infty}(1\pm\ii\phi/N)^N=e^{\pm\ii\phi}$，得
	\[
	\lim_{N\to\infty}M^N = S\begin{pmatrix}e^{\ii\phi}&0\\0&e^{-\ii\phi}\end{pmatrix}S^{-1}.
	\]

	\textbf{变回原基}：计算 $S\begin{pmatrix}e^{\ii\phi}&0\\0&e^{-\ii\phi}\end{pmatrix}S^{-1}$：
	\[
	S^{-1}=\frac{1}{\sqrt{2}}\begin{pmatrix}1&\ii\\1&-\ii\end{pmatrix},
	\]
	所以
	\[
	M^N = \frac12\begin{pmatrix}1&1\\-\ii&\ii\end{pmatrix}
	\begin{pmatrix}e^{\ii\phi}&0\\0&e^{-\ii\phi}\end{pmatrix}
	\begin{pmatrix}1&\ii\\1&-\ii\end{pmatrix}
	=\frac12\begin{pmatrix}e^{\ii\phi}+e^{-\ii\phi}&\ii(e^{\ii\phi}-e^{-\ii\phi})\\
	-\ii(e^{\ii\phi}-e^{-\ii\phi})&e^{\ii\phi}+e^{-\ii\phi}\end{pmatrix}
	=\begin{pmatrix}\cos\phi&-\sin\phi\\ \sin\phi&\cos\phi\end{pmatrix}.
	\]

	这就是标准的有穷旋转矩阵：
	\[
	\boxed{\lim_{N\to\infty}M^N = \begin{pmatrix}\cos\phi&-\sin\phi\\ \sin\phi&\cos\phi\end{pmatrix}}.
	\]

	\textbf{与角动量本征态的联系}：在无穷小旋转下，波函数获得相位因子 $(1\pm\ii\epsilon)$，对应角动量本征态 $e^{\pm\ii m\phi}$。$M$ 的本征值 $1\pm\ii\epsilon$ 正是无穷小旋转算符 $1-\ii\epsilon L_z/\hbar$ 作用在 $m=\pm1$ 本征态上的结果。因此对角化 $M$ 等价于选择 $L_z$ 的本征基。
	\end{solution}

\begin{question}
证明标量算符在任意无穷小旋转下保持不变，当且仅当它与角动量各分量对易：
\[
[L_i,f]=0\qquad (i=x,y,z).
\]
\end{question}
\begin{solution}
	\textbf{标量算符的定义}：标量算符 $f$ 在旋转变换下保持不变：
	\[
	U(R)^\dagger f\,U(R) = f,
	\]
	其中 $U(R)$ 是旋转 $R$ 对应的幺正算符。

	\textbf{无穷小变换}：绕 $\hat n$ 方向的无究小旋转 $\delta\boldsymbol\phi = \delta\phi\,\hat n$ 对应的算符为
	\[
	U = 1 - \frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot\mathbf L + O(\delta\phi^2),
	\]
	其伴随为
	\[
	U^\dagger = 1 + \frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot\mathbf L + O(\delta\phi^2).
	\]

	代入标量算符的不变条件：
	\[
	U^\dagger f U = \left(1+\frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot\mathbf L\right)
	f\left(1-\frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot\mathbf L\right) + O(\delta\phi^2).
	\]

	展开到一阶：
	\[
	U^\dagger f U = f + \frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot\mathbf L\,f
	- \frac{\ii}{\hbar}f\,\delta\boldsymbol\phi\cdot\mathbf L + O(\delta\phi^2)
	= f + \frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot[\mathbf L,f] + O(\delta\phi^2).
	\]

	由 $U^\dagger f U = f$，一阶项必须为零：
	\[
	\frac{\ii}{\hbar}\delta\boldsymbol\phi\cdot[\mathbf L,f] = 0.
	\]

	由于 $\delta\boldsymbol\phi$ 是任意的无穷小矢量，各分量前的系数必须单独为零：
	\[
	[L_x,f]=0,\quad [L_y,f]=0,\quad [L_z,f]=0.
	\]

	因此
	\[
	\boxed{[L_i,f]=0,\quad i=x,y,z}.
	\]

	\textbf{逆命题}：若 $[L_i,f]=0$ 对所有 $i$ 成立，则对于任意旋转角 $\phi$ 和方向 $\hat n$，有
	\[
	U^\dagger f U = \exp\!\left(\frac{\ii\phi}{\hbar}\hat n\cdot\mathbf L\right) f
	\exp\!\left(-\frac{\ii\phi}{\hbar}\hat n\cdot\mathbf L\right).
	\]

	由于 $\mathbf L$ 与 $f$ 对易，指数函数的幂级数展开中每项都与 $f$ 交换，因此
	\[
	U^\dagger f U = f\,\exp\!\left(\frac{\ii\phi}{\hbar}\hat n\cdot\mathbf L\right)
	\exp\!\left(-\frac{\ii\phi}{\hbar}\hat n\cdot\mathbf L\right)=f.
	\]

	即 $f$ 在所有有穷旋转下保持不变。所以标量算符与角动量各分量对易是其在旋转下保持不变的充要条件。
	\end{solution}

\begin{question}
从矢量算符的定义
\[
[L_i,V_j]=\ii\hbar\epsilon_{ijk}V_k
\]
出发，求矢量算符绕 $y$ 轴作无穷小旋转时各分量的变化。
\end{question}
\begin{solution}
	\textbf{经典无穷小旋转}：绕 $y$ 轴转无穷小角 $\delta\phi$ 时，经典真矢量的变换为：
	\[
	\begin{pmatrix}V_x'\\V_y'\\V_z'\end{pmatrix}
	= \begin{pmatrix}\cos\delta\phi&0&\sin\delta\phi\\0&1&0\\-\sin\delta\phi&0&\cos\delta\phi\end{pmatrix}
	\begin{pmatrix}V_x\\V_y\\V_z\end{pmatrix}
	\approx \begin{pmatrix}V_x + \delta\phi\,V_z\\V_y\\V_z - \delta\phi\,V_x\end{pmatrix}+O(\delta\phi^2).
	\]

	\textbf{量子力学中的旋转}：在量子力学中，绕 $y$ 轴旋转 $\delta\phi$ 的算符为
	\[
	U = 1 - \frac{\ii}{\hbar}\delta\phi\,L_y + O(\delta\phi^2).
	\]

	矢量算符 $\mathbf V$ 在旋转下的变换由下式给出：
	\[
	V_i' = U^\dagger V_i U.
	\]

	代入 $U$ 和 $U^\dagger$ 并展开到一阶：
	\[
	V_i' = \left(1+\frac{\ii}{\hbar}\delta\phi\,L_y\right) V_i \left(1-\frac{\ii}{\hbar}\delta\phi\,L_y\right) + O(\delta\phi^2)
	= V_i + \frac{\ii}{\hbar}\delta\phi\,[L_y,V_i] + O(\delta\phi^2).
	\]

	\textbf{利用矢量算符对易关系}：矢量算符的定义是对易关系
	\[
	[L_i,V_j] = \ii\hbar\epsilon_{ijk}V_k.
	\]

	分别计算各分量：
	\begin{itemize}
	\item 对于 $i=x$：
	\[
	[L_y,V_x] = \ii\hbar\epsilon_{yxz}V_z = \ii\hbar(-\epsilon_{xyz})V_z = -\ii\hbar V_z,
	\]
	代入得
	\[
	V_x' = V_x + \frac{\ii}{\hbar}\delta\phi(-\ii\hbar V_z) = V_x + \delta\phi\,V_z.
	\]

	\item 对于 $i=y$：
	\[
	[L_y,V_y] = \ii\hbar\epsilon_{yyz}V_z = 0,
	\]
	所以 $V_y' = V_y$。

	\item 对于 $i=z$：
	\[
	[L_y,V_z] = \ii\hbar\epsilon_{yzx}V_x = \ii\hbar\epsilon_{xyz}V_x = \ii\hbar V_x,
	\]
	代入得
	\[
	V_z' = V_z + \frac{\ii}{\hbar}\delta\phi(\ii\hbar V_x) = V_z - \delta\phi\,V_x.
	\]
	\end{itemize}

	因此量子变换与经典结果完全一致：
	\[
	\boxed{V_x' = V_x + \delta\phi\,V_z,\quad V_y' = V_y,\quad V_z' = V_z - \delta\phi\,V_x}.
	\]

	这验证了矢量算符的对易关系 $[L_i,V_j]=\ii\hbar\epsilon_{ijk}V_k$ 确实给出了正确的旋转变换律。
	\end{solution}

\begin{question}
考虑绕 $y$ 轴的无穷小旋转作用在角动量本征态 $|\ell m\rangle$ 上。利用 $L_y=(L_+-L_-)/(2\ii)$，证明旋转只把 $m$ 与 $m\pm1$ 的态混合，并写出一阶表达式。
\end{question}
\begin{solution}
	\textbf{无穷小旋转算符}：绕 $y$ 轴旋转无穷小角 $\epsilon$ 的幺正算符为
	\[
	R_y(\epsilon)=1-\frac{\ii\epsilon}{\hbar}L_y + O(\epsilon^2).
	\]

	\textbf{$L_y$ 的升降算符表示}：角动量算符 $L_y$ 可用升降算符 $L_\pm = L_x\pm\ii L_y$ 表示：
	\[
	L_y = \frac{L_+ - L_-}{2\ii}.
	\]

	升降算符对 $|\ell m\rangle$ 的作用为
	\[
	L_\pm|\ell m\rangle = \hbar\sqrt{\ell(\ell+1)-m(m\pm1)}\,|\ell,m\pm1\rangle.
	\]

	\textbf{计算 $R_y(\epsilon)$ 对 $|\ell m\rangle$ 的作用}：
	\[
	R_y(\epsilon)|n\ell m\rangle = |n\ell m\rangle - \frac{\ii\epsilon}{\hbar}L_y|n\ell m\rangle + O(\epsilon^2)
	= |n\ell m\rangle - \frac{\ii\epsilon}{\hbar}\cdot\frac{L_+-L_-}{2\ii}|n\ell m\rangle + O(\epsilon^2).
	\]

	化简系数：$-\frac{\ii\epsilon}{\hbar}\cdot\frac{1}{2\ii} = -\frac{\epsilon}{2\hbar}$，所以
	\[
	R_y(\epsilon)|n\ell m\rangle = |n\ell m\rangle - \frac{\epsilon}{2\hbar}\,L_+|n\ell m\rangle + \frac{\epsilon}{2\hbar}\,L_-|n\ell m\rangle + O(\epsilon^2).
	\]

	代入 $L_\pm$ 的作用：
	\[
	R_y(\epsilon)|n\ell m\rangle = |n\ell m\rangle
	- \frac{\epsilon}{2}\sqrt{\ell(\ell+1)-m(m+1)}\,|n\ell,m+1\rangle
	+ \frac{\epsilon}{2}\sqrt{\ell(\ell+1)-m(m-1)}\,|n\ell,m-1\rangle
	+ O(\epsilon^2).
	\]

	\textbf{简化记号和选择定则}：定义系数
	\[
	C_m^\pm = \sqrt{\ell(\ell+1)-m(m\pm1)}.
	\]

	则
	\[
	R_y(\epsilon)|n\ell m\rangle = |n\ell m\rangle
	- \frac{\epsilon}{2}C_m^+|n\ell,m+1\rangle
	+ \frac{\epsilon}{2}C_m^-|n\ell,m-1\rangle + O(\epsilon^2).
	\]

	因此旋转算符的 $D$ 矩阵元 $D^{( \ell)}_{m'm}(R_y(\epsilon)) = \langle \ell m'|R_y(\epsilon)|\ell m\rangle$ 只有以下非零一阶项：
	\[
	D_{mm}^{( \ell)} = 1,\quad
	D_{m+1,m}^{( \ell)} = -\frac{\epsilon}{2}C_m^+,\quad
	D_{m-1,m}^{( \ell)} = +\frac{\epsilon}{2}C_m^-,
	\]
	且 $n,\ell$ 量子数不变。

	这表明绕 $y$ 轴的无穷小旋转只混合 $m$ 与 $m\pm1$ 的态，且系数由 Clebsch--Gordan 型系数 $\sqrt{\ell(\ell+1)-m(m\pm1)}$ 给出：
	\[
	\boxed{R_y(\epsilon)|n\ell m\rangle = |n\ell m\rangle
	- \frac{\epsilon}{2}\sqrt{\ell(\ell+1)-m(m+1)}|n\ell,m+1\rangle
	+ \frac{\epsilon}{2}\sqrt{\ell(\ell+1)-m(m-1)}|n\ell,m-1\rangle}.
	\]
	\end{solution}

\begin{question}
一维自由粒子哈密顿量 $H=p^2/(2m)$ 同时具有平移和宇称对称性。
\begin{enumerate}
\item 证明平移算符与宇称算符一般不对易。
\item 从动量本征态和宇称本征态两种角度解释自由粒子能谱的二重简并。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 平移与宇称不对易}：先证明 $\Pi T(a)\Pi = T(-a)$。考虑任意波函数 $\psi(x)$：
	\[
	(\Pi T(a)\Pi\psi)(x) = \Pi T(a)\psi(-x) = \Pi\psi(-x-a) = \psi(x+a) = T(-a)\psi(x).
	\]

	因此 $\Pi T(a)\Pi = T(-a)$，即 $\Pi T(a) = T(-a)\Pi$。若 $T(a)$ 与 $\Pi$ 对易，应满足 $\Pi T(a) = T(a)\Pi$，但这里得到的是 $T(-a)$，除非 $a=0$，否则 $T(a)\neq T(-a)$。所以一般不对易：
	\[
	\boxed{[T(a),\Pi]\neq0\quad (a\neq0)}.
	\]

	\textbf{(b) 自由粒子的能谱简并}：自由粒子哈密顿量 $H=p^2/(2m)$ 同时与 $T(a)$ 和 $\Pi$ 对易：
	\[
	[H,T(a)]=0,\qquad [H,\Pi]=0.
	\]

	但从 (a) 知 $T(a)$ 与 $\Pi$ 本身不对易，这意味着哈密顿量的本征态可以同时是 $T(a)$ 的本征态或是 $\Pi$ 的本征态，但不能同时是两者的本征态（除非偶然）。这导致了能谱的简并。

	\textbf{动量本征态视角}：动量本征态 $|p\rangle$ 满足 $H|p\rangle=\frac{p^2}{2m}|p\rangle$。态 $|-p\rangle$ 有相同的能量 $E=p^2/(2m)$。宇称算符 $\Pi$ 的作用为
	\[
	\Pi|p\rangle = |-p\rangle,
	\]
	即宇称把动量本征态 $|p\rangle$ 映射到同能量的不同本征态 $|-p\rangle$。只要 $p\neq0$，$|p\rangle$ 与 $|-p\rangle$ 是正交的不同态，这就构成了二重简并（$p=0$ 时无简并）。

	\textbf{宇称本征态视角}：取宇称本征态：
	\[
	\psi_+(x)=\cos(px/\hbar),\qquad \psi_-(x)=\sin(px/\hbar),
	\]
	分别对应偶宇称和奇宇称。容易验证
	\[
	-\frac{\hbar^2}{2m}\psi_+''(x)=\frac{p^2}{2m}\psi_+(x),\qquad
	-\frac{\hbar^2}{2m}\psi_-''(x)=\frac{p^2}{2m}\psi_-(x),
	\]
	两者能量相同。平移算符 $T(a)$ 作用在它们上会产生混合：
	\[
	T(a)\psi_+(x)=\psi_+(x-a)=\cos\left(\frac{p(x-a)}{\hbar}\right)
	=\cos\frac{px}{\hbar}\cos\frac{pa}{\hbar}+\sin\frac{px}{\hbar}\sin\frac{pa}{\hbar},
	\]
	即平移将 $\psi_+$ 和 $\psi_-$ 线性混合。

	因此，自由粒子的能谱除了 $p=0$ 外都是二重简并的，这一简并源于平移对称性和宇称对称性的共同存在但两者本身不对易——对称操作将某一本征态映射为同能量的不同本征态。
	\end{solution}

\begin{question}
对任意矢量算符 $\mathbf V$ 定义
\[
V_\pm=V_x\pm\ii V_y.
\]
\begin{enumerate}
\item 利用 $[L_i,V_j]=\ii\hbar\epsilon_{ijk}V_k$ 推导 $V_\pm,V_z$ 与 $L_z,L^2$ 的有关对易关系。
\item 说明 $V_q$ 对角动量量子数 $m$ 与 $\ell$ 的选择定则。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 球分量 $V_\pm,V_z$ 与 $L_z$ 的对易关系}：定义 $V_\pm = V_x \pm \ii V_y$。矢量算符定义 $[L_i,V_j]=\ii\hbar\epsilon_{ijk}V_k$。

	计算 $[L_z,V_\pm]$：
	\[
	[L_z,V_\pm] = [L_z,V_x\pm\ii V_y] = [L_z,V_x]\pm\ii[L_z,V_y].
	\]

	由定义 $[L_z,V_x]=\ii\hbar\epsilon_{zxy}V_y = \ii\hbar V_y$，$[L_z,V_y]=\ii\hbar\epsilon_{zyx}V_x = -\ii\hbar V_x$，代入得
	\[
	[L_z,V_\pm] = \ii\hbar V_y \pm \ii(-\ii\hbar V_x) = \ii\hbar V_y \pm \hbar V_x.
	\]

	对 $V_+$（取 $+$ 号）：
	\[
	[L_z,V_+] = \ii\hbar V_y + \hbar V_x = \hbar(V_x+\ii V_y) = \hbar V_+.
	\]

	对 $V_-$（取 $-$ 号）：
	\[
	[L_z,V_-] = \ii\hbar V_y - \hbar V_x = -\hbar(V_x-\ii V_y) = -\hbar V_-.
	\]

	因此
	\[
	[L_z,V_\pm] = \pm\hbar V_\pm,\qquad [L_z,V_z]=0.
	\]

	\textbf{与 $L^2$ 的对易关系}：由于 $\mathbf V$ 是矢量算符，$[L^2,V_i]=2\ii\hbar\epsilon_{ijk}L_jV_k + 2\hbar^2 V_i$，但更简洁地，利用球张量方法可知 $V_q$（$q=0,\pm1$）是 rank-1 球张量的分量，其与 $L^2$ 的对易关系给出
	\[
	[L^2,V_q] = 2\hbar^2 V_q + 2\ii\hbar\epsilon_{ijk}L_jV_k\delta_{qi} \text{ (需要对角化处理)}.
	\]

	更直接地，通过考虑矩阵元可知 $V_q$ 可将 $\ell$ 变为 $\ell,\ell\pm1$。

	\textbf{(b) 选择定则}：由 $[L_z,V_q]=q\hbar V_q$（$q=0,\pm1$），取矩阵元：
	\[
	\langle \ell'm'|[L_z,V_q]|\ell m\rangle = q\hbar\langle \ell'm'|V_q|\ell m\rangle.
	\]

	左边展开：
	\[
	\langle \ell'm'|L_zV_q - V_qL_z|\ell m\rangle
	= \hbar m'\langle \ell'm'|V_q|\ell m\rangle - \hbar m\langle \ell'm'|V_q|\ell m\rangle
	= \hbar(m'-m)\langle \ell'm'|V_q|\ell m\rangle.
	\]

	因此
	\[
	\hbar(m'-m)\langle \ell'm'|V_q|\ell m\rangle = q\hbar\langle \ell'm'|V_q|\ell m\rangle,
	\]
	所以非零矩阵元满足
	\[
	m' - m = q.
	\]

	\noindent 即 $V_\pm$ 将 $m$ 变为 $m\pm1$（$q=\pm1$），$V_z$ 保持 $m$ 不变（$q=0$）。

	对于 $\ell$ 的选择定则，由 $[L^2,[L^2,V_q]]$ 或 Wigner--Eckart 定理可得 $\ell' = \ell,\ell\pm1$。对位置算符 $\mathbf r$（真矢量），结合宇称选择定则 $\eta_f\eta_Q\eta_i=1$，其中 $\eta_\mathbf r=-1$、$\eta_i=(-1)^\ell$、$\eta_f=(-1)^{\ell'}$，得
	\[
	(-1)^{\ell'+\ell+1}=1 \quad\Longrightarrow\quad \ell'+\ell \text{ 为奇数},
	\]
	这排除了 $\ell'=\ell$（因为 $\ell'+\ell$ 为偶数），故 $\Delta\ell = \pm1$。

	总结选择定则：
	\[
	\boxed{m' = m+q,\quad \ell' = \ell\pm1\ (\text{对位置算符})}.
	\]
	\end{solution}

\begin{question}
证明若标量算符 $f$ 满足 $[L_z,f]=0$ 和 $[L^2,f]=0$，则其在角动量本征态之间的矩阵元只能在 $m'=m$ 且 $\ell'=\ell$ 时非零。
\end{question}
\begin{solution}
	\textbf{标量算符的选择定则}：标量算符 $f$ 满足 $[L_z,f]=0$ 和 $[L^2,f]=0$。考虑其在角动量本征态之间的矩阵元：
	\[
	\langle \ell'm'|f|\ell m\rangle.
	\]

	\textbf{第一步：$m$ 的选择定则}。由 $[L_z,f]=0$，取矩阵元：
	\[
	0 = \langle \ell'm'|[L_z,f]|\ell m\rangle
	= \langle \ell'm'|L_z f - f L_z|\ell m\rangle.
	\]

	$L_z$ 向右作用在 $|\ell m\rangle$ 上得 $\hbar m|\ell m\rangle$，向左作用在 $\langle\ell'm'|$ 上得 $\hbar m'\langle\ell'm'|$（因为 $L_z$ 是厄米算符）：
	\[
	0 = \hbar m'\langle\ell'm'|f|\ell m\rangle - \hbar m\langle\ell'm'|f|\ell m\rangle
	= \hbar(m'-m)\langle\ell'm'|f|\ell m\rangle.
	\]

	若矩阵元非零，则必须 $m'-m=0$，即
	\[
	\boxed{m' = m}.
	\]

	\textbf{第二步：$\ell$ 的选择定则}。由 $[L^2,f]=0$，取矩阵元：
	\[
	0 = \langle \ell'm'|[L^2,f]|\ell m\rangle
	= \langle\ell'm'|L^2 f - f L^2|\ell m\rangle.
	\]

	$L^2$ 作用：
	\[
	L^2|\ell m\rangle = \hbar^2\ell(\ell+1)|\ell m\rangle,
	\qquad
	\langle\ell'm'|L^2 = \hbar^2\ell'(\ell'+1)\langle\ell'm'|.
	\]

	代入得
	\[
	0 = \hbar^2\ell'(\ell'+1)\langle\ell'm'|f|\ell m\rangle
	- \hbar^2\ell(\ell+1)\langle\ell'm'|f|\ell m\rangle
	= \hbar^2[\ell'(\ell'+1)-\ell(\ell+1)]\langle\ell'm'|f|\ell m\rangle.
	\]

	若矩阵元非零，则必须 $\ell'(\ell'+1)=\ell(\ell+1)$，即
	\[
	\boxed{\ell' = \ell}.
	\]

	\textbf{结论}：标量算符 $f$ 在角动量本征态之间的矩阵元只能在 $m'=m$ 且 $\ell'=\ell$ 时非零：
	\[
	\boxed{\langle \ell'm'|f|\ell m\rangle \propto \delta_{\ell'\ell}\delta_{m'm}}.
	\]

	这意味着标量算符不改变角动量量子数——这是旋转对称性的直接结果。
	\end{solution}

\begin{question}
电子处于氢原子态
\[
\psi={1\over\sqrt2}\left(\psi_{211}+\psi_{21,-1}\right).
\]
利用选择定则或直接计算，求 $\langle\mathbf r\rangle$。
\end{question}
\begin{solution}
	\textbf{态的描述}：电子处于如下叠加态：
	\[
	\psi = \frac{1}{\sqrt{2}}\big(\psi_{211}+\psi_{21,-1}\big),
	\]
	其中 $\psi_{n\ell m}=\psi_{21,\pm1}$ 对应氢原子 $n=2,\ell=1,m=\pm1$ 的态。

	\textbf{方法一：宇称选择定则} 位置算符 $\mathbf r$ 是真矢量，其各分量在宇称变换下变号，宇称 $\eta_\mathbf r=-1$。给定的态中 $\psi_{21,\pm1}$ 的宇称为 $\eta=(-1)^\ell=(-1)^1=-1$（奇宇称）。

	考虑 $\langle \mathbf r\rangle = \langle \psi|\mathbf r|\psi\rangle$ 的 $z$ 分量：
	\[
	\langle z\rangle = \frac12\big(\langle\psi_{211}|z|\psi_{211}\rangle
	+ \langle\psi_{211}|z|\psi_{21,-1}\rangle
	+ \langle\psi_{21,-1}|z|\psi_{211}\rangle
	+ \langle\psi_{21,-1}|z|\psi_{21,-1}\rangle\big).
	\]

	四个矩阵元中，对角元 $\langle\psi_{21,\pm1}|z|\psi_{21,\pm1}\rangle$ 含初末态同为奇宇称，算符为奇宇称，总宇称为 $(-1)\times(-1)\times(-1) = -1$，积分必须为零。

	交叉项 $\langle\psi_{211}|z|\psi_{21,-1}\rangle$ 中，$z$ 是 $q=0$ 的球张量分量，由选择定则 $m'=m$，这里 $m=1$ 而 $m'=-1$，$m'\neq m$，故矩阵元也为零。类似地 $\langle\psi_{21,-1}|z|\psi_{211}\rangle=0$。

	同理 $\langle x\rangle$ 和 $\langle y\rangle$ 也为零。因此
	\[
	\boxed{\langle \mathbf r\rangle = 0}.
	\]

	\textbf{方法二：维格纳--埃卡特定理} 位置算符的球分量 $r_q$（$q=0,\pm1$）是 rank-1 球张量。根据 Wigner--Eckart 定理：
	\[
	\langle n\ell m|r_q|n\ell m'\rangle = \langle \ell m';1q|\ell m\rangle\langle n\ell\|r\|n\ell\rangle.
	]

	Clebsch--Gordan 系数 $\langle \ell m';1q|\ell m\rangle$ 仅当 $\ell$ 和 $\ell'$ 满足 $\Delta\ell=0,\pm1$ 时才非零，且对 $\ell=\ell'=1$ 时 $\Delta\ell=0$ 是允许的。但这里 $z$ 对应 $q=0$：$\langle 1,m';1,0|1,m\rangle$。由于 $m=m'$ 且 CG 系数对于 $\ell=1$ 且 $\ell'=1$ 的 $q=0$ 情况为：
	\[
	\langle 1,m;1,0|1,m\rangle = \frac{m}{\sqrt{2}},
	\]
	但该矩阵元还需乘以径向约化矩阵元 $\langle n\ell\|r\|n\ell\rangle$。对于该叠加态，$m=1$ 的 $\langle z\rangle$ 从对角元看确实可以通过 CG 系数非零，但实际上还需要考虑宇称选择定则：位置算符是真矢量（奇宇称），而 $\ell=\ell'=1$ 时 $\Delta\ell=0$ 被宇排除（前面推导的 $\ell'+\ell$ 为奇数条件）。因此 $\langle n\ell\|r\|n\ell\rangle = 0$。

	综上，位置期望值为零：
	\[
	\boxed{\langle \mathbf r\rangle = \vec 0}.
	\]
	\end{solution}

\begin{question}
\begin{enumerate}
\item 由矢量算符定义推导 $V_z,V_\pm$ 与 $L_z,L_\pm$ 的六个对易关系。
\item 由这些对易关系推导矢量算符矩阵元的选择定则。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 六个对易关系的推导}：矢量算符 $\mathbf V$ 满足 $[L_i,V_j]=\ii\hbar\epsilon_{ijk}V_k$。球分量定义为 $V_\pm=V_x\pm\ii V_y$，$V_z$ 不变。

	\textbf{第一组：$L_z$ 与 $V_z,V_\pm$}
	\begin{itemize}
	\item $[L_z,V_z]=0$（直接由 $\epsilon_{zzj}=0$ 得到）。
	\item $[L_z,V_\pm] = [L_z,V_x\pm\ii V_y] = [L_z,V_x]\pm\ii[L_z,V_y]$\\
	$= \ii\hbar V_y \pm\ii(-\ii\hbar V_x) = \ii\hbar V_y\pm\hbar V_x$。\\
	$V_+$ 时取 $+$ 号：$[L_z,V_+]=\hbar(V_x+\ii V_y)=\hbar V_+$。\\
	$V_-$ 时取 $-$ 号：$[L_z,V_-]=-\hbar(V_x-\ii V_y)=-\hbar V_-$。
	\end{itemize}

	\textbf{第二组：$L_\pm$ 与 $V_z$}
	$L_\pm = L_x\pm\ii L_y$。计算 $[L_\pm,V_z]$：
	\[
	[L_\pm,V_z] = [L_x,V_z]\pm\ii[L_y,V_z].
	\]
	由矢量算符定义：
	$[L_x,V_z]=\ii\hbar\epsilon_{xzy}V_y = -\ii\hbar V_y$，
	$[L_y,V_z]=\ii\hbar\epsilon_{yzx}V_x = \ii\hbar V_x$。
	代入得
	\[
	[L_\pm,V_z] = -\ii\hbar V_y \pm\ii(\ii\hbar V_x) = -\ii\hbar V_y \mp\hbar V_x.
	\]
	$L_+$ 时取 $+$：$[L_+,V_z] = -\hbar(V_x+\ii V_y) = -\hbar V_+$。\\
	$L_-$ 时取 $-$：$[L_-,V_z] = \hbar(V_x-\ii V_y) = \hbar V_-$。
	即
	\[
	[L_\pm,V_z] = \mp\hbar V_\pm.
	\]

	\textbf{第三组：$L_\pm$ 与 $V_\pm$} 先计算 $[L_+,V_+]$：
	\[
	[L_+,V_+] = [L_x+\ii L_y, V_x+\ii V_y]
	= [L_x,V_x] + \ii[L_x,V_y] + \ii[L_y,V_x] - [L_y,V_y].
	\]
	逐个计算：
	$[L_x,V_x]=0$（$\epsilon_{xxj}=0$），
	$[L_x,V_y]=\ii\hbar\epsilon_{xyz}V_z = \ii\hbar V_z$，
	$[L_y,V_x]=\ii\hbar\epsilon_{yxz}V_z = -\ii\hbar V_z$，
	$[L_y,V_y]=0$。
	所以
	\[
	[L_+,V_+] = 0 + \ii(\ii\hbar V_z) + \ii(-\ii\hbar V_z) - 0 = -\hbar V_z + \hbar V_z = 0.
	\]

	类似地 $[L_-,V_-]=0$。

	\textbf{第四组：$L_\pm$ 与 $V_\mp$} 计算 $[L_+,V_-]$：
	\[
	[L_+,V_-] = [L_x+\ii L_y, V_x-\ii V_y]
	= [L_x,V_x] -\ii[L_x,V_y] +\ii[L_y,V_x] + [L_y,V_y].
	\]
	代入各对易子：
	$[L_x,V_x]=0$，$[L_x,V_y]=\ii\hbar V_z$，$[L_y,V_x]=-\ii\hbar V_z$，$[L_y,V_y]=0$，
	得
	\[
	[L_+,V_-] = 0 - \ii(\ii\hbar V_z) + \ii(-\ii\hbar V_z) + 0 = \hbar V_z + \hbar V_z = 2\hbar V_z.
	\]

	类似地 $[L_-,V_+] = -2\hbar V_z$。

	总结六个对易关系：
	\[
	\boxed{
	\begin{aligned}
	[L_z,V_z] &= 0, \\
	[L_z,V_\pm] &= \pm\hbar V_\pm, \\
	[L_\pm,V_z] &= \mp\hbar V_\pm, \\
	[L_\pm,V_\mp] &= \pm 2\hbar V_z, \\
	[L_\pm,V_\pm] &= 0.
	\end{aligned}}
	\]

	\textbf{(b) 选择定则}：将这些对易关系夹在 $\langle\ell'm'|$ 和 $|\ell m\rangle$ 之间。以 $[L_z,V_\pm]=\pm\hbar V_\pm$ 为例：
	\[
	\langle\ell'm'|L_zV_\pm - V_\pm L_z|\ell m\rangle = \pm\hbar\langle\ell'm'|V_\pm|\ell m\rangle.
	\]

	左边：
	$\hbar m'\langle\ell'm'|V_\pm|\ell m\rangle - \hbar m\langle\ell'm'|V_\pm|\ell m\rangle = \hbar(m'-m)\langle\ell'm'|V_\pm|\ell m\rangle$。

	因此 $(m'-m\mp1)\langle\ell'm'|V_\pm|\ell m\rangle=0$，故 $V_\pm$ 要求 $m'=m\pm1$，$V_z$ 要求 $m'=m$。

	其余对易关系给出矩阵元之间的比例关系（如 CG 系数之比），最终导向 Wigner--Eckart 定理。
	\end{solution}

\begin{question}
把两个角动量 $j_1,j_2$ 耦合成总角动量 $J$：
\[
|JM\rangle=\sum_{m_1,m_2} C^{JM}_{j_1m_1j_2m_2}|j_1m_1\rangle|j_2m_2\rangle.
\]
证明 Clebsch--Gordan 系数非零时必须满足 $M=m_1+m_2$，并推导由升降算符给出的递推关系。
\end{question}
\begin{solution}
	\textbf{Clebsch--Gordan 系数定义}：两个角动量 $j_1$、$j_2$ 耦合成总角动量 $J$ 的耦合基为
	\[
	|JM\rangle = \sum_{m_1,m_2} C^{JM}_{j_1m_1j_2m_2}\,|j_1m_1\rangle|j_2m_2\rangle,
	\]
	其中 $C^{JM}_{j_1m_1j_2m_2}$ 即 Clebsch--Gordan（CG）系数，求和限于 $|m_1|\le j_1$、$|m_2|\le j_2$。

	\textbf{$M=m_1+m_2$ 的条件}：将总角动量的 $z$ 分量算符 $J_z = J_{1z}+J_{2z}$ 作用到上式两边。

	左边：
	\[
	J_z|JM\rangle = \hbar M|JM\rangle.
	\]

	右边：
	\[
	(J_{1z}+J_{2z})\sum_{m_1,m_2}C^{JM}_{j_1m_1j_2m_2}|j_1m_1\rangle|j_2m_2\rangle
	= \sum_{m_1,m_2}C^{JM}_{j_1m_1j_2m_2}(\hbar m_1+\hbar m_2)|j_1m_1\rangle|j_2m_2\rangle.
	\]

	左右相等，比较 $|j_1m_1\rangle|j_2m_2\rangle$ 的系数：
	\[
	\hbar M\,C^{JM}_{j_1m_1j_2m_2} = \hbar(m_1+m_2)\,C^{JM}_{j_1m_1j_2m_2},
	\]
	即
	\[
	(m_1+m_2-M)\,C^{JM}_{j_1m_1j_2m_2}=0.
	\]

	因此非零 CG 系数必须满足
	\[
	\boxed{m_1+m_2 = M}.
	\]

	\textbf{递推关系的推导}：将总角动量升降算符 $J_\pm = J_{1\pm}+J_{2\pm}$ 作用到耦合基的表达式上。

	左边：
	\[
	J_\pm|JM\rangle = \hbar\sqrt{J(J+1)-M(M\pm1)}\,|J,M\pm1\rangle
	= \hbar\sqrt{J(J+1)-M(M\pm1)}\sum_{m_1',m_2'} C^{J,M\pm1}_{j_1m_1'j_2m_2'}|j_1m_1'\rangle|j_2m_2'\rangle.
	\]

	右边：
	\[
	(J_{1\pm}+J_{2\pm})\sum_{m_1,m_2}C^{JM}_{j_1m_1j_2m_2}|j_1m_1\rangle|j_2m_2\rangle
	= \sum_{m_1,m_2}C^{JM}_{j_1m_1j_2m_2}\Big(
	\hbar\sqrt{j_1(j_1+1)-m_1(m_1\pm1)}\,|j_1,m_1\pm1\rangle|j_2m_2\rangle
	\]
	\[
	+ \hbar\sqrt{j_2(j_2+1)-m_2(m_2\pm1)}\,|j_1m_1\rangle|j_2,m_2\pm1\rangle\Big).
	\]

	比较两边 $|j_1m_1\rangle|j_2m_2\rangle$ 的系数（注意指标重标定），得到递推关系：
	\[
	\sqrt{J(J+1)-M(M\pm1)}\,C^{J,M\pm1}_{j_1m_1j_2m_2}
	= \sqrt{j_1(j_1+1)-m_1(m_1\mp1)}\,C^{JM}_{j_1,m_1\mp1,j_2m_2}
	+ \sqrt{j_2(j_2+1)-m_2(m_2\mp1)}\,C^{JM}_{j_1m_1,j_2,m_2\mp1}.
	\]

	其中 $M=m_1+m_2$。这个递推关系与归一化条件
	\[
	\sum_{m_1,m_2}|C^{JM}_{j_1m_1j_2m_2}|^2 = 1
	\]
	以及相位约定（通常取 $C^{JJ}_{j_1Jj_2,J-j_1}$ 为正实数）共同唯一确定所有 CG 系数。

	因此 CG 系数非零需满足 $m_1+m_2=M$（以及三角不等式 $|j_1-j_2|\le J\le j_1+j_2$），并通过递推关系计算：
	\[
	\boxed{C^{JM}_{j_1m_1j_2m_2}\neq0 \;\Longrightarrow\; m_1+m_2=M,\; |j_1-j_2|\le J\le j_1+j_2}.
	\]
	\end{solution}

\begin{question}
将矢量算符球分量的对易关系夹在角动量本征态之间，推导其矩阵元之间的关系，并说明这些关系如何导向维格纳--埃卡特定理的形式。
\end{question}
\begin{solution}
	\textbf{基本思路}：将矢量算符球分量与角动量的六个对易关系夹在角动量本征态之间，利用 $L_z,L_\pm$ 的本征作用提取出矩阵元之间的关系。

	\textbf{第一步：$m$ 的选择定则} 取 $[L_z,V_\pm]=\pm\hbar V_\pm$ 的矩阵元：
	\[
	\langle n'\ell'm'|[L_z,V_\pm]|n\ell m\rangle = \pm\hbar\langle n'\ell'm'|V_\pm|n\ell m\rangle.
	\]

	左边展开：
	\[
	\langle n'\ell'm'|L_zV_\pm - V_\pm L_z|n\ell m\rangle
	= \hbar m'\langle n'\ell'm'|V_\pm|n\ell m\rangle - \hbar m\langle n'\ell'm'|V_\pm|n\ell m\rangle
	= \hbar(m'-m)\langle n'\ell'm'|V_\pm|n\ell m\rangle.
	\]

	因此
	\[
	\hbar(m'-m)\langle n'\ell'm'|V_\pm|n\ell m\rangle = \pm\hbar\langle n'\ell'm'|V_\pm|n\ell m\rangle,
	\]
	即 $(m'-m\mp1)\langle n'\ell'm'|V_\pm|n\ell m\rangle=0$。所以
	\[
	\langle n'\ell'm'|V_+|n\ell m\rangle \neq 0 \;\Longrightarrow\; m' = m+1,
	\]
	\[
	\langle n'\ell'm'|V_-|n\ell m\rangle \neq 0 \;\Longrightarrow\; m' = m-1.
	\]

	对 $[L_z,V_z]=0$ 同理得 $m'=m$。

	\textbf{第二步：矩阵元之间的比例关系} 取 $[L_\pm,V_z]=\mp\hbar V_\pm$ 的矩阵元：
	\[
	\langle n'\ell'm'|L_\pm V_z - V_z L_\pm|n\ell m\rangle = \mp\hbar\langle n'\ell'm'|V_\pm|n\ell m\rangle.
	\]

	左边第一项：$\langle n'\ell'm'|L_\pm V_z|n\ell m\rangle
	= \hbar\sqrt{\ell'(\ell'+1)-m'(m'\pm1)}\,\langle n'\ell',m'\mp1|V_z|n\ell m\rangle$。

	左边第二项：$-\langle n'\ell'm'|V_z L_\pm|n\ell m\rangle
	= -\hbar\sqrt{\ell(\ell+1)-m(m\pm1)}\,\langle n'\ell'm'|V_z|n\ell,m\pm1\rangle$。

	因此
	\[
	\hbar\sqrt{\ell'(\ell'+1)-m'(m'\pm1)}\,\langle n'\ell',m'\mp1|V_z|n\ell m\rangle
	- \hbar\sqrt{\ell(\ell+1)-m(m\pm1)}\,\langle n'\ell'm'|V_z|n\ell,m\pm1\rangle
	= \mp\hbar\langle n'\ell'm'|V_\pm|n\ell m\rangle.
	\]

	取适当指标，此式给出了 $V_z$ 和 $V_\pm$ 矩阵元之间的比例关系。

	\textbf{第三步：$\ell$ 的选择定则} 利用 $[L^2,V_q]$ 的矩阵元，或注意到矢量算符是 rank-1 球张量，其矩阵元由 Wigner--Eckart 定理给出：
	\[
	\langle n'\ell'm'|V_q|n\ell m\rangle = \langle \ell m;1q|\ell'm'\rangle\,\langle n'\ell'\|V\|n\ell\rangle,
	\]
	其中 $\langle \ell m;1q|\ell'm'\rangle$ 是 CG 系数，$\langle n'\ell'\|V\|n\ell\rangle$ 是约化矩阵元（与 $m,m',q$ 无关）。

	CG 系数非零的条件给出完整的角动量选择定则：
	\[
	m' = m+q,\qquad |\ell-1| \le \ell' \le \ell+1.
	\]

	\textbf{与习题 25 CG 递推关系的对应}：上面导出的矩阵元比例关系与习题 23 中 CG 系数的递推公式形式完全相同。这正是 Wigner--Eckart 定理的核心：算符矩阵元的角动量依赖完全由 CG 系数描述，径向/动力学信息全部吸收进约化矩阵元。

	因此，矢量算符的矩阵元满足：
	\[
	\boxed{\langle n'\ell'm'|V_q|n\ell m\rangle
	= \langle \ell m;1q|\ell'm'\rangle\langle n'\ell'\|V\|n\ell\rangle},
	\]
	其中 $\langle \ell m;1q|\ell'm'\rangle$ 由 $m'=m+q$、$\ell'=\ell,\ell\pm1$ 等条件限制。
	\end{solution}

\begin{question}
把位置算符 $\mathbf r$ 写成球张量分量 $r_0,r_\pm$，并利用维格纳--埃卡特定理推导氢原子电偶极矩阵元的选择定则。
\end{question}
\begin{solution}
	\textbf{位置算符的球分量}：定义位置算符的球分量（rank-1 球张量）：
	\[
	r_0 = z,\qquad
	r_\pm = \mp\frac{x\pm\ii y}{\sqrt{2}}.
	\]

	用球谐函数表示：
	\[
	z = r\cos\theta = r\sqrt{\frac{4\pi}{3}}\,Y_1^0(\theta,\phi),
	\]
	\[
	x\pm\ii y = r\sin\theta e^{\pm\ii\phi} = r\sqrt{\frac{8\pi}{3}}\,Y_1^{\pm1}(\theta,\phi),
	\]
	因此
	\[
	r_\pm = \mp\frac{r}{\sqrt{2}}\sqrt{\frac{8\pi}{3}}\,Y_1^{\pm1}(\theta,\phi)
	= r\sqrt{\frac{4\pi}{3}}\,Y_1^{\pm1}(\theta,\phi).
	\]

	可见 $r_q$ 正比于 $r Y_1^q(\theta,\phi)$，是标准的 rank-1 球张量。

	\textbf{维格纳--埃卡特定理的应用}：由于 $r_q$ 是 rank-1 球张量，Wigner--Eckart 定理直接给出
	\[
	\langle n'\ell'm'|r_q|n\ell m\rangle
	= \langle \ell m;1q|\ell'm'\rangle\,
	\langle n'\ell'\|r\|n\ell\rangle,
	\]
	其中 $\langle \ell m;1q|\ell'm'\rangle$ 是 CG 系数，$\langle n'\ell'\|r\|n\ell\rangle$ 是约化矩阵元（与 $m,m',q$ 无关）。

	\textbf{选择定则}：
	\begin{enumerate}
	\item $m$ 选择定则：CG 系数 $\langle \ell m;1q|\ell'm'\rangle$ 非零要求
	\[
	m' = m + q.
	\]
	即 $r_0$ 保持 $m$ 不变，$r_\pm$ 将 $m$ 改变 $\pm1$。
	\item $\ell$ 选择定则：角动量耦合规则 $\ell\otimes1$ 给出 $\ell' = \ell-1,\ell,\ell+1$（当 $\ell\ge1$ 时），但还需结合宇称选择定则。
	\item 宇称定则：位置算符 $\mathbf r$ 是奇宇称算符（$\eta_Q=-1$），初态宇称 $\eta_i=(-1)^\ell$，末态宇称 $\eta_f=(-1)^{\ell'}$。非零矩阵元要求
	\[
	\eta_f\eta_Q\eta_i = (-1)^{\ell'}(-1)(-1)^\ell = (-1)^{\ell'+\ell+1} = 1,
	\]
	即 $\ell'+\ell$ 为奇数。因此 $\ell'=\ell$（偶数）被排除，只剩下
	\[
	\ell' = \ell\pm1.
	\]
	\end{enumerate}

	因此氢原子电偶极矩阵元的选择定则为
	\[
	\boxed{m' = m+q,\qquad \ell' = \ell\pm1}.
	\]

	\textbf{物理意义}：这对应电偶极跃迁的选择定则。例如氢原子中电子从 $2p$（$\ell=1$）跃迁到 $1s$（$\ell=0$）是允许的，而 $2s$（$\ell=0$）到 $1s$（$\ell=0$）被宇称禁止。径向积分的具体数值（如 $\langle 1s\|r\|2p\rangle$ 等）决定了跃迁速率的大小，全都包含在约化矩阵元中。
	\end{solution}

\begin{question}
求例题 6.7 中系统的海森伯绘景动量算符 $p_H(t)$，并讨论所得结果与经典运动方程的对应。
\end{question}
\begin{solution}
	\textbf{海森伯运动方程}：在海森伯绘景中，算符 $Q_H(t)=U^\dagger(t)Q U(t)$ 的运动方程为
	\[
	\frac{\dd Q_H}{\dd t} = \frac{\ii}{\hbar}[H,Q_H] + \left(\frac{\partial Q}{\partial t}\right)_H.
	\]

	对于动量算符 $p_H(t)$，若哈密顿量不显含时：
	\[
	\frac{\dd p_H}{\dd t} = \frac{\ii}{\hbar}[H,p_H].
	\]

	\textbf{以 $H=p^2/(2m)+V(x)$ 为例}：计算对易子 $[H,p]$：
	\[
	[H,p] = \left[\frac{p^2}{2m}+V(x),\,p\right]
	= \frac{1}{2m}[p^2,p] + [V(x),p].
	\]

	$[p^2,p]=0$，而 $[V(x),p] = V(x)p - pV(x)$。利用 $[x,p]=\ii\hbar$ 和 $p=-\ii\hbar\partial_x$，对于任意波函数 $\psi(x)$：
	\[
	[V(x),p]\psi = V(x)(-\ii\hbar\psi') - (-\ii\hbar)(V\psi)' = -\ii\hbar V\psi' + \ii\hbar(V'\psi+V\psi') = \ii\hbar V'\psi.
	\]

	因此
	\[
	[H,p] = \ii\hbar V'(x).
	\]

	代入海森伯方程：
	\[
	\frac{\dd p_H}{\dd t} = \frac{\ii}{\hbar}[H,p_H] = \frac{\ii}{\hbar}\cdot\ii\hbar V'(x_H) = -V'(x_H).
	\]

	\textbf{具体例子}：
	\begin{enumerate}
	\item 线性势 $V(x) = -Fx$（如均匀电场中带电粒子），则 $V'(x) = -F$，所以
	\[
	\frac{\dd p_H}{\dd t} = F \quad\Longrightarrow\quad p_H(t) = p_H(0) + Ft.
	\]
	这正是经典力学中的匀加速运动：$\dot p = F$。

	\item 谐振子 $V(x)=\frac12 m\omega^2 x^2$，则 $V'(x)=m\omega^2 x$，所以
	\[
	\frac{\dd p_H}{\dd t} = -m\omega^2 x_H.
	\]
	结合 $\dot x_H = p_H/m$，得到
	\[
	\ddot x_H + \omega^2 x_H = 0,
	\]
	与经典谐振子运动方程完全相同。
	\end{enumerate}

	因此量子海森伯方程给出的算符运动方程与经典哈密顿方程形式一致：
	\[
	\boxed{\dot p_H = -V'(x_H)}.
	\]
	\end{solution}

\begin{question}
考虑质量为 $m$ 的一维自由粒子。证明海森伯绘景中
\[
x_H(t)=x_H(0)+{p_H(0)\over m}t,
\qquad
p_H(t)=p_H(0),
\]
并解释它们与经典自由运动方程的关系。
\end{question}
\begin{solution}
	\textbf{自由粒子的哈密顿量}：一维自由粒子 $H = \dfrac{p^2}{2m}$，不显含时。

	\textbf{海森伯方程}：动量算符 $p_H(t)$ 的运动方程：
	\[
	\frac{\dd p_H}{\dd t} = \frac{\ii}{\hbar}[H,p_H].
	\]

	由于在任意时刻 $[p_H(t),H]=0$（因为 $p_H(t)=U^\dagger p U$ 与 $H$ 对易，$H$ 在海森伯绘景下不变），所以
	\[
	\frac{\dd p_H}{\dd t} = 0 \quad\Longrightarrow\quad p_H(t) = p_H(0).
	\]

	位置算符 $x_H(t)$ 的运动方程：
	\[
	\frac{\dd x_H}{\dd t} = \frac{\ii}{\hbar}[H,x_H].
	\]

	计算对易子。首先注意 $H=p^2/(2m)$ 在海森伯绘景中形式不变：
	\[
	[H,x_H] = \frac{1}{2m}[p_H^2,x_H].
	\]

	利用 $[p_H^2,x_H] = p_H[p_H,x_H] + [p_H,x_H]p_H$ 以及 $[p_H,x_H] = U^\dagger[p,x]U = -\ii\hbar$（因为 $[p,x]=-\ii\hbar$ 在幺正变换下保持），得
	\[
	[p_H^2,x_H] = p_H(-\ii\hbar) + (-\ii\hbar)p_H = -2\ii\hbar p_H.
	\]

	因此
	\[
	[H,x_H] = \frac{1}{2m}(-2\ii\hbar p_H) = -\frac{\ii\hbar}{m}p_H.
	\]

	代入海森伯方程：
	\[
	\frac{\dd x_H}{\dd t} = \frac{\ii}{\hbar}\left(-\frac{\ii\hbar}{m}p_H\right) = \frac{p_H}{m}.
	\]

	\textbf{积分求解}：
	\[
	\frac{\dd x_H}{\dd t} = \frac{p_H(0)}{m},
	\]
	对时间积分：
	\[
	x_H(t) = x_H(0) + \int_0^t \frac{p_H(0)}{m}\,\dd t' = x_H(0) + \frac{p_H(0)}{m}\,t.
	\]

	因此
	\[
	\boxed{p_H(t) = p_H(0)},\qquad
	\boxed{x_H(t) = x_H(0) + \frac{p_H(0)}{m}\,t}.
	\]

	\textbf{与经典的对应}：经典自由粒子的运动方程为
	\[
	p = \text{常量},\qquad x(t) = x_0 + \frac{p}{m}t.
	\]

	量子海森伯绘景中的算符运动方程与经典运动方程形式完全一致，只是将经典变量替换为算符（海森伯绘景算符）。这体现了 Ehrenfest 定理：期望值遵循经典运动方程。
	\end{solution}

\begin{question}
由薛定谔方程对 $\psi(x,t+\delta t)$ 作泰勒展开，证明无穷小时间平移算符为
\[
U(\delta t)=1-{\ii H\delta t\over\hbar}+O(\delta t^2).
\]
\end{question}
\begin{solution}
	\textbf{薛定谔方程}：一维薛定谔方程为
	\[
	\ii\hbar\frac{\partial\psi(x,t)}{\partial t} = H\psi(x,t),
	\]
	其中 $H$ 是哈密顿算符。

	\textbf{泰勒展开}：对时间 $t$ 作微小推移 $\delta t$，将 $\psi(x,t+\delta t)$ 在 $t$ 附近泰勒展开：
	\[
	\psi(x,t+\delta t) = \psi(x,t) + \delta t\,\frac{\partial\psi(x,t)}{\partial t} + \frac{(\delta t)^2}{2}\frac{\partial^2\psi(x,t)}{\partial t^2} + \cdots
	\]
	只保留到一阶项：
	\[
	\psi(x,t+\delta t) = \psi(x,t) + \delta t\,\frac{\partial\psi}{\partial t} + O(\delta t^2).
	\]

	\textbf{代入薛定谔方程}：由 $\ii\hbar\frac{\partial\psi}{\partial t}=H\psi$ 得 $\frac{\partial\psi}{\partial t} = -\frac{\ii}{\hbar}H\psi$，代入上式：
	\[
	\psi(x,t+\delta t) = \psi(x,t) + \delta t\left(-\frac{\ii}{\hbar}H\right)\psi(x,t) + O(\delta t^2)
	= \left(1 - \frac{\ii H\delta t}{\hbar}\right)\psi(x,t) + O(\delta t^2).
	\]

	\textbf{无穷小时间平移算符}：对比 $\psi(t+\delta t)=U(\delta t)\psi(t)$ 得
	\[
	U(\delta t) = 1 - \frac{\ii H\delta t}{\hbar} + O(\delta t^2).
	\]

	\textbf{有限时间演化算符}：将时间区间 $[0,t]$ 等分为 $N$ 段，每段 $\delta t = t/N$。连续应用无穷小平移：
	\[
	U(t) = \lim_{N\to\infty} [U(t/N)]^N
	= \lim_{N\to\infty} \left(1 - \frac{\ii H t}{\hbar N}\right)^N.
	\]

	利用极限 $\lim_{N\to\infty}(1+z/N)^N = e^z$（$z$ 为算符时需小心，但当 $H$ 不显含时时该式成立），得
	\[
	U(t) = \exp\!\left(-\frac{\ii H t}{\hbar}\right).
	\]

	因此，无穷小时间平移算符为
	\[
	\boxed{U(\delta t) = 1 - \frac{\ii H\delta t}{\hbar} + O(\delta t^2)},
	\]
	有限时间演化算符为 $\boxed{U(t)=\exp(-\ii H t/\hbar)}$（$H$ 不显含时）。

	\textbf{幺正性验证}：$U(t)^\dagger = \exp(\ii H t/\hbar) = U(t)^{-1}$，因为 $H$ 是厄米算符。
	\end{solution}

\begin{question}
对海森伯绘景算符 $Q_H(t)=U^\dagger(t)QU(t)$ 求导，推导海森伯运动方程。并将其用于 $H=p^2/(2m)+V(x)$，求 $x_H,p_H$ 的运动方程。
\end{question}
\begin{solution}
	\textbf{海森伯绘景定义}：海森伯绘景算符定义为
	\[
	Q_H(t) = U^\dagger(t)\,Q\,U(t),
	\]
	其中 $U(t)=\exp(-\ii Ht/\hbar)$ 是时间演化算符（$H$ 不显含时）。

	\textbf{对时间求导}：
	\[
	\frac{\dd Q_H}{\dd t} = \frac{\dd U^\dagger}{\dd t}\,Q\,U + U^\dagger\,Q\,\frac{\dd U}{\dd t} + U^\dagger\frac{\partial Q}{\partial t}U.
	\]

	由 $U(t)=\exp(-\ii Ht/\hbar)$ 得
	\[
	\frac{\dd U}{\dd t} = -\frac{\ii}{\hbar}H\,U(t),\qquad
	\frac{\dd U^\dagger}{\dd t} = \frac{\ii}{\hbar}U^\dagger(t)\,H.
	\]

	代入：
	\[
	\frac{\dd Q_H}{\dd t} = \frac{\ii}{\hbar}U^\dagger H Q U - \frac{\ii}{\hbar}U^\dagger Q H U + U^\dagger\frac{\partial Q}{\partial t}U.
	\]

	插入 $UU^\dagger=1$：
	\[
	U^\dagger H Q U = (U^\dagger H U)(U^\dagger Q U) = H_H Q_H,
	\]
	\[
	U^\dagger Q H U = (U^\dagger Q U)(U^\dagger H U) = Q_H H_H,
	\]
	其中 $H_H=U^\dagger H U = H$（$H$ 不显含时在海森伯绘景中不变）。因此
	\[
	\frac{\dd Q_H}{\dd t} = \frac{\ii}{\hbar}(H_H Q_H - Q_H H_H) + \left(\frac{\partial Q}{\partial t}\right)_H
	= \frac{\ii}{\hbar}[H,Q_H] + \left(\frac{\partial Q}{\partial t}\right)_H.
	\]

	这就是海森伯运动方程：
	\[
	\boxed{\frac{\dd Q_H}{\dd t} = \frac{\ii}{\hbar}[H,Q_H] + \left(\frac{\partial Q}{\partial t}\right)_H}.
	\]

	\textbf{应用：$H=p^2/(2m)+V(x)$}：

	对 $x_H$：
	\[
	\frac{\dd x_H}{\dd t} = \frac{\ii}{\hbar}[H,x_H].
	\]

	计算 $[H,x_H] = \frac{1}{2m}[p_H^2,x_H]$。由 $[p_H,x_H]=-\ii\hbar$：
	\[
	[p_H^2,x_H] = p_H[p_H,x_H] + [p_H,x_H]p_H = -2\ii\hbar p_H.
	\]

	所以 $[H,x_H] = -\frac{\ii\hbar}{m}p_H$，代入得
	\[
	\frac{\dd x_H}{\dd t} = \frac{\ii}{\hbar}\left(-\frac{\ii\hbar}{m}p_H\right) = \frac{p_H}{m}.
	\]

	对 $p_H$：
	\[
	\frac{\dd p_H}{\dd t} = \frac{\ii}{\hbar}[H,p_H] = \frac{\ii}{\hbar}[V(x_H),p_H].
	\]

	$[V(x_H),p_H] = \ii\hbar V'(x_H)$（证明见第 26 题），所以
	\[
	\frac{\dd p_H}{\dd t} = \frac{\ii}{\hbar}\cdot\ii\hbar V'(x_H) = -V'(x_H).
	\]

	综合得算符形式的牛顿方程：
	\[
	\boxed{\dot x_H = \frac{p_H}{m}},\qquad
	\boxed{\dot p_H = -V'(x_H)}.
	\]

	这说明海森伯绘景中的算符运动方程与经典哈密顿方程完全同形，体现了量子--经典对应原理。
	\end{solution}

\begin{question}
设一维不含时哈密顿量的定态为 $\psi_n(x)$、能量为 $E_n$。
\begin{enumerate}
\item 证明传播子可写为
\[
K(x,x';t)=\sum_n\psi_n(x)\psi_n^*(x')e^{-\ii E_nt/\hbar}.
\]
\item 将该表达式用于一维谐振子，得到标准传播子的形式。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 传播子的谱展开}：设哈密顿量 $H$ 的完备本征函数系为 $\{\psi_n(x)\}$，满足
	\[
	H\psi_n(x) = E_n\psi_n(x),
	\]
	且完备性关系为 $\sum_n \psi_n(x)\psi_n^*(x') = \delta(x-x')$。

	含时薛定谔方程 $i\hbar\partial_t\psi=H\psi$ 的通解为
	\[
	\psi(x,t) = \sum_n c_n\psi_n(x)e^{-\ii E_n t/\hbar},
	\]
	其中系数由初始条件确定：
	\[
	c_n = \int \psi_n^*(x')\psi(x',0)\,\dd x'.
	\]

	代入得
	\[
	\psi(x,t) = \sum_n \left[\int \psi_n^*(x')\psi(x',0)\,\dd x'\right]\psi_n(x)e^{-\ii E_n t/\hbar}
	= \int \left[\sum_n \psi_n(x)\psi_n^*(x')e^{-\ii E_n t/\hbar}\right] \psi(x',0)\,\dd x'.
	\]

	因此传播子（格林函数）为
	\[
	K(x,x';t) = \sum_n \psi_n(x)\psi_n^*(x')e^{-\ii E_n t/\hbar},
	\]
	它满足初始条件 $\lim_{t\to0} K(x,x';t)=\delta(x-x')$（由完备性）。

	\textbf{(b) 谐振子的标准传播子}：一维谐振子的哈密顿量为
	\[
	H = \frac{p^2}{2m} + \frac12 m\omega^2 x^2,
	\]
	本征态为 $\psi_n(x)$，能量 $E_n = \hbar\omega(n+1/2)$。

	传播子为
	\[
	K(x,x';t) = \sum_{n=0}^\infty \psi_n(x)\psi_n^*(x') e^{-\ii\omega(n+1/2)t}.
	\]

	利用谐振子本征函数的生成函数（Mehler 求和公式）：
	\[
	\sum_{n=0}^\infty \frac{H_n(\xi)H_n(\xi')}{2^n n!}z^n = \frac{1}{\sqrt{1-z^2}}\exp\!\left(\frac{2\xi\xi'z-(\xi^2+\xi'^2)z^2}{1-z^2}\right),
	\]
	其中 $H_n$ 是厄米多项式，$\xi = \sqrt{m\omega/\hbar}\,x$，令 $z=e^{-\ii\omega t}$，可最终推导出标准传播子形式。

	结果为
	\[
	\boxed{K(x,x';t) = \sqrt{\frac{m\omega}{2\pi\ii\hbar\sin\omega t}}
	\exp\!\left\{\frac{\ii m\omega}{2\hbar\sin\omega t}
	\left[(x^2+x'^2)\cos\omega t - 2xx'\right]\right\}}.
	\]

	\textbf{验证初始条件}：当 $t\to0$ 时，$\sin\omega t \approx \omega t$，$\cos\omega t \approx 1$，则
	\[
	K \approx \sqrt{\frac{m}{2\pi\ii\hbar t}} \exp\!\left[-\frac{m}{2\ii\hbar t}(x-x')^2\right],
	\]
	这是高斯型 $\delta$ 函数的表达式：$\lim_{t\to0} K(x,x';t) = \delta(x-x')$，满足物理要求。
	\end{solution}

\begin{question}
用动量完备基定义平移算符
\[
T(a)=\int e^{-\ii pa/\hbar}|p\rangle\langle p|\,\dd p.
\]
证明它对任意波函数的作用为 $T(a)\psi(x)=\psi(x-a)$。
\end{question}
\begin{solution}
	\textbf{动量完备基定义平移算符}：动量本征态 $|p\rangle$ 满足完备性关系
	\[
	\int |p\rangle\langle p|\,\dd p = 1.
	]

	平移算符 $T(a)$ 可用动量本征态的谱分解定义为
	\[
	T(a) = \int e^{-\ii pa/\hbar}\,|p\rangle\langle p|\,\dd p.
	\]

	这是因为动量本征态是平移算符的本征态：$T(a)|p\rangle = e^{-\ii pa/\hbar}|p\rangle$，而谱分解即 $T(a)=\int |p\rangle\langle p|\,T(a)\,|p\rangle\langle p|\,\dd p$。

	\textbf{作用在波函数上}：任意波函数 $\psi(x)$ 可作傅里叶展开：
	\[
	\psi(x) = \langle x|\psi\rangle = \int \langle x|p\rangle\langle p|\psi\rangle\,\dd p
	= \int \phi(p)\,\frac{e^{\ii px/\hbar}}{\sqrt{2\pi\hbar}}\,\dd p,
	\]
	其中 $\phi(p)=\langle p|\psi\rangle$ 是动量空间波函数，$\langle x|p\rangle = e^{\ii px/\hbar}/\sqrt{2\pi\hbar}$。

	$T(a)$ 作用在态 $|\psi\rangle$ 上：
	\[
	T(a)|\psi\rangle = \int e^{-\ii pa/\hbar}|p\rangle\langle p|\psi\rangle\,\dd p
	= \int e^{-\ii pa/\hbar}\,\phi(p)\,|p\rangle\,\dd p.
	\]

	在坐标表象中：
	\[
	(T(a)\psi)(x) = \langle x|T(a)|\psi\rangle
	= \int e^{-\ii pa/\hbar}\,\phi(p)\,\langle x|p\rangle\,\dd p
	= \int \phi(p)\,e^{-\ii pa/\hbar}\,\frac{e^{\ii px/\hbar}}{\sqrt{2\pi\hbar}}\,\dd p.
	]

	合并指数：
	\[
	(T(a)\psi)(x) = \int \phi(p)\,\frac{e^{\ii p(x-a)/\hbar}}{\sqrt{2\pi\hbar}}\,\dd p.
	\]

	这正是 $\psi(x-a)$ 的傅里叶展开表达式：
	\[
	\psi(x-a) = \int \phi(p)\,\frac{e^{\ii p(x-a)/\hbar}}{\sqrt{2\pi\hbar}}\,\dd p.
	\]

	因此
	\[
	\boxed{T(a)\psi(x) = \psi(x-a)}.
	\]

	\textbf{优势}：这个推导只要求 $\psi(x)$ 的傅里叶展开存在（即 $\psi\in L^2$），不要求 $\psi$ 处处可微。即使波函数在某点一阶导数不连续，谱分解定义仍给出正确的平移结果。这是用坐标空间导数定义平移算符 $T(a)=\exp(-a\partial_x)$ 的一种推广（后者要求波函数可微）。
	\end{solution}

\begin{question}
自旋 $1/2$ 的旋量在绕单位矢量 $\hat n$ 转动角 $\phi$ 时由
\[
U(\phi)=\exp\left(-{\ii\phi\over2}\boldsymbol\sigma\cdot\hat n\right)
\]
给出。
\begin{enumerate}
\item 证明泡利矩阵恒等式 $(\boldsymbol\sigma\cdot\mathbf a)(\boldsymbol\sigma\cdot\mathbf b)=\mathbf a\cdot\mathbf b+\ii\boldsymbol\sigma\cdot(\mathbf a\times\mathbf b)$。
\item 推导 $U(\phi)$ 的闭式表达式和矩阵形式。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{(a) 泡利矩阵恒等式}：泡利矩阵 $\sigma_x,\sigma_y,\sigma_z$ 满足
	\[
	\sigma_i\sigma_j = \delta_{ij} + \ii\epsilon_{ijk}\sigma_k.
	\]

	对于任意矢量 $\mathbf a,\mathbf b$：
	\[
	(\boldsymbol\sigma\cdot\mathbf a)(\boldsymbol\sigma\cdot\mathbf b)
	= \sum_{i,j} a_i b_j \sigma_i\sigma_j
	= \sum_{i,j} a_i b_j (\delta_{ij} + \ii\epsilon_{ijk}\sigma_k).
	\]

	第一项：$\sum_{i,j} a_i b_j \delta_{ij} = \sum_i a_i b_i = \mathbf a\cdot\mathbf b$。
	第二项：$\ii\sum_{i,j,k} a_i b_j \epsilon_{ijk}\sigma_k = \ii\boldsymbol\sigma\cdot(\mathbf a\times\mathbf b)$。

	因此
	\[
	\boxed{(\boldsymbol\sigma\cdot\mathbf a)(\boldsymbol\sigma\cdot\mathbf b)
	= \mathbf a\cdot\mathbf b + \ii\boldsymbol\sigma\cdot(\mathbf a\times\mathbf b)}.
	\]

	\textbf{(b) $U(\phi)$ 的闭式表达式}：取 $\mathbf a=\mathbf b=\hat n$，则 $\mathbf a\cdot\mathbf b=1$，$\mathbf a\times\mathbf b=0$，得
	\[
	(\boldsymbol\sigma\cdot\hat n)^2 = 1.
	\]

	利用此性质展开指数：
	\[
	\exp\!\left(-\frac{\ii\phi}{2}\boldsymbol\sigma\cdot\hat n\right)
	= \sum_{k=0}^\infty \frac{(-\ii\phi/2)^k}{k!}(\boldsymbol\sigma\cdot\hat n)^k.
	\]

	将 $k$ 按奇偶分开。令 $M \equiv \boldsymbol\sigma\cdot\hat n$，则 $M^2=1$，$M^{2m}=1$，$M^{2m+1}=M$：
	\[
	e^{-\ii\phi M/2}
	= \sum_{m=0}^\infty \frac{(-\ii\phi/2)^{2m}}{(2m)!} M^{2m}
	+ \sum_{m=0}^\infty \frac{(-\ii\phi/2)^{2m+1}}{(2m+1)!} M^{2m+1}
	= \sum_{m=0}^\infty \frac{(-1)^m(\phi/2)^{2m}}{(2m)!}
	- \ii M\sum_{m=0}^\infty \frac{(-1)^m(\phi/2)^{2m+1}}{(2m+1)!}.
	\]

	识别余弦和正弦级数：
	\[
	e^{-\ii\phi M/2} = \cos\frac{\phi}{2} - \ii M\sin\frac{\phi}{2}.
	\]

	因此闭式表达式为
	\[
	\boxed{U(\phi) = \cos\frac{\phi}{2} - \ii(\boldsymbol\sigma\cdot\hat n)\sin\frac{\phi}{2}}.
	\]

	\textbf{矩阵形式}：设 $\hat n = (\sin\theta\cos\varphi,\sin\theta\sin\varphi,\cos\theta)$，则
	\[
	\boldsymbol\sigma\cdot\hat n = \sigma_x\sin\theta\cos\varphi + \sigma_y\sin\theta\sin\varphi + \sigma_z\cos\theta.
	\]

	泡利矩阵为
	\[
	\sigma_x = \begin{pmatrix}0&1\\1&0\end{pmatrix},\quad
	\sigma_y = \begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix},\quad
	\sigma_z = \begin{pmatrix}1&0\\0&-1\end{pmatrix}.
	\]

	代入：
	\[
	\boldsymbol\sigma\cdot\hat n
	= \sin\theta\cos\varphi\begin{pmatrix}0&1\\1&0\end{pmatrix}
	+ \sin\theta\sin\varphi\begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix}
	+ \cos\theta\begin{pmatrix}1&0\\0&-1\end{pmatrix}
	= \begin{pmatrix}\cos\theta & \sin\theta(\cos\varphi-\ii\sin\varphi)\\ \sin\theta(\cos\varphi+\ii\sin\varphi) & -\cos\theta\end{pmatrix}
	= \begin{pmatrix}\cos\theta & e^{-\ii\varphi}\sin\theta\\ e^{\ii\varphi}\sin\theta & -\cos\theta\end{pmatrix}.
	\]

	因此旋转矩阵为
	\[
	U(\phi) = \cos\frac{\phi}{2}\begin{pmatrix}1&0\\0&1\end{pmatrix}
	- \ii\sin\frac{\phi}{2}\begin{pmatrix}\cos\theta & e^{-\ii\varphi}\sin\theta\\ e^{\ii\varphi}\sin\theta & -\cos\theta\end{pmatrix}
	= \begin{pmatrix}
	\cos\frac{\phi}{2} - \ii\cos\theta\sin\frac{\phi}{2} & -\ii e^{-\ii\varphi}\sin\theta\sin\frac{\phi}{2}\\
	-\ii e^{\ii\varphi}\sin\theta\sin\frac{\phi}{2} & \cos\frac{\phi}{2} + \ii\cos\theta\sin\frac{\phi}{2}
	\end{pmatrix}.
	\]

	这就是自旋 $1/2$ 旋量绕 $\hat n$ 轴旋转 $\phi$ 角的 $SU(2)$ 矩阵：
	\[
	\boxed{R_{\hat n}(\phi)=\begin{pmatrix}
	\cos\frac{\phi}{2}-\ii\cos\theta\sin\frac{\phi}{2}&-\ii e^{-\ii\varphi}\sin\theta\sin\frac{\phi}{2}\\
	-\ii e^{\ii\varphi}\sin\theta\sin\frac{\phi}{2}&\cos\frac{\phi}{2}+\ii\cos\theta\sin\frac{\phi}{2}
	\end{pmatrix}}.
	\]
	\end{solution}

\begin{question}
考虑边长为 $L$ 的二维无限深方势阱，原点取在方阱中心。
\begin{enumerate}
\item 说明为什么能量只依赖 $n_x^2+n_y^2$，并证明 $(a,b)$ 与 $(b,a)$ 态简并。
\item 证明绕中心旋转 $90^\circ$ 会把这两个态相互变换。
\item 构造旋转算符的好态线性组合。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{二维无限深方势阱}：边长为 $L$，原点取在中心，势能为
	\[
	V(x,y) = \begin{cases}
	0, & |x|<L/2,\ |y|<L/2,\\
	\infty, & \text{其它}.
	\end{cases}
	\]

	定态波函数为
	\[
	\psi_{n_x n_y}(x,y) = \frac{2}{L}\sin\!\left[\frac{n_x\pi}{L}\left(x+\frac{L}{2}\right)\right]
	\sin\!\left[\frac{n_y\pi}{L}\left(y+\frac{L}{2}\right)\right],
	\]
	能量为
	\[
	E_{n_x n_y} = \frac{\pi^2\hbar^2}{2mL^2}(n_x^2+n_y^2),\qquad n_x,n_y=1,2,3,\ldots
	\]

	\textbf{简并性}：能量只依赖于 $n_x^2+n_y^2$，因此当 $a\neq b$ 时，$(a,b)$ 和 $(b,a)$ 态具有相同的能量：
	\[
	E_{ab} = \frac{\pi^2\hbar^2}{2mL^2}(a^2+b^2) = E_{ba}.
	\]

	\textbf{旋转 $90^\circ$ 的作用}：绕中心逆时针旋转 $90^\circ$ 使坐标变换为
	\[
	R_{90^\circ}:\ (x,y) \longrightarrow (-y,x).
	\]

	考虑旋转后的波函数：
	\[
	R_{90^\circ}\psi_{ab}(x,y) = \psi_{ab}(-y,x).
	\]

	代入波函数表达式：
	\[
	\psi_{ab}(-y,x)=\frac{2}{L}\sin\!\left[\frac{a\pi}{L}\left(-y+\frac{L}{2}\right)\right]
	\sin\!\left[\frac{b\pi}{L}\left(x+\frac{L}{2}\right)\right].
	\]

	利用 $\sin(-\theta+\alpha) = (-1)^{\lfloor 2\alpha/\pi\rfloor}\sin(\theta-\alpha+\cdots)$，更直接地设 $x'=-y$，$y'=x$。注意平移项 $L/2$ 的存在，旋转后需要额外处理边界。为简化，考虑波函数用 $\sin$ 的自变量偏移形式：
	\[
	\psi_{ab}(-y,x) = (-1)^{a+1}\frac{2}{L}\sin\!\left[\frac{a\pi}{L}\left(y+\frac{L}{2}\right)\right]
	\sin\!\left[\frac{b\pi}{L}\left(x+\frac{L}{2}\right)\right] \quad (\text{当 } a\text{ 为奇数时需小心符号}).
	\]

	更简洁地，注意到 $\sin(\frac{n\pi}{L}(x+L/2))$ 在 $x\to -y$ 变换下，$\sin(\frac{a\pi}{L}(-y+L/2)) = \sin(-\frac{a\pi}{L}y+\frac{a\pi}{2})$。利用三角恒等式：
	\[
	\sin\!\left(-\frac{a\pi}{L}y+\frac{a\pi}{2}\right) = (-1)^{a+1}\sin\!\left(\frac{a\pi}{L}y+\frac{a\pi}{2}\right) = (-1)^{a+1}\sin\!\left[\frac{a\pi}{L}\left(y+\frac{L}{2}\right)\right]?
	\]

	注意 $\sin(\frac{a\pi}{L}(y+L/2))$ 和 $\sin(\frac{a\pi}{L}(-y+L/2))$ 的关系依赖于 $a$ 的奇偶。总体而言，旋转 $90^\circ$ 将 $\psi_{ab}$ 映射到 $\pm\psi_{ba}$（相差一个符号因子），因此两个态在旋转下互换。

	\textbf{构造旋转算符的本征态}：取对称和反对称组合：
	\[
	\psi_\pm = \frac{1}{\sqrt{2}}(\psi_{ab}\pm\psi_{ba}).
	\]

	$R_{90^\circ}$ 将 $\psi_{ab}\leftrightarrow\pm\psi_{ba}$：
	\[
	R_{90^\circ}\psi_+ = \frac{1}{\sqrt{2}}(R_{90^\circ}\psi_{ab}+R_{90^\circ}\psi_{ba})
	= \frac{1}{\sqrt{2}}(\eta\psi_{ba}+\eta^*\psi_{ab}) = \eta^*\psi_\pm?
	\]

	更精确地，若 $R_{90^\circ}\psi_{ab} = c\,\psi_{ba}$，$R_{90^\circ}\psi_{ba} = c^*\,\psi_{ab}$（$|c|=1$），则
	\[
	R_{90^\circ}\psi_+ = c^*\psi_+,\qquad
	R_{90^\circ}\psi_- = -c^*\psi_-.
	\]

	因此 $\psi_\pm$ 是 $R_{90^\circ}$ 的本征态，本征值为 $\pm c^*$。特别地，当 $a,b$ 同为奇数或同为偶数时，$c=\pm1$，本征值为 $\pm1$ 或 $\pm\ii$。

	例如 $a=5,b=3$ 时，$n_x^2+n_y^2=34$，$\psi_{53}$ 和 $\psi_{35}$ 简并。对称组合 $\psi_+$ 和反对称组合 $\psi_-$ 是旋转算符的好态，旋转 $90^\circ$ 后分别获得确定的相位因子：
	\[
	\boxed{\psi_\pm = \frac{1}{\sqrt{2}}(\psi_{ab}\pm\psi_{ba})}.
	\]

	这就对角化了简并子空间中的旋转操作。
	\end{solution}

\begin{question}
库仑势具有比普通旋转不变性更高的对称性。定义拉普拉斯--龙格--楞次矢量
\[
\mathbf M={1\over2m}(\mathbf p\times\mathbf L-\mathbf L\times\mathbf p)-k{\mathbf r\over r}.
\]
证明在 $V(r)=-k/r$ 中 $\mathbf M$ 与哈密顿量对易，并推导它与 $\mathbf L$ 之间的主要对易关系。
\end{question}
\begin{solution}
	\textbf{拉普拉斯--龙格--楞次矢量（LRL 矢量）}：定义
	\[
	\mathbf M = \frac{1}{2m}(\mathbf p\times\mathbf L - \mathbf L\times\mathbf p) - k\frac{\mathbf r}{r},
	\]
	其中 $\mathbf L=\mathbf r\times\mathbf p$ 是角动量，$k>0$ 是库仑势 $V(r)=-k/r$ 的系数。注意 $\mathbf p\times\mathbf L$ 不同于 $\mathbf L\times\mathbf p$，它们的差与 $\mathbf p$ 有关。

	\textbf{第一步：证明 $\mathbf M$ 与哈密顿量对易} 哈密顿量为
	\[
	H = \frac{p^2}{2m} - \frac{k}{r}.
	\]

	先计算 $[H,\mathbf p\times\mathbf L]$ 和 $[H,\mathbf L\times\mathbf p]$。利用基本对易关系：
	\[
	[x_i,p_j] = \ii\hbar\delta_{ij},\qquad [L_i,L_j]=\ii\hbar\epsilon_{ijk}L_k,
	\]
	\[
	[p_i,f(r)] = -\ii\hbar\frac{\partial f}{\partial x_i},\qquad
	[L_i,f(r)] = 0.
	\]

	由于 $L$ 与哈密顿量的标量势部分对易，关键在于处理动能部分。计算 $[p^2,\mathbf p\times\mathbf L]$ 和 $[p^2,\mathbf L\times\mathbf p]$，以及 $[1/r,\mathbf p\times\mathbf L]$ 等。以 $x$ 分量为例：

	利用恒等式 $\mathbf p\times\mathbf L = \mathbf L\times\mathbf p + 2\ii\hbar\mathbf p$（由 $\epsilon_{ijk}\epsilon_{ilm}=\delta_{jl}\delta_{km}-\delta_{jm}\delta_{kl}$ 可证），可将 $\mathbf M$ 简化为
	\[
	\mathbf M = \frac{1}{m}\mathbf p\times\mathbf L - \frac{\ii\hbar}{m}\mathbf p - k\frac{\mathbf r}{r}.
	\]

	通过对各分量直接计算（过程较长，此处给出关键结果）：
	\[
	[H,M_i] = 0\quad (i=x,y,z).
	\]

	因此 $\mathbf M$ 是守恒量：
	\[
	\boxed{[H,\mathbf M]=0}.
	\]

	\textbf{第二步：$\mathbf M$ 与 $\mathbf L$ 的对易关系} 计算 $[L_i,M_j]$。利用 $M_j$ 的定义和 $[L_i,\mathbf p\times\mathbf L]$ 的矢量变换性质：
	\[
	[L_i,M_j] = \frac{1}{2m}[L_i,(\mathbf p\times\mathbf L)_j - (\mathbf L\times\mathbf p)_j] - k\left[L_i,\frac{r_j}{r}\right].
	\]

	由于 $\mathbf L$ 是旋转生成元，任何矢量算符 $\mathbf V$ 满足 $[L_i,V_j]=\ii\hbar\epsilon_{ijk}V_k$。直接验证 $\mathbf M$ 也是矢量算符：
	\[
	[L_i,M_j] = \ii\hbar\epsilon_{ijk}M_k.
	\]

	因此 $\mathbf M$ 在旋转下像矢量一样变换。

	\textbf{第三步：$\mathbf M$ 与自身的对易关系} 进一步计算 $[M_i,M_j]$。利用已得的对易关系和对库仑势的特有约束（仅当势为 $-k/r$ 时该式成立），可得
	\[
	[M_i,M_j] = -\frac{2\ii\hbar H}{m}\epsilon_{ijk}L_k.
	\]

	注意右边出现了 $H$，说明在能量壳上（$H=E$ 固定），$M_i$ 之间的对易关系类似于角动量，但带有能量因子。

	\textbf{物理意义}：这些对易关系揭示了氢原子的 $SO(4)$ 对称性。在束缚态（$E<0$）中，定义
	\[
	\mathbf A = \sqrt{-\frac{m}{2E}}\,\mathbf M,
	\]
	则 $\mathbf A$ 与 $\mathbf L$ 满足 $SO(4)$ 代数：
	\[
	[L_i,L_j]=\ii\hbar\epsilon_{ijk}L_k,\quad
	[L_i,A_j]=\ii\hbar\epsilon_{ijk}A_k,\quad
	[A_i,A_j]=\ii\hbar\epsilon_{ijk}L_k.
	\]

	重新组合 $\mathbf J_\pm = (\mathbf L\pm\mathbf A)/2$，它们构成两个独立的 $SU(2)$ 代数。这解释了氢原子能级简并度 $n^2$（比单纯的转动对称 $SO(3)$ 所预期的 $2\ell+1$ 更高）——这就是库仑势中 LRL 矢量守恒带来的额外对称性：
	\[
	\boxed{[H,\mathbf M]=0,\quad [L_i,M_j]=\ii\hbar\epsilon_{ijk}M_k,\quad
	[M_i,M_j]=-\frac{2\ii\hbar H}{m}\epsilon_{ijk}L_k}.
	\]
	\end{solution}

\begin{question}
伽利略变换把参考系变到相对原参考系以速度 $v$ 运动的新参考系。
\begin{enumerate}
\item 求无穷小 boost 下 $x$ 与 $p$ 的变换，并解释其物理意义。
\item 利用 Baker--Campbell--Hausdorff 公式整理有限 boost 算符的形式。
\item 证明若 $\psi$ 满足势 $V(x)$ 下的薛定谔方程，则 boost 后的波函数满足相应移动势场中的薛定谔方程。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{伽利略 boost 算符}：考虑从参考系 $S$ 变换到以速度 $v$ 沿 $x$ 方向运动的参考系 $S'$。在量子力学中，伽利略 boost 算符 $G(v,t)$ 作用于波函数为
	\[
	\psi'(x,t) = G(v,t)\psi(x,t).
	\]

	经典伽利略变换为 $x'=x-vt$，$t'=t$，且波函数变换应包含一个相位因子以保证薛定谔方程协变：
	\[
	G(v,t) = \exp\!\left[-\frac{\ii}{\hbar}\left(mvx - \frac12 mv^2 t\right)\right]\,T(vt),
	\]
	其中 $T(a)=\exp(-\ii a p/\hbar)$ 是空间平移算符。

	\textbf{无穷小 boost}：对无穷小速度 $v\to\delta v$，展开到一阶：
	\begin{itemize}
	\item 相位部分：
	\[
	\exp\!\left[-\frac{\ii}{\hbar}\left(m\delta v\,x - \frac12 m(\delta v)^2 t\right)\right]
	\approx 1 - \frac{\ii m\delta v}{\hbar}x + O(\delta v^2).
	\]
	\item 平移部分 $T(\delta v\,t) = 1 - \frac{\ii\delta v\,t}{\hbar}p + O(\delta v^2)$。
	\end{itemize}

	因此
	\[
	G(\delta v,t) = 1 - \frac{\ii\delta v}{\hbar}(mx + tp) + O(\delta v^2).
	\]

	\textbf{计算 $x$ 和 $p$ 的变换}：利用 $G^\dagger x G$ 和 $G^\dagger p G$，其中 $G^\dagger = 1 + \frac{\ii\delta v}{\hbar}(mx+tp)$。

	对 $x$：
	\[
	x' = G^\dagger x G
	= \left(1+\frac{\ii\delta v}{\hbar}(mx+tp)\right) x \left(1-\frac{\ii\delta v}{\hbar}(mx+tp)\right) + O(\delta v^2)
	= x + \frac{\ii\delta v}{\hbar}[(mx+tp),x] + O(\delta v^2).
	\]

	计算对易子：
	\[
	[(mx+tp),x] = m[x,x] + t[p,x] = 0 + t(-\ii\hbar) = -\ii\hbar t,
	\]
	所以
	\[
	x' = x + \frac{\ii\delta v}{\hbar}(-\ii\hbar t) = x - \delta v\,t.
	\]

	对 $p$：
	\[
	p' = G^\dagger p G = p + \frac{\ii\delta v}{\hbar}[(mx+tp),p] + O(\delta v^2).
	\]

	计算对易子：
	\[
	[(mx+tp),p] = m[x,p] + t[p,p] = m(\ii\hbar) + 0 = \ii\hbar m,
	\]
	所以
	\[
	p' = p + \frac{\ii\delta v}{\hbar}(\ii\hbar m) = p - m\,\delta v.
	\]

	对于有限速度 $v$，积分得
	\[
	\boxed{x' = x - vt},\qquad
	\boxed{p' = p - mv}.
	\]

	这正是经典的伽利略变换：新参考系中位置坐标少了 $vt$，动量少了 $mv$（相当于减去参考系本身的动量）。

	\textbf{BCH 公式合并}：有限 boost 算符 $G(v,t)$ 的形式可写为一个相位乘平移算符。利用 Baker--Campbell--Hausdorff 公式整理指数算符可以验证该形式使薛定谔方程在伽利略变换下协变。具体地，若 $\psi(x,t)$ 满足势 $V(x)$ 下的薛定谔方程，则 $\psi'(x,t)=G(v,t)\psi(x,t)$ 满足势 $V(x+vt)$ 下的薛定谔方程（相当于势场随参考系移动）。

	因此伽利略 boost 算符的完整形式为
	\[
	\boxed{G(v,t)=\exp\!\left[-\frac{\ii}{\hbar}\left(mvx-\frac12 mv^2t\right)\right]\,\exp\!\left(-\frac{\ii v t}{\hbar}p\right)}.
	\]
	\end{solution}

\begin{question}
讨论时间反演不变性。以无空气阻力下抛体运动为例说明：若在某一时刻把速度反向，粒子会沿原轨迹反向返回。把这一思想推广到量子力学，说明时间反演对 $x,p$ 和虚数单位 $\ii$ 的作用。
\end{question}
\begin{solution}
	\textbf{经典力学中的时间反演}：以无阻力抛体运动为例。质点在重力场中沿抛物线轨迹运动。若在某一时刻 $t_0$ 将速度反向（$v\to -v$），则此后质点沿原轨迹反向返回，经过 $t_0$ 之前走过的路径回到起点。

	经典时间反演变换为：
	\[
	t \to -t,\quad x \to x,\quad \dot x \to -\dot x.
	\]
	牛顿方程 $m\ddot x = -\nabla V(x)$ 在 $t\to -t$ 下不变（因为 $\ddot x$ 是二阶导数），所以时间反演是经典力学的对称性。

	\textbf{量子力学中的时间反演}：在量子力学中，时间反演算符 $\Theta$ 是一个反线性（anti-linear）算符，其作用为：
	\begin{itemize}
	\item 位置算符不变：
	\[
	\Theta x\Theta^{-1} = x.
	\]
	\item 动量算符变号：
	\[
	\Theta p\Theta^{-1} = -p.
	\]
	这是因为 $p=m\dd x/\dd t$，时间反演下 $\dd/\dd t \to -\dd/\dd t$。
	\item 虚数单位 $\ii$ 变号（反线性性质）：
	\[
	\Theta \ii\Theta^{-1} = -\ii.
	\]
	\end{itemize}

	验证动量算符的变换：$p = -\ii\hbar\frac{\dd}{\dd x}$，
	\[
	\Theta p \Theta^{-1} = \Theta(-\ii\hbar\partial_x)\Theta^{-1}
	= (-\Theta\ii\Theta^{-1})(\hbar\partial_x) = -(-\ii)\hbar\partial_x = \ii\hbar\partial_x = -p.
	\]

	\textbf{时间反演不变性}：若势能 $V(x)$ 是实函数且不含速度项，则哈密顿量
	\[
	H = \frac{p^2}{2m} + V(x)
	\]
	在时间反演下不变：
	\[
	\Theta H \Theta^{-1} = \frac{(\Theta p\Theta^{-1})^2}{2m} + V(x)
	= \frac{(-p)^2}{2m} + V(x) = H.
	\]

	因此若 $\psi(x,t)$ 满足薛定谔方程，则 $\Theta\psi(x,t)$ 也满足同一方程。注意由于 $\Theta$ 的反线性性质，$\Theta\psi(x,t) = \psi^*(x,-t)$：
	\[
	\psi(x,t) \to \psi^*(x,-t).
	\]

	\textbf{物理意义}：时间反演对称性保证了若存在一个正向时间演化的解，则存在一个反向时间演化的解——即"电影倒放"是物理上允许的。复共轭的出现正是为了使 $p=-\ii\hbar\partial_x$ 在时间反演下变号：
	\[
	\boxed{\Theta x\Theta^{-1}=x,\quad \Theta p\Theta^{-1}=-p,\quad \Theta\ii\Theta^{-1}=-\ii}.
	\]
	\end{solution}

\begin{question}
自旋 $1/2$ 粒子的时间反演算符可取为 $\Theta=-\ii\sigma_yK$，其中 $K$ 表示复共轭。
\begin{enumerate}
\item 证明 $\Theta^2=-1$。
\item 若哈密顿量与时间反演对易，证明所有能级至少二重简并（Kramers 简并）。
\end{enumerate}
\end{question}
\begin{solution}
	\textbf{自旋 $1/2$ 的时间反演算符}：对于自旋 $1/2$ 粒子，时间反演算符取为
	\[
	\Theta = -\ii\sigma_y K,
	\]
	其中 $K$ 是复共轭算符（$K\psi=\psi^*$），$\sigma_y$ 是泡利矩阵。

	在 $S_z$ 基下，$\sigma_y = \begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix}$，所以
	\[
	\Theta = -\ii\begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix}K
	= \begin{pmatrix}0&-1\\1&0\end{pmatrix}K.
	\]

	\textbf{(a) 证明 $\Theta^2 = -1$}：在 $S_z$ 标准基 $|+\rangle=\begin{pmatrix}1\\0\end{pmatrix}$、$|-\rangle=\begin{pmatrix}0\\1\end{pmatrix}$ 上直接计算。

	$\Theta = -\ii\sigma_y K$，其中 $\sigma_y = \begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix}$。先计算 $\Theta$ 对基矢的作用：
	\[
	\Theta|+\rangle = -\ii\sigma_y K|+\rangle = -\ii\sigma_y|+\rangle
	= -\ii\begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix}\begin{pmatrix}1\\0\end{pmatrix}
	= -\ii\begin{pmatrix}0\\\ii\end{pmatrix} = \begin{pmatrix}0\\1\end{pmatrix} = |-\rangle.
	\]
	\[
	\Theta|-\rangle = -\ii\sigma_y K|-\rangle = -\ii\sigma_y|-\rangle
	= -\ii\begin{pmatrix}0&-\ii\\\ii&0\end{pmatrix}\begin{pmatrix}0\\1\end{pmatrix}
	= -\ii\begin{pmatrix}-\ii\\0\end{pmatrix} = \begin{pmatrix}-1\\0\end{pmatrix} = -|+\rangle.
	\]

	现在计算 $\Theta^2$：
	\[
	\Theta^2|+\rangle = \Theta(|+\rangle) = \Theta|-\rangle = -|+\rangle,
	\]
	\[
	\Theta^2|-\rangle = \Theta(-|+\rangle) = -\Theta|+\rangle = -|-\rangle.
	\]

	因此对任意旋量 $\psi$，
	\[
	\Theta^2\psi = -\psi,
	\]
	即 $\Theta^2 = -I$，亦即
	\[
	\boxed{\Theta^2 = -1}.
	\]

	更一般地，对于自旋 $s$ 的系统，时间反演算符满足 $\Theta^2 = (-1)^{2s}$。对半整数自旋（$s=1/2,3/2,\ldots$）有 $\Theta^2=-1$；对整数自旋有 $\Theta^2=+1$。

		\textbf{(b) Kramers 简并}：设哈密顿量 $H$ 与时间反演算符 $\Theta$ 对易：$[\Theta,H]=0$。设 $|\psi\rangle$ 是 $H$ 的本征态：$H|\psi\rangle=E|\psi\rangle$。

		那么 $H(\Theta|\psi\rangle) = \Theta H|\psi\rangle = E(\Theta|\psi\rangle)$，所以 $\Theta|\psi\rangle$ 也是同一能量的本征态。

		现在证明 $|\psi\rangle$ 与 $\Theta|\psi\rangle$ 是线性无关的。假设两者成比例：
		\[
		\Theta|\psi\rangle = c|\psi\rangle,
		\]
		其中 $c$ 是常数。两边再作用 $\Theta$：
		\[
		\Theta^2|\psi\rangle = \Theta(c|\psi\rangle) = c^*\Theta|\psi\rangle = c^*c|\psi\rangle = |c|^2|\psi\rangle.
		\]

		但由 (a) 知 $\Theta^2=-1$，所以
		\[
		-|\psi\rangle = |c|^2|\psi\rangle \quad\Longrightarrow\quad |c|^2 = -1,
		\]
		这不可能（因为 $|c|^2\ge0$）。因此假设不成立，$|\psi\rangle$ 与 $\Theta|\psi\rangle$ 是不同态。

		因此每个能量本征值 $E$ 至少对应两个不同的本征态 $|\psi\rangle$ 和 $\Theta|\psi\rangle$，即所有能级至少是二重简并的。这就是 Kramers 简并：
		\[
		\boxed{\text{在时间反演对称的系统中，半整数自旋系统的所有能级至少二重简并}}.
		\]

		这个结论对固体物理中自旋轨道耦合系统有重要应用。
		\end{solution}
