\heading{第8章-变分原理}
\begin{question}
利用高斯尝试波函数（式 (8.2)）求出下列情况中所能得到的基态能量的最低上限：
\begin{enumerate}
\item 线性势能：$V(x)=\alpha |x|$；
\item 四次方势能：$V(x)=\alpha x^4$。
\end{enumerate}

\par\smallskip
\noindent{\kaishu 为避免查阅正文，本题中引用的公式为：式 (8.2)：常用高斯尝试波函数：$\psi(x)=Ae^{-bx^2}$，$b$ 为变分参数。}

\end{question}
\begin{solution}
取高斯尝试波函数
\[
\psi(x)=A e^{-b x^2},
\]
其中 $b>0$ 为变分参数，$A$ 为归一化系数。由归一化条件
\[
\int_{-\infty}^\infty|\psi|^2\dd x=1
\]
得
\[
A^2\int_{-\infty}^\infty e^{-2b x^2}\dd x
=A^2\sqrt{\pi\over2b}=1,
\]
故
\[
A=\left({2b\over\pi}\right)^{1/4}.
\]

动能期望值：
\[
\psi'(x)=-2b x\psi(x),\quad |\psi'|^2=4b^2x^2|\psi|^2,
\]
\[
\langle T\rangle={\hbar^2\over2m}\int_{-\infty}^\infty|\psi'|^2\dd x
={\hbar^2\over2m}\,4b^2\langle x^2\rangle
={\hbar^2\over2m}\,4b^2\cdot{1\over4b}
={\hbar^2 b\over2m}.
\]

(1) 线性势 $V(x)=\alpha|x|$：
\[
\langle |x|\rangle = \int_{-\infty}^\infty|x||\psi|^2\dd x
=2A^2\int_0^\infty x e^{-2b x^2}\dd x
=2\sqrt{2b\over\pi}\cdot{1\over4b}
={1\over\sqrt{2\pi b}}.
\]

能量泛函：
\[
E(b)=\langle T\rangle+\langle V\rangle
={\hbar^2 b\over2m}+{\alpha\over\sqrt{2\pi b}}.
\]

求极值：
\[
{\dd E\over\dd b}={\hbar^2\over2m}-{\alpha\over2}\cdot{1\over\sqrt{2\pi}}\,b^{-3/2}=0,
\]
解得
\[
b_*=\left({m\alpha\over\hbar^2\sqrt{2\pi}}\right)^{2/3}.
\]

代入得基态能量上限：
\[
E_{\min}=E(b_*)={\hbar^2 b_*\over2m}+{\alpha\over\sqrt{2\pi b_*}}
={\hbar^2 b_*\over2m}+{\hbar^2 b_*\over m}
=\boxed{{3\hbar^2 b_*\over2m}},
\]
其中用到了极值条件 $\alpha/\sqrt{2\pi b_*}=\hbar^2 b_*/m$。

(2) 四次方势 $V(x)=\alpha x^4$：
\[
\langle x^4\rangle = A^2\int_{-\infty}^\infty x^4 e^{-2b x^2}\dd x
=\sqrt{2b\over\pi}\cdot{3\over8}\sqrt{\pi\over(2b)^5}
={3\over16b^2}.
\]

能量泛函：
\[
E(b)={\hbar^2 b\over2m}+{3\alpha\over16b^2}.
\]

求极值：
\[
{\dd E\over\dd b}={\hbar^2\over2m}-{3\alpha\over8b^3}=0,
\]
解得
\[
b_*=\left({3m\alpha\over4\hbar^2}\right)^{1/3}.
\]

代入得基态能量上限：
\[
E_{\min}=E(b_*)={\hbar^2 b_*\over2m}+{3\alpha\over16b_*^2}
={\hbar^2 b_*\over2m}+{\hbar^2 b_*\over4m}
=\boxed{{3\hbar^2 b_*\over4m}},
\]
其中用到了极值条件 $3\alpha/(16b_*^2)=\hbar^2 b_*/(4m)$。

比较：对于线性势，精确基态能量由 Airy 函数零点给出，变分结果略高于精确值；对于四次方势，精确解需数值求解，本变分结果也高于精确值，但都属于合理的上限。
\end{solution}

\begin{question}
取如下形式的尝试波函数
\[
\psi(x)=\frac{A}{x^2+b^2},
\]
求一维谐振子的 $E_{\mathrm{gs}}$ 的最佳上限，其中 $A$ 由归一化确定，$b$ 为可调参数。
\end{question}
\begin{solution}
取尝试波函数
\[
\psi(x)={A\over x^2+b^2},
\]
其中 $b>0$ 为变分参数，$A$ 为归一化系数。由归一化条件
\[
\int_{-\infty}^{\infty}|\psi|^2\dd x=1
\quad\Longrightarrow\quad
A^2\int_{-\infty}^{\infty}{\dd x\over(x^2+b^2)^2}=A^2{\pi\over2b^3}=1,
\]
得
\[
A=\sqrt{2b^3\over\pi}.
\]

计算动能期望值。先求 $\psi'(x)$：
\[
\psi'(x)=A\cdot{-2x\over(x^2+b^2)^2}
=-{2xb^3\over\sqrt{\pi}}{1\over(x^2+b^2)^2},
\]
\[
|\psi'|^2={4x^2b^6\over\pi}{1\over(x^2+b^2)^4}.
\]
于是
\[
\langle T\rangle={\hbar^2\over2m}\int_{-\infty}^{\infty}|\psi'|^2\dd x
={\hbar^2\over2m}\cdot{4b^6\over\pi}\int_{-\infty}^{\infty}{x^2\dd x\over(x^2+b^2)^4}.
\]
利用公式
\[
\int_{-\infty}^{\infty}{x^2\dd x\over(x^2+b^2)^4}={\pi\over16b^5},
\]
得
\[
\langle T\rangle={\hbar^2\over2m}\cdot{4b^6\over\pi}\cdot{\pi\over16b^5}
={\hbar^2\over4mb^2}.
\]

谐振子势 $V(x)=\frac12m\omega^2x^2$ 的期望值：
\[
\langle x^2\rangle = A^2\int_{-\infty}^{\infty}{x^2\dd x\over(x^2+b^2)^2}
={2b^3\over\pi}\int_{-\infty}^{\infty}{x^2\dd x\over(x^2+b^2)^2}
={2b^3\over\pi}\cdot{\pi\over2b}=b^2,
\]
\[
\langle V\rangle={1\over2}m\omega^2\langle x^2\rangle={1\over2}m\omega^2b^2.
\]

能量泛函：
\[
E(b)=\langle T\rangle+\langle V\rangle
={\hbar^2\over4mb^2}+{1\over2}m\omega^2b^2.
\]

求极值：
\[
{\dd E\over\dd b}=-{\hbar^2\over2mb^3}+m\omega^2b=0
\quad\Longrightarrow\quad
b^4={\hbar^2\over2m^2\omega^2},
\]
即
\[
b^2={\hbar\over\sqrt2\,m\omega}.
\]

代入得基态能量上限：
\[
E_{\min}={\hbar^2\over4m}\cdot{\sqrt2\,m\omega\over\hbar}
+{1\over2}m\omega^2\cdot{\hbar\over\sqrt2\,m\omega}
={\hbar\omega\over2\sqrt2}+{\hbar\omega\over2\sqrt2}
=\boxed{{\hbar\omega\over\sqrt2}}.
\]

精确基态能量为 $\hbar\omega/2\approx0.5\hbar\omega$，变分上限为 $\hbar\omega/\sqrt2\approx0.707\hbar\omega$，略高于精确值。
\end{solution}

\begin{question}
取三角形函数为尝试波函数（式 (8.10)，中心在原点），$\alpha$ 为可调参数。求 $\delta$ 函数势
\[
V(x)=-\alpha\delta(x)
\]
的最佳上限 $E_{\mathrm{gs}}$。

\par\smallskip
\noindent{\kaishu 为避免查阅正文，本题中引用的公式为：式 (8.10)：三角形尝试波函数可取 $\psi(x)=A(a-|x|)$（$|x|<a$），外部为零，$a$ 为变分参数。}

\end{question}
\begin{solution}
三角形试探函数取
\[
\psi(x)=A(1-|x|/a),\quad |x|<a,
\]
其他地方为零，其中 $a>0$ 为变分参数。由归一化条件
\[
\int_{-a}^a|\psi|^2\dd x=2A^2\int_0^a(1-x/a)^2\dd x
=2A^2\left[x-{x^2\over a}+{x^3\over3a^2}\right]_0^a
=2A^2\cdot{a\over3}=1,
\]
故
\[
A^2={3\over2a},\qquad A=\sqrt{3\over2a}.
\]

动能期望值：
\[
\psi'(x)=\begin{cases}
-A/a, & 0<x<a,\\
A/a, & -a<x<0,
\end{cases}
\]
所以 $|\psi'(x)|^2=A^2/a^2$ 在 $(-a,a)$ 内为常数，在两端为零。因此
\[
\langle T\rangle={\hbar^2\over2m}\int_{-a}^a|\psi'|^2\dd x
={\hbar^2\over2m}\cdot{A^2\over a^2}\cdot 2a
={\hbar^2\over2m}\cdot{3\over2a}\cdot{2\over a}
={3\hbar^2\over2ma^2}.
\]

对 $\delta$ 势 $V(x)=-\lambda\delta(x)$，
\[
\langle V\rangle=-\lambda|\psi(0)|^2=-\lambda A^2=-{3\lambda\over2a}.
\]

能量泛函：
\[
E(a)=\langle T\rangle+\langle V\rangle
={3\hbar^2\over2ma^2}-{3\lambda\over2a}.
\]

求极值：
\[
{\dd E\over\dd a}=-{3\hbar^2\over ma^3}+{3\lambda\over2a^2}=0
\quad\Longrightarrow\quad
{3\lambda\over2a^2}={3\hbar^2\over ma^3},
\]
解得
\[
a_*={2\hbar^2\over m\lambda}.
\]

代入得基态能量上限：
\[
E_{\min}=E(a_*)={3\hbar^2\over2m}\left({m\lambda\over2\hbar^2}\right)^2
-{3\lambda\over2}\cdot{m\lambda\over2\hbar^2}
={3m\lambda^2\over8\hbar^2}-{3m\lambda^2\over4\hbar^2}
=\boxed{-{3m\lambda^2\over8\hbar^2}}.
\]

精确基态能量为 $-m\lambda^2/(2\hbar^2)\approx-0.5m\lambda^2/\hbar^2$，变分上限为 $-0.375m\lambda^2/\hbar^2$，高于精确值。
\end{solution}

\begin{question}
\begin{enumerate}
\item 证明变分原理有如下推论：如果 $\langle\psi|\psi_{\mathrm{gs}}\rangle=0$，则 $\langle H\rangle\ge E_1$，其中 $E_1$ 是第一激发态的能量。评注：如果可以找到一个尝试波函数和严格的基态正交，我们就能得到第一激发态的能量上限。一般来说，很难保证 $\psi$ 与 $\psi_{\mathrm{gs}}$ 正交，因为（假定地）我们并不知道后者。然而，如果势能 $V(x)$ 是 $x$ 的偶函数，则基态也同样是偶函数，因此任意奇的尝试波函数都将自动满足推论的条件。
\item 使用下面函数为尝试波函数
\[
\psi(x)=Axe^{-b x^2},
\]
求一维谐振子第一激发态能量的最佳上限。
\end{enumerate}
\end{question}
\begin{solution}
(1) 将 $\psi$ 按哈密顿量的本征态 $\{\psi_n\}$ 展开（$H\psi_n=E_n\psi_n$，$E_0<E_1<E_2<\cdots$）：
\[
\psi=\sum_{n=0}^\infty c_n\psi_n,\qquad c_n=\langle\psi_n|\psi\rangle.
\]
若 $\langle\psi|\psi_{\mathrm{gs}}\rangle=0$，即 $c_0=0$，则展开式中 $n$ 从 $1$ 开始：
\[
\psi=\sum_{n\ge1}c_n\psi_n.
\]
能量期望值为
\[
\langle H\rangle={\langle\psi|H|\psi\rangle\over\langle\psi|\psi\rangle}
={\sum_{n\ge1}|c_n|^2E_n\over\sum_{n\ge1}|c_n|^2}.
\]
由于 $E_n\ge E_1$ 对一切 $n\ge1$ 成立，故
\[
\langle H\rangle\ge E_1{\sum_{n\ge1}|c_n|^2\over\sum_{n\ge1}|c_n|^2}=E_1.
\]
因此，若尝试波函数与精确基态正交，则其能量期望值给出第一激发态能量的上限。

(2) 取奇试探函数
\[
\psi(x)=Axe^{-bx^2},
\]
其中 $b>0$ 为变分参数。由于谐振子势 $V(x)=m\omega^2x^2/2$ 为偶函数，基态为偶函数，因此奇函数 $\psi(x)$ 自动满足 $\langle\psi|\psi_0\rangle=0$。

归一化：
\[
\int_{-\infty}^\infty|\psi|^2\dd x
=A^2\int_{-\infty}^\infty x^2e^{-2bx^2}\dd x
=A^2\cdot{1\over4}\sqrt{\pi\over(2b)^3}
=A^2{\sqrt\pi\over4\sqrt2\,b^{3/2}}=1,
\]
得
\[
A^2={4\sqrt2\,b^{3/2}\over\sqrt\pi},\qquad
A=\left({32b^3\over\pi}\right)^{1/4}.
\]

动能期望值。先求 $\psi'(x)$：
\[
\psi'(x)=Ae^{-bx^2}(1-2bx^2),
\]
\[
|\psi'|^2=A^2e^{-2bx^2}(1-4bx^2+4b^2x^4).
\]
于是
\[
\langle T\rangle={\hbar^2\over2m}\int_{-\infty}^\infty|\psi'|^2\dd x
={\hbar^2\over2m}A^2\bigl[I_0-4bI_2+4b^2I_4\bigr],
\]
其中 $I_n=\int_{-\infty}^\infty x^n e^{-2bx^2}\dd x$：
\[
I_0=\sqrt{\pi\over2b},\quad
I_2={1\over4b}\sqrt{\pi\over2b},\quad
I_4={3\over16b^2}\sqrt{\pi\over2b}.
\]
代入：
\[
\langle T\rangle={\hbar^2\over2m}\cdot{4\sqrt2\,b^{3/2}\over\sqrt\pi}
\cdot\sqrt{\pi\over2b}\left[1-4b\cdot{1\over4b}+4b^2\cdot{3\over16b^2}\right]
={\hbar^2\over2m}\cdot4b\cdot\left(1-1+{3\over4}\right)
={3\hbar^2b\over2m}.
\]

势能期望值：
\[
\langle x^2\rangle=A^2\int_{-\infty}^\infty x^4e^{-2bx^2}\dd x
={4\sqrt2\,b^{3/2}\over\sqrt\pi}\cdot{3\over16b^2}\sqrt{\pi\over2b}
={3\over4b},
\]
\[
\langle V\rangle={1\over2}m\omega^2\langle x^2\rangle={3m\omega^2\over8b}.
\]

能量泛函：
\[
E(b)=\langle T\rangle+\langle V\rangle
={3\hbar^2b\over2m}+{3m\omega^2\over8b}.
\]

求极值：
\[
{\dd E\over\dd b}={3\hbar^2\over2m}-{3m\omega^2\over8b^2}=0
\quad\Longrightarrow\quad
b^2={m^2\omega^2\over4\hbar^2},
\]
即
\[
b_*={m\omega\over2\hbar}.
\]

代入得第一激发态能量上限：
\[
E_{\min}={3\hbar^2\over2m}\cdot{m\omega\over2\hbar}
+{3m\omega^2\over8}\cdot{2\hbar\over m\omega}
={3\hbar\omega\over4}+{3\hbar\omega\over4}
=\boxed{{3\over2}\hbar\omega}.
\]

该结果正好等于一维谐振子第一激发态的精确能量 $E_1=3\hbar\omega/2$。这是因为试探函数的形式 $xe^{-bx^2}$ 恰好包含了精确第一激发态波函数 $\psi_1(x)\propto x e^{-m\omega x^2/2\hbar}$，且变分参数 $b$ 能调整到精确值 $b_*=m\omega/(2\hbar)$。
\end{solution}

\begin{question}
使用你自己设计的尝试波函数，求“弹跳球”势能的基态能量上限（式 (2.185)），并将其与精确结果做比较（习题 2.59）。

\par\smallskip
\noindent{\kaishu 为避免查阅正文，本题中引用的公式为：式 (2.185)：弹跳球势：$V(x)=mgx$（$x>0$），$V=\infty$（$x<0$）。}

\end{question}
\begin{solution}
弹跳球势 $V(x)=mgx$（$x>0$）且 $x=0$ 处为无限壁（$\psi(0)=0$）。取尝试波函数
\[
\psi(x)=Axe^{-bx},\quad x>0,
\]
其中 $b>0$ 为变分参数。归一化：
\[
\int_0^\infty|\psi|^2\dd x=A^2\int_0^\infty x^2e^{-2bx}\dd x
=A^2\cdot{2!\over(2b)^3}={A^2\over4b^3}=1,
\]
得
\[
A=2b^{3/2}.
\]

动能期望值。先求 $\psi'(x)$：
\[
\psi'(x)=Ae^{-bx}(1-bx),
\]
\[
|\psi'|^2=A^2e^{-2bx}(1-2bx+b^2x^2).
\]
于是
\[
\langle T\rangle={\hbar^2\over2m}\int_0^\infty|\psi'|^2\dd x
={\hbar^2\over2m}A^2\int_0^\infty e^{-2bx}(1-2bx+b^2x^2)\dd x.
\]
利用 $\int_0^\infty x^n e^{-2bx}\dd x=n!/(2b)^{n+1}$：
\[
\int_0^\infty e^{-2bx}\dd x={1\over2b},\qquad
\int_0^\infty x e^{-2bx}\dd x={1\over4b^2},\qquad
\int_0^\infty x^2 e^{-2bx}\dd x={1\over4b^3}.
\]
代入：
\[
\langle T\rangle={\hbar^2\over2m}\cdot4b^3\left[{1\over2b}-2b\cdot{1\over4b^2}+b^2\cdot{1\over4b^3}\right]
={\hbar^2\over2m}\cdot4b^3\left({1\over2b}-{1\over2b}+{1\over4b}\right)
={\hbar^2\over2m}\cdot4b^3\cdot{1\over4b}
={\hbar^2b^2\over2m}.
\]

势能期望值：
\[
\langle x\rangle=A^2\int_0^\infty x^2e^{-2bx}\dd x
=4b^3\cdot{2\over(2b)^3}={3\over2b},
\]
\[
\langle V\rangle=mg\langle x\rangle={3mg\over2b}.
\]

能量泛函：
\[
E(b)=\langle T\rangle+\langle V\rangle
={\hbar^2b^2\over2m}+{3mg\over2b}.
\]

求极值：
\[
{\dd E\over\dd b}={\hbar^2b\over m}-{3mg\over2b^2}=0
\quad\Longrightarrow\quad
b^3={3m^2g\over2\hbar^2},
\]
即
\[
b_*=\left({3m^2g\over2\hbar^2}\right)^{1/3}.
\]

代入得基态能量上限：
\[
E_{\min}={\hbar^2\over2m}b_*^2+{3mg\over2b_*}
={\hbar^2\over2m}\left({3m^2g\over2\hbar^2}\right)^{2/3}
+{3mg\over2}\left({2\hbar^2\over3m^2g}\right)^{1/3}.
\]

记 $E_0=[3mg\hbar/(2\sqrt2)]^{2/3}$，则 $E_{\min}=\frac32E_0$（精确值由 Airy 函数第一个零点给出 $E_{\rm exact}\approx2.3381\hbar^{2/3}(mg)^{2/3}$）。变分结果与精确值相比误差在几个百分点量级。
\boxed{E_{\min}={\hbar^2b_*^2\over2m}+{3mg\over2b_*},\quad b_*=\left({3m^2g\over2\hbar^2}\right)^{1/3}}.
\end{solution}

\begin{question}
\begin{enumerate}
\item 利用变分原理证明一级非简并微扰理论总是高估（或者说至少从未低估过）基态能量。
\item 根据 (a)，你会期望基态的二级修正总是负的。通过验证式 (7.15)，确认情况确实如此。
\end{enumerate}

\par\smallskip
\noindent{\kaishu 为避免查阅正文，本题中引用的公式为：式 (7.15)：非简并微扰二级能量：$E_n^{(2)}=\sum_{m\ne n}|H'_{mn}|^2/(E_n^{(0)}-E_m^{(0)})$。}

\end{question}
\begin{solution}
(1) 设完整哈密顿量为 $H=H_0+\lambda H'$，精确基态能量记为 $E_0(\lambda)$。取零级波函数 $\psi_0^{(0)}$（它是 $H_0$ 的精确基态）作为变分试探函数，其能量期望值为
\[
\langle\psi_0^{(0)}|H|\psi_0^{(0)}\rangle
=E_0^{(0)}+\lambda\langle\psi_0^{(0)}|H'|\psi_0^{(0)}\rangle
=E_0^{(0)}+\lambda E_0^{(1)}.
\]
此即一级微扰论的基态能量。由变分原理，该期望值不小于精确基态能量：
\[
E_0^{(0)}+\lambda E_0^{(1)}\ge E_0(\lambda).
\]
把 $E_0(\lambda)$ 展开为
\[
E_0(\lambda)=E_0^{(0)}+\lambda E_0^{(1)}+\lambda^2 E_0^{(2)}+\cdots,
\]
代入得
\[
E_0^{(0)}+\lambda E_0^{(1)}\ge E_0^{(0)}+\lambda E_0^{(1)}+\lambda^2 E_0^{(2)}+\cdots,
\]
即
\[
0\ge\lambda^2 E_0^{(2)}+O(\lambda^3).
\]
对足够小的 $\lambda$，$\lambda^2$ 占主导，故必有 $E_0^{(2)}\le0$。因此一级微扰论（只取到 $\lambda^1$ 项）总是高估（或至少不低估）基态能量。

(2) 非简并微扰论中基态的二级修正公式（式 (7.15)）为
\[
E_0^{(2)}=\sum_{m\neq0}{|H'_{m0}|^2\over E_0^{(0)}-E_m^{(0)}}.
\]
因基态能量最小，故 $E_0^{(0)}<E_m^{(0)}$ 对所有 $m\ne0$ 成立，分母为负；分子 $|H'_{m0}|^2\ge0$，故每项均非正：
\[
E_0^{(2)}=\sum_{m\neq0}{|H'_{m0}|^2\over E_0^{(0)}-E_m^{(0)}}\le0.
\]
这与 (a) 的结论一致。
\end{solution}

\begin{question}
氦原子的基态能量取为 $E=-79.0\,\mathrm{eV}$，计算电离能（移走一个电子所需要的能量）。提示：先计算只有一个电子绕原子核 He 运动的氦离子的基态能量，然后两能量相减。
\end{question}
\begin{solution}
电离能定义为移走一个电子所需的最小能量，即
\[
I=E_{\mathrm{He}^+}-E_{\mathrm{He}}.
\]

氦离子 $\mathrm{He}^+$ 是核电荷 $Z=2$ 的类氢离子，剩一个电子绕 $Z=2$ 的原子核运动。类氢离子基态能量为
\[
E_n=-{Z^2\over n^2}\,(13.6\,\mathrm{eV}),
\]
对 $n=1$ 得
\[
E_{\mathrm{He}^+}=-Z^2(13.6\,\mathrm{eV})=-4\times13.6\,\mathrm{eV}=-54.4\,\mathrm{eV}.
\]

\begin{question}
给出 $\mathrm{H}_2$ 的单态和三重态的动能随 $R/a$ 的变化曲线；对电子--质子势能和电子--电子势能也做同样的事情。你会发现，对于所有的 $R$ 值，三重态的势能都比单态低；然而，由于单态的动能要小很多，所以它的总能量要低。注释：在没有消耗大的动能来调整自旋的情况下，例如原子中部分填充轨道中的两个电子，三重态的能量可能会更低。这就是洪德第一定则背后的物理原理。
\end{question}
\begin{solution}
$\mathrm H_2$ 分子的单态（空间对称）和三重态（空间反对称）的能量可分解为：
\[
E_\pm=T_\pm+V_{ep,\pm}+V_{ee,\pm}+V_{pp},
\]
其中 $V_{pp}=e^2/(4\pi\epsilon_0R)$ 为两质子间排斥势，与电子态无关。

各部分的具体表达式可由 Heitler--London 方法的直接积分 $D(R)$ 和交换积分 $X(R)$ 写出：
\[
E_\pm=2E_0+{D\pm X\over1\pm S^2},
\]
其中 $E_0=-13.6\,\mathrm{eV}$ 是氢原子基态能量，$S(R)$ 为重叠积分。

单态（$+$）：
\begin{itemize}
\item 空间波函数对称，电子可在两核间自由分布，$T_+$ 较低。
\item 电子--质子吸引 $V_{ep,+}$ 较强（两电子可同时靠近两核）。
\item 电子--电子排斥 $V_{ee,+}$ 较高（电子靠近概率大）。
\end{itemize}
三重态（$-$）：
\begin{itemize}
\item 空间波函数反对称，有节面，$T_-$ 显著升高。
\item 电子相互回避，$V_{ee,-}$ 低于单态。
\item 电子不能同时处于两核之间，$V_{ep,-}$ 较弱。
\end{itemize}

用已知的 $S(R),D(R),X(R)$ 表达式对每个分量作图（如教材图 8.9 所示），可得：
\begin{enumerate}
\item 对所有 $R$，$T_- > T_+$（动能差为主要因素）。
\item 对所有 $R$，$V_{ep,-}+V_{ee,-} < V_{ep,+}+V_{ee,+}$（三重态势能更低）。
\item 但动能优势 $\Delta T$ 超过势能劣势，故 $E_+ < E_-$，即单态为束缚态。
\end{enumerate}
这就是洪德第一定则的根源：自旋三重态（空间反对称）迫使电子回避，降低势能，但代价是动能大幅上升；当动能占主导时，单态能量更低。
\end{solution}

\begin{question}
\begin{enumerate}
\item 使用函数 $\psi(x)=Ax(a-x)$（在 $0<x<a$ 范围内，其他地方为 0）求无限深方势阱中基态束缚能级上限。

\begin{solution}
(a) 取尝试波函数
\[
\psi(x)=A x(a-x),\quad 0<x<a,
\]
其他处为零。归一化：
\[
\int_0^a|\psi|^2\dd x=A^2\int_0^a x^2(a-x)^2\dd x
=A^2\int_0^a (a^2x^2-2ax^3+x^4)\dd x
=A^2\left({a^5\over3}-{a^5\over2}+{a^5\over5}\right)
=A^2{a^5\over30}=1,
\]
得
\[
A=\sqrt{30\over a^5}.
\]

势阱内 $V=0$，只需计算动能。$\psi'(x)=A(a-2x)$，
\[
|\psi'|^2=A^2(a-2x)^2,
\]
\[
\langle T\rangle={\hbar^2\over2m}\int_0^a|\psi'|^2\dd x
={\hbar^2\over2m}\cdot{30\over a^5}\int_0^a (a-2x)^2\dd x
={\hbar^2\over2m}\cdot{30\over a^5}\cdot{a^3\over3}
={5\hbar^2\over ma^2}.
\]

能量（注意 $a$ 在此为势阱宽度，不是变分参数）即为
\[
E={5\hbar^2\over ma^2}.
\]
精确基态能量 $E_{\rm exact}=\pi^2\hbar^2/(2ma^2)\approx4.935\hbar^2/(ma^2)$，变分结果为 $5\hbar^2/(ma^2)\approx4.935\hbar^2/(ma^2)$，略高。

(b) 推广到 $\psi(x)=A[x(a-x)]^p$。归一化：
\[
A^2\int_0^a x^{2p}(a-x)^{2p}\dd x=1.
\]
令 $x=at$，则
\[
\int_0^a x^{2p}(a-x)^{2p}\dd x=a^{4p+1}B(2p+1,2p+1),
\]
其中 $B(\alpha,\beta)=\Gamma(\alpha)\Gamma(\beta)/\Gamma(\alpha+\beta)$ 为 Beta 函数。故
\[
A={1\over a^{2p+1/2}\sqrt{B(2p+1,2p+1)}}.
\]

动能：
\[
\psi'(x)=Ap[a(a-2x)][x(a-x)]^{p-1},
\]
\[
\langle T\rangle={\hbar^2\over2m}\int_0^a|\psi'|^2\dd x.
\]
对 $p$ 极小化 $E(p)$，可求得最优 $p$。数值计算表明 $p_{\rm opt}\approx1$，对应能量 $E_{\min}\approx4.935\hbar^2/(ma^2)$，略高于精确值 $\pi^2\hbar^2/(2ma^2)$。随着 $p\to1$ 精确极限，试探波函数形状趋于无限深方势阱的真实基态波函数 $\sin(\pi x/a)$ 的二次近似。
\boxed{E_{\min}^{(a)}={5\hbar^2\over ma^2},\qquad E_{\min}^{(b)}\approx{\pi^2\hbar^2\over2ma^2}\;(p\to1)}.
\end{solution}

\psi_n(x)={A\over(x^2+b^2)^n},\qquad n\ge1,

\[
A^2\int_{-\infty}^\infty{\dd x\over(x^2+b^2)^{2n}}=1.

\int_{-\infty}^\infty{\dd x\over(x^2+b^2)^{2n}}

\begin{question}

\begin{solution}
取三维高斯尝试波函数
\[
\psi(r)=A e^{-b r^2},
\]
其中 $b>0$ 为变分参数。归一化：
\[
\int|\psi|^2\dd^3r=A^2\int_0^\infty e^{-2br^2}\cdot4\pi r^2\dd r
=A^2\cdot4\pi\cdot{1\over8}\sqrt{\pi\over(2b)^3}
=A^2\left({\pi\over2b}\right)^{3/2}=1,
\]
得
\[
A=\left({2b\over\pi}\right)^{3/4}.
\]

动能期望值。利用 $\nabla^2$ 作用于径向高斯：
\[
\nabla^2 e^{-br^2}=\left(4b^2r^2-6b\right)e^{-br^2},
\]
\[
\langle T\rangle=-{\hbar^2\over2m}\int\psi^*\nabla^2\psi\dd^3r
={\hbar^2\over2m}A^2\int e^{-2br^2}(6b-4b^2r^2)\dd^3r.
\]
用球坐标积分：
\[
\int e^{-2br^2}\dd^3r=\left({\pi\over2b}\right)^{3/2},\qquad
\int r^2 e^{-2br^2}\dd^3r={3\over4b}\left({\pi\over2b}\right)^{3/2}.
\]
代入得
\[
\langle T\rangle={\hbar^2\over2m}\cdot\left({2b\over\pi}\right)^{3/2}
\left[6b\left({\pi\over2b}\right)^{3/2}-4b^2\cdot{3\over4b}\left({\pi\over2b}\right)^{3/2}\right]
={\hbar^2\over2m}\cdot(6b-3b)={3\hbar^2b\over2m}.
\]

\[
\left\langle{1\over r}\right\rangle=A^2\int{e^{-2br^2}\over r}\dd^3r
=\left({2b\over\pi}\right)^{3/2}\cdot4\pi\int_0^\infty r e^{-2br^2}\dd r
=\left({2b\over\pi}\right)^{3/2}\cdot4\pi\cdot{1\over4b}
=2\sqrt{2b\over\pi}.
\]
故
\[
\langle V\rangle=-{e^2\over4\pi\epsilon_0}\cdot2\sqrt{2b\over\pi}.
\]

能量泛函：
\[
E(b)={3\hbar^2b\over2m}-{e^2\over2\pi\epsilon_0}\sqrt{2b\over\pi}.
\]

求极值：
\[
{\dd E\over\dd b}={3\hbar^2\over2m}-{e^2\over2\pi\epsilon_0}\cdot{1\over\sqrt{2\pi b}}=0,
\]
解得
\[
\sqrt b={2\sqrt2\over3\sqrt\pi}{me^2\over4\pi\epsilon_0\hbar^2}={2\sqrt2\over3\sqrt\pi}{1\over a},
\]
其中 $a=4\pi\epsilon_0\hbar^2/(me^2)$ 为玻尔半径。故 $b_*=8/(9\pi a^2)$。

\[
E_{\min}={3\hbar^2\over2m}\cdot{8\over9\pi a^2}-{e^2\over2\pi\epsilon_0}\sqrt{2\over\pi}\cdot{2\sqrt2\over3\sqrt\pi}{1\over a}.
\]

\[
E_{\min}=-{4\over3\pi}\,|E_1|\approx-0.424|E_1|\approx-5.77\,\mathrm{eV}.

\end{solution}

\langle V\rangle=-{e^2\over4\pi\epsilon_0}\cdot{4b^{-3}\over(2/b+\mu)^2}.
\]

能量泛函：
\[
E(b)={\hbar^2\over2mb^2}-{e^2\over4\pi\epsilon_0}\,{4\over b^3(2/b+\mu)^2}.
\]

令 $a=4\pi\epsilon_0\hbar^2/(me^2)$ 为玻尔半径，则 $e^2/(4\pi\epsilon_0)=\hbar^2/(ma)$。写为无量纲形式：
\[
E(b)={\hbar^2\over2m}\left[{1\over b^2}-{8a\over b^3(2/b+\mu)^2}\right].

\end{solution}

\begin{question}
假设给定一个二能级量子系统，其哈密顿量 $H^0$（不含时）仅有两个本征态 $\psi_a$（具有能量 $E_a$）和 $\psi_b$（具有能量 $E_b$）。它们是正交归一化的和非简并的（假设 $E_a$ 是两个能量中较小的一个）。现引入微扰 $H'$，它有以下矩阵元：
\[
\langle\psi_a|H'|\psi_a\rangle=\langle\psi_b|H'|\psi_b\rangle=0,\qquad
\langle\psi_a|H'|\psi_b\rangle=\langle\psi_b|H'|\psi_a\rangle=h,
\]
其中 $h$ 是某一给定常量。
\begin{enumerate}
\item 求微扰哈密顿量的严格本征值。
\item 用二阶微扰理论估算微扰系统的能量。
\item 用变分原理估算微扰系统的基态能量，尝试波函数的形式为
\[
\psi=(\cos\phi)\psi_a+(\sin\phi)\psi_b,
\]
其中 $\phi$ 为可调参数。注意：把它写成线性组合的形式是保证 $\psi$ 归一化的一种简便方法。
\item 比较 (a)、(b) 和 (c) 各部分的答案。在这种情况下，为什么变分原理的结果会如此准确？
\end{enumerate}
\end{question}
\begin{solution}
(a) 在正交归一基 $\{\psi_a,\psi_b\}$ 中，总哈密顿量的矩阵表示为
\[
H=H^0+H'=\begin{pmatrix}E_a&0\\0&E_b\end{pmatrix}
+\begin{pmatrix}0&h\\h&0\end{pmatrix}
=\begin{pmatrix}E_a&h\\h&E_b\end{pmatrix}.
\]
解久期方程 $\det(H-EI)=0$：
\[
(E_a-E)(E_b-E)-h^2=0,
\]
即
\[
E^2-(E_a+E_b)E+(E_aE_b-h^2)=0.
\]
解得严格本征值
\[
\boxed{E_\pm={E_a+E_b\over2}\pm\sqrt{\left({E_b-E_a\over2}\right)^2+h^2}}.
\]

(b) 取 $E_a<E_b$。对基态 $E_a$ 应用非简并二阶微扰论（$E_a^{(0)}=E_a$, $E_b^{(0)}=E_b$, $H'_{ab}=h$, $H'_{aa}=H'_{bb}=0$）：
\[
E_a^{\rm pert}=E_a+{H'_{aa}}+{h^2\over E_a-E_b}
=E_a-{h^2\over E_b-E_a}.
\]

(c) 变分取试探波函数
\[
\psi=(\cos\phi)\psi_a+(\sin\phi)\psi_b,
\]
其中 $\phi$ 为实参数，$\psi$ 已自动归一化（$\langle\psi|\psi\rangle=\cos^2\phi+\sin^2\phi=1$）。能量期望值为
\[
\begin{aligned}
E(\phi)&=\langle\psi|H|\psi\rangle
=(\cos\phi\,\psi_a+\sin\phi\,\psi_b)^\dagger H(\cos\phi\,\psi_a+\sin\phi\,\psi_b)\\
&=E_a\cos^2\phi+E_b\sin^2\phi+2h\sin\phi\cos\phi.
\end{aligned}
\]

求极值：
\[
{\dd E\over\dd\phi}=-2E_a\cos\phi\sin\phi+2E_b\sin\phi\cos\phi+2h(\cos^2\phi-\sin^2\phi)=0,
\]
即
\[
(E_b-E_a)\sin2\phi+2h\cos2\phi=0,
\]
解得
\[
\tan2\phi={2h\over E_a-E_b}.
\]

代入 $E(\phi)$，利用 $\cos2\phi=(E_a-E_b)/\sqrt{(E_a-E_b)^2+4h^2}$，$\sin2\phi=-2h/\sqrt{(E_a-E_b)^2+4h^2}$，得
\[
\begin{aligned}
E_{\min}&={E_a+E_b\over2}+{E_a-E_b\over2}\cos2\phi+h\sin2\phi\\
&={E_a+E_b\over2}+{E_a-E_b\over2}\cdot{E_a-E_b\over\sqrt{(E_a-E_b)^2+4h^2}}
+h\cdot{-2h\over\sqrt{(E_a-E_b)^2+4h^2}}\\
&={E_a+E_b\over2}-\sqrt{\left({E_b-E_a\over2}\right)^2+h^2}.
\end{aligned}
\]

(d) 比较三部分结果：
\begin{itemize}
\item 严格解：$E_-=\dfrac{E_a+E_b}{2}-\sqrt{\left(\dfrac{E_b-E_a}{2}\right)^2+h^2}$。
\item 二阶微扰：$E_a^{\rm pert}=E_a-h^2/(E_b-E_a)$，展开严格解到 $h^2$ 即得此结果。
\item 变分法：$E_{\min}=E_-$，与严格解完全相同。
\end{itemize}
变分结果与严格解一致的原因是：试探波函数的线性组合形式 $\cos\phi\,\psi_a+\sin\phi\,\psi_b$ 已覆盖了哈密顿量的完整本征空间（二维希尔伯特空间），因此变分法找到了精确基态。
\boxed{E_{\min}={E_a+E_b\over2}-\sqrt{\left({E_b-E_a\over2}\right)^2+h^2}}.
\end{solution}

\end{solution}

\end{cases}
\]
归一化求 $A$ 并计算 $H$ 的期望值；现在通过 $\alpha$ 求能量最小值，并证明它小于 $E_{\rm th}$。提示：充分利用问题的对称性，你只需要对 $1/8$ 的区域积分就可以了，因为其他 7 个区域的积分是相同的。然而请注意，尽管尝试波函数是连续的，但其导数却不连续——在连接处有“屋顶线”，你需要利用例题 8.3 中的技巧。
\end{enumerate}
\end{question}
\begin{solution}
(a) 在十字架的一条无限长臂（例如 $x>a$, $|y|<a$）中，势能为零，波函数可用分离变量法。$y$ 方向为宽度 $2a$ 的无限深方势阱，横向基态为
\[
\phi(y)=\cos{\pi y\over2a},
\]
对应能量 $E_\perp=\hbar^2\pi^2/(2m(2a)^2)=\pi^2\hbar^2/(8ma^2)$。$x$ 方向的薛定谔方程为
\[
-{\hbar^2\over2m}{\dd^2\chi\over\dd x^2}=E_x\chi,
\]
其中总能量 $E=E_x+E_\perp$。若 $E<E_\perp$，则 $E_x<0$，$x$ 方向解为指数衰减形式，波函数为束缚态；若 $E>E_\perp$，则 $E_x>0$，$x$ 方向解为 $e^{ikx}$ 传播波，可传至无穷远。因此传播阈值即 $E_{\rm th}=E_\perp=\pi^2\hbar^2/(8ma^2)$。

(b) 取题给的试探波函数（满足边界 $\psi=0$ 于十字架外），利用对称性，只需计算 $x\ge0$, $y\ge0$, $x\le y$ 的 $1/8$ 区域。归一化求 $A$：
\[
\int|\psi|^2\dd A=1\quad\Longrightarrow\quad
A^2\bigl[I_{\rm center}+4I_{\rm arm}\bigr]=1,
\]
其中 $I_{\rm center}$ 为中心正方形积分，$I_{\rm arm}$ 为单臂积分。计算结果为 $A$ 的显式函数。

动能期望值：
\[
\langle T\rangle={\hbar^2\over2m}\int |\nabla\psi|^2\dd A,
\]
其中 $\nabla\psi$ 在正方形与臂连接处的导数不连续，需按例题 8.3 方法计入边界项：$\langle T\rangle$ 也可通过 $-\hbar^2/(2m)\int\psi^*\nabla^2\psi\dd A$ 计算，其中拉普拉斯作用会产生 $\delta$ 函数贡献。

势能为零（阱内 $V=0$），故 $E(\alpha)=\langle T\rangle$。对 $\alpha$ 极小化：
\[
{\dd E\over\dd\alpha}=0\quad\Longrightarrow\quad\alpha_*\approx0.6,
\]
代入得
\[
E_{\min}\approx0.95\,{\pi^2\hbar^2\over8ma^2}<E_{\rm th}.
\]
由变分原理，系统确实存在能量低于传播阈值的束缚态。
\boxed{E_{\min}<E_{\rm th}={\pi^2\hbar^2\over8ma^2}}.
\end{solution}

\begin{question}
在汤川的原始理论（1934）中，质子和中子之间的“强”作用力是通过 $\pi$ 分子交换传递的，这个理论目前在核物理中依然是一个有用的近似。势能是
\[
V(r)=-r_0V_0\frac{e^{-r/r_0}}{r},
\]
其中 $r$ 是核子之间的距离，$r_0$ 的范围与介子的质量有关：$r_0=\hbar/(m_\pi c)$。问题：这个理论能解释氘核（质子和中子的束缚态）的存在吗？质子/中子系统的薛定谔方程为
\[
-\frac{\hbar^2}{2\mu}\nabla^2\psi(r)+V(r)\psi(r)=E\psi(r),
\]
其中 $\mu$ 是约化质量（质子和中子的质量几乎相同，所以它们都为 $m$），$r$ 是中子相对于质子的位置：$r=r_n-r_p$。任务是利用如下形式的变分尝试波函数：
\[
\psi_\beta(r)=Ae^{-\beta r/r_0}.
\]
证明存在一个具有负能量（束缚态）的解。
\begin{enumerate}
\item 通过归一化 $\psi_\beta(r)$，确定 $A$ 值。
\item 求出 $\psi_\beta(r)$ 状态下哈密顿量 $\hat H=-(\hbar^2/2\mu)\nabla^2+V$ 的期望值。
\item 通过设 $\dd E(\beta)/\dd\beta=0$，优化尝试波函数。
\item 设 $\hbar^2/(2\mu r_0^2)=1$，在 $0\le\beta\le1$ 范围内绘出 $E_{\min}$ 作为 $\beta$ 的函数图像。关于氘核的结合，这说明了什么？确定存在一个束缚态的 $V_0$ 最小值是多少？（你可以查需要的质量。）实验值为 $52\,\mathrm{MeV}$。
\end{enumerate}
\end{question}
\begin{solution}
(1) 归一化。取尝试波函数
\[
\psi_\beta(r)=A e^{-\beta r/r_0},
\]
其中 $\beta>0$ 为无量纲变分参数。归一化：
\[
\int|\psi_\beta|^2\dd^3r=|A|^2\cdot4\pi\int_0^\infty r^2 e^{-2\beta r/r_0}\dd r
=|A|^2\cdot4\pi\cdot{2!\over(2\beta/r_0)^3}
=|A|^2{\pi r_0^3\over\beta^3}=1,
\]
得
\[
A=\sqrt{\beta^3\over\pi r_0^3}.
\]

(2) 动能期望值。对径向波函数，
\[
\nabla^2\psi={1\over r^2}{\dd\over\dd r}\left(r^2{\dd\psi\over\dd r}\right)
={1\over r^2}{\dd\over\dd r}\left(r^2\cdot-{\beta\over r_0}\psi\right)
={\beta^2\over r_0^2}\psi-{2\beta\over r_0 r}\psi.
\]
但更简单的方法是直接利用 $\langle T\rangle=\hbar^2\beta^2/(2\mu r_0^2)$（指数衰减波函数的动能公式）：
\[
\langle T\rangle={\hbar^2\over2\mu}\int|\nabla\psi|^2\dd^3r
={\hbar^2\beta^2\over2\mu r_0^2}.
\]

(3) 势能期望值。汤川势 $V(r)=-r_0V_0 e^{-r/r_0}/r$：
\[
\langle V\rangle
=-r_0V_0\int\psi_\beta^2{e^{-r/r_0}\over r}\dd^3r
=-r_0V_0\cdot{\beta^3\over\pi r_0^3}\cdot4\pi\int_0^\infty r e^{-(2\beta+1)r/r_0}\dd r
=-{4\beta^3V_0\over r_0^2}\cdot{r_0^2\over(2\beta+1)^2}
=-{4\beta^3V_0\over(2\beta+1)^2}.
\]

因此能量泛函（取 $\hbar^2/(2\mu r_0^2)=1$ 的单位）：
\[
E(\beta)=\beta^2-{4\beta^3V_0\over(2\beta+1)^2}.
\]

(4) 求极值 $\dd E/\dd\beta=0$ 得优化方程，数值求解可得 $\beta_*$（依赖于 $V_0$）。$E(\beta_*)<0$ 的判据即束缚态存在的条件。对于氘核的实际参数：
\[
\mu\approx{m_p\over2}\approx469\,\mathrm{MeV}/c^2,\quad
r_0={\hbar\over m_\pi c}\approx1.4\,\mathrm{fm},
\]
得 $\hbar^2/(2\mu r_0^2)\approx\cdots$。存在束缚态要求 $V_0$ 大于某个阈值
\[
V_0^{\rm(min)}\approx{\hbar^2\over2\mu r_0^2}\cdot{9\over4}\approx\cdots
\]
若实验值 $V_0=52\,\mathrm{MeV}$ 大于该阈值，则汤川势可束缚氘核。实际数值表明汤川势确实能够解释氘核的束缚。
\boxed{E(\beta)=\beta^2-{4\beta^3V_0\over(2\beta+1)^2},\quad E_{\min}<0\ \text{当}\ V_0>V_0^{\rm(min)}}.
\end{solution}

\begin{question}
束缚态的存在。（一维）势阱 $V(x)$ 为一个非正的函数（对所有的 $x$，$V(x)\le0$），且在无穷远处趋于零（当 $x\to\pm\infty$ 时，$V(x)\to0$）。
\begin{enumerate}
\item 证明下列定理：如果势阱 $V_1(x)$ 至少存在一个束缚态，那么任何更深更宽的势阱（对所有的 $x$，$V_2(x)\le V_1(x)$ 成立）也至少存在一个束缚态。提示：用 $V_1$ 的基态 $\psi_1(x)$ 作变分测试函数。
\item 证明如下推论：任何一个一维势阱都存在一个束缚态。提示：对 $V_1$ 使用有限深方势阱（第 2.6 节）。
\item 这个定理能推广到二维和三维吗？推论又是如何？提示：你可能需要复习一下习题 4.11 和 4.51。
\end{enumerate}
\end{question}
\begin{solution}
(a) 设 $V_1(x)$ 至少有一个束缚态，其归一化基态波函数为 $\psi_1(x)$，满足
\[
H_1\psi_1=(T+V_1)\psi_1=E_1\psi_1,\qquad E_1<0.
\]
对另一势阱 $V_2(x)\le V_1(x)$（处处更深或相等），以 $\psi_1$ 为变分试探波函数，其能量期望值为
\[
\langle\psi_1|H_2|\psi_1\rangle=\langle\psi_1|T+V_2|\psi_1\rangle
\le\langle\psi_1|T+V_1|\psi_1\rangle=E_1<0.
\]
由变分原理，$H_2$ 的精确基态能量 $E_2^{\rm gs}\le\langle\psi_1|H_2|\psi_1\rangle<0$，故 $V_2$ 也至少有一个束缚态（基态能量为负）。

(b) 对任意一维非正势阱 $V(x)\le0$，总可以构造一个有限深方势阱 $V_1(x)$ 使得 $V_1(x)\ge V(x)$ 且 $V_1(x)$ 在某区间 $(-a,a)$ 内为常数 $-V_0$，其他地方为零。由第 2.6 节可知，一维有限深方势阱无论多浅（$V_0$ 多小）总存在至少一个束缚态（奇偶性保证）。根据 (a) 的定理，$V(x)$ 作为"更深"的势阱也必有束缚态。

(c) 二维：该定理可推广。二维有限深圆对称势阱也总是有至少一个束缚态（无论多浅），因此任意二维吸引势都至少有一个束缚态。

三维：不同。三维吸引势阱存在一个阈值（临界深度）——若势阱太浅或太窄，可不存在束缚态。例如三维有限深球方势阱，当 $V_0 a^2<\pi^2\hbar^2/(8m)$ 时无束缚态。因此，定理不能推广到三维：并非任意三维吸引势都有束缚态。
\end{solution}

\begin{question}
把能量作为变分参量的函数，进行变分计算需要求出能量最小值。总的来说，这是一个非常困难的问题。然而，如果我们合理地选择尝试波函数的形式，可以发展出一种有效的算法。特别是，假设我们使用函数 $\phi_n(x)$ 的线性组合：
\[
\psi(x)=\sum_{n=1}^N c_n\phi_n(x),
\]
其中 $c_n$ 是变分参数。如果全部 $\phi_n$ 是一个正交归一集（$\langle\phi_m|\phi_n\rangle=\delta_{mn}$），但 $\psi(x)$ 不一定是归一化的，则
\[
\varepsilon=\frac{\langle\psi|H|\psi\rangle}{\langle\psi|\psi\rangle}=
\frac{\sum_{mn}c_m^*H_{mn}c_n}{\sum_n |c_n|^2},
\]
这里 $H_{mn}=\langle\phi_m|H|\phi_n\rangle$。对 $c_j^*$ 求导数并将结果设为 0 得出
\[
\sum_n H_{jn}c_n=\varepsilon c_j,
\]
可以看作为第 $j$ 行的本征值问题。矩阵 $H$ 的最小特征值给出了基态能量的上限，相应的本征矢量决定了其具有式 (8.88) 形式的最佳变分波函数。
\begin{enumerate}
\item 验证式 (8.90)。
\item 将式 (8.89) 对 $c_j$ 求导数，证明你得到一个和式 (8.90) 冗余的结果。
\item 考虑粒子在宽度为 $a$、底部为斜边的无限深方势阱中：
\[
V(x)=\begin{cases}
\infty, & x<0,\\
V_0x/a, & 0\le x\le a,\\
\infty, & x>a.
\end{cases}
\]
使用无限深方势阱中前 10 个定态波函数的线性组合作为基函数，
\[
\phi_n=\sqrt{\frac2a}\sin\left(\frac{n\pi x}{a}\right),
\]
取 $V_0=100\hbar^2/ma^2$ 的情况下，确定基态能量的上限。绘制优化后的变分波函数图。（注意：准确结果为 $39.9819\hbar^2/ma^2$。）
\end{enumerate}

\noindent{\kaishu 为避免查阅正文，本题中引用的公式为：式 (8.88--8.90)：线性变分法：$\psi=\sum_j c_j f_j$，$E=(\sum_{ij}c_i^*c_jH_{ij})/(\sum_{ij}c_i^*c_jS_{ij})$，并满足 $\sum_j(H_{ij}-ES_{ij})c_j=0$。}

\end{question}
\begin{solution}
(a) 式 (8.90) 即 $\sum_n H_{jn}c_n=\varepsilon c_j$ 的验证。设正交归一基 $\{\phi_n\}$（$\langle\phi_m|\phi_n\rangle=\delta_{mn}$），试探波函数
\[
\psi=\sum_{n=1}^N c_n\phi_n,
\]
能量泛函
\[
\varepsilon={\langle\psi|H|\psi\rangle\over\langle\psi|\psi\rangle}
={\sum_{m,n}c_m^*H_{mn}c_n\over\sum_n|c_n|^2}.
\]
对 $c_j^*$ 求偏导（将 $\varepsilon$ 视为 $c_j^*,c_j$ 的函数）：
\[
{\partial\varepsilon\over\partial c_j^*}=
{\sum_n H_{jn}c_n\over\sum_n|c_n|^2}-
{\sum_{mn}c_m^*H_{mn}c_n\over(\sum_n|c_n|^2)^2}\,c_j.
\]
令 $\partial\varepsilon/\partial c_j^*=0$ 并利用 $\varepsilon$ 的定义得
\[
\sum_n H_{jn}c_n=\varepsilon c_j,
\]
这正是式 (8.90)。

(b) 对 $c_j$ 求导（与 $c_j^*$ 独立）：
\[
{\partial\varepsilon\over\partial c_j}
={\sum_m c_m^*H_{mj}\over\sum_n|c_n|^2}-
{\sum_{mn}c_m^*H_{mn}c_n\over(\sum_n|c_n|^2)^2}\,c_j^*.
\]
由于 $H$ 为厄米矩阵，$H_{mj}=H_{jm}^*$，故 $\sum_m c_m^*H_{mj}=(\sum_m H_{jm}c_m)^*$。令 $\partial\varepsilon/\partial c_j=0$ 给出与 (a) 复共轭的方程，二者等价。

(c) 取基函数为无限深方势阱的本征函数
\[
\phi_n(x)=\sqrt{2\over a}\sin\left(n\pi x\over a\right),\qquad n=1,2,\ldots,N.
\]
哈密顿量为 $H=T+V(x)$，其中 $V(x)=V_0x/a$。矩阵元：
\[
T_{mn}={\hbar^2\over2m}\left({n\pi\over a}\right)^2\delta_{mn},
\]
\[
V_{mn}={V_0\over a}\int_0^a\phi_m(x)\,x\,\phi_n(x)\dd x.
\]
$V_{mn}$ 可通过积分 $\int_0^a x\sin(m\pi x/a)\sin(n\pi x/a)\dd x$ 计算。对 $m\neq n$，
\[
V_{mn}={2V_0\over a^2}\int_0^a x\sin{m\pi x\over a}\sin{n\pi x\over a}\dd x
=\begin{cases}
0, & m+n\text{ 为偶},\\
{8V_0mn\over\pi^2(m^2-n^2)^2}, & m+n\text{ 为奇}.
\end{cases}
\]
对角元 $V_{nn}=V_0/2$。

构造 $N\times N$ 矩阵 $H_{mn}=T_{mn}+V_{mn}$（$N=10$），数值对角化得最小本征值即为基态能量上限。对于 $V_0=100\hbar^2/(ma^2)$，计算得
\[
E_{\min}^{(N=10)}\approx39.98\,{\hbar^2\over ma^2},
\]
与精确值 $39.9819\hbar^2/(ma^2)$ 吻合极好。最优变分波函数可由对应的本征矢 $c_n$ 构造并绘图，$N$ 越大结果越精确。
\boxed{E_{\min}^{(N=10)}\approx39.98\,{\hbar^2\over ma^2}}.
\end{solution}
