% ── \heading{…} at top of each exam file ──────────────────
\newcommand{\heading}[1]{%
  \setcounter{question}{0}%
  \def\HeadingCourse{#1}%
  \def\HeadingName{#1}%
  \IfSubStr{#1}{-}{%
    \StrBefore{#1}{-}[\HeadingCourse]%
    \StrBehind{#1}{-}[\HeadingName]%
  }{}%
  \refstepcounter{subsection}%
  \phantomsection
  \addcontentsline{toc}{subsection}{#1}%
  \markboth{\HeadingCourse}{\HeadingName}%
  \examtitle{\HeadingCourse}{\HeadingName}%
}
